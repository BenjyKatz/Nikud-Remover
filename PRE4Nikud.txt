(א) מַעֲשֶׂה בְּרַבִּי אֱלִיעֶזֶר בֶּן הוּרְקָנוּס, שֶׁהָיוּ לְאָבִיו חוֹרְשִׁים,(ב) וְהָיוּ חוֹרְשִׁין עַל גַּבֵּי הַמַּעֲנָה, וְהוּא הָיָה חוֹרֵשׁ בִּטְרָשִׁין,(ג) יָשַׁב לוֹ וְהָיָה בּוֹכֶה. אָמַר לוֹ אָבִיו מִפְּנֵי מָה אַתָּה בּוֹכֶה,(ד) שֶׁמָּא מִצְטַעֵר אַתָּה שֶׁאַתָּה חוֹרֵשׁ בִּטְרָשִׁין, עַכְשָׁו אַתָּה חוֹרֵשׁ עַל גַּבֵּי הַמַּעֲנָה. יָשַׁב לוֹ עַל גַּבֵּי הַמַּעֲנָה וְהָיָה בּוֹכֶה. אָמַר לוֹ מִפְּנֵי מָה אַתָּה בּוֹכֶה,(ה) שֶׁמָּא מִצְטַעֵר אַתָּה שֶׁאַתָּה חוֹרֵשׁ עַל גַּבֵּי הַמַּעֲנָה. אָמַר לוֹ לָאו, וְלָמָּה אַתָּה בּוֹכֶה, אָמַר לוֹ שֶׁאֲנִי מְבַקֵּשׁ לִלְמֹד תּוֹרָה, אָמַר לוֹ(ו) וַהֲלֹא בֶּן עֶשְׂרִים וּשְׁמוֹנֶה שָׁנִים אָתָּה(ז) וְאַתָּה מְבַקֵּשׁ לִלְמֹד תּוֹרָה,(ח) אֶלָּא קַח לְךָ אִשָּׁה וְתוֹלִיד לְךָ בָּנִים(ט) וְאַתָּה מוֹלִיכָן לְבֵית הַסֵּפֶר.(י) עָשָׂה שְׁתֵּי שַׁבָּתוֹת וְלֹא טָעַם כְּלוּם,(יא) עַד שֶׁנִּגְלָה לוֹ אֵלִיָּהוּ זָכוּר לַטוֹב(יב) וְאָמַר לוֹ, בֶּן הוּרְקָנוּס(יג) לָמָּה אַתָּה בּוֹכֶה, אָמַר לוֹ מִפְּנֵי שֶׁאֲנִי מְבַקֵּשׁ לִלְמֹד תּוֹרָה, אָמַר לוֹ אִם אַתָּה מְבַקֵּשׁ לִלְמֹד תּוֹרָה, עֲלֵה לִירוּשָׁלַיִם אֵצֶל רַבָּן יוֹחָנָן בֶּן זַכַּאי. עָמַד וְהָלַךְ אֵצֶל רַבָּן יוֹחָנָן בֶּן זַכַּאי, יָשַׁב לוֹ וְהָיָה בּוֹכֶה, אָמַר לוֹ מִפְּנֵי מָה אַתָּה בּוֹכֶה, אָמַר לוֹ מִפְּנֵי שֶׁאֲנִי מְבַקֵּשׁ לִלְמֹד תּוֹרָה.(יד) אָמַר לוֹ בֶּן מִי אָתָּה, וְלֹא הִגִּיד לוֹ,(טו) אָמַר לוֹ מִיָּמֶיךָ לֹא לָמַדְתָּ קְרִיאַת שְׁמַע וְלֹא תְפִלָּה וְלֹא בִּרְכַּת הַמָּזוֹן,(טז) אָמַר לוֹ לָאו, אָמַר לוֹ(יז) עֲמֹד וַאֲלַמֶּדְךָ שְׁלָשְׁתָּן.(יח) יָשַׁב וְהָיָה בּוֹכֶה, אָמַר לוֹ בְּנִי מִפְּנֵי מָה אַתָּה בּוֹכֶה, אָמַר לוֹ שֶׁאֲנִי מְבַקֵּשׁ לִלְמֹד תּוֹרָה.(יט) וְהָיָה אוֹמֵר לוֹ שְׁתֵּי הֲלָכוֹת(כ) כָּל יְמֵי הַשָּׁבוּעַ וְהָיָה חוֹזֵר לוֹ עֲלֵיהֶן(כא) וּמְדַבְּקָן.(כב) עָשָׂה שְׁמוֹנָה יָמִים(כג) וְלֹא טָעַם כְּלוּם(כד) עַד שֶׁעָלָה רֵיחַ פִּיו לִפְנֵי רַבָּן יוֹחָנָן בֶּן זַכַּאי,(כה) וְהֶעֱמִידוֹ מִלְּפָנָיו. יָשַׁב וְהָיָה בּוֹכֶה, אָמַר לוֹ מִפְּנֵי מָה אַתָּה בּוֹכֶה, אָמַר לוֹ מִפְּנֵי שֶׁהֶעֱמַדְתַּנִי מִפָּנֶיךָ כְּאָדָם(כו) שֶׁמַעֲמִיד מִלְּפָנָיו מֻכֵּה שְׁחִין. אָמַר לוֹ בְּנִי(כז) כְּשֵׁם שֶׁעָלָה רֵיחַ פִּיךָ מִלְּפָנַי,(כח) כָּךְ יַעֲלֶה רֵיחַ חֻקֵּי תוֹרָה מִפִּיךָ לַשָּׁמַיִם. אָמַר לוֹ בְּנִי, בֶּן מִי אָתָּה, אָמַר לוֹ, בֶּן הוּרְקָנוֹס אָנִי, אָמַר לוֹ וַהֲלֹא(כט) בֶּן גְּדוֹלֵי עוֹלָם אַתָּה, וְלֹא הָיִיתָ(ל) מַגִּיד לִי. אָמַר לוֹ חַיֶּיךָ, הַיּוֹם אַתָּה סוֹעֵד אֶצְלִי,(לא) אָמַר לוֹ כְּבָר סָעַדְתִּי אֵצֶל אַכְסַנְיָא שֶׁלִי, אָמַר לוֹ וּמִי הוּא אַכְסַנְיָא שֶׁלָּךְ, אָמַר לוֹ, רַבִּי יְהוֹשֻׁעַ בֶּן חֲנַנְיָה וְרַבִּי יוֹסֵי הַכֹּהֵן. שָׁלַח וְשָׁאַל אַכְסַנְיָא שֶׁלּוֹ אָמַר לָהֶם אֶצְלְכֶם סָעַד אֱלִיעֶזֶר הַיּוֹם, אָמְרוּ לוֹ לָאו, וַהֲלֹא יֶשׁ לוֹ שְׁמוֹנָה יָמִים שֶׁלֹּא טָעַם כְּלוּם.(לב) אַחֲרֵי כֵן הָלְכוּ רַבִּי יְהוֹשֻׁעַ בֶּן חֲנַנְיָה וְרַבִּי יוֹסֵי הַכֹּהֵן וְאָמְרוּ לְרַבָּן יוֹחָנָן בֶּן זַכַּאי, הֲרֵי יֶשׁ לוֹ שְׁמוֹנָה יָמִים שֶׁלֹא טָעַם כְּלוּם.

בַּשְּׁלִישִׁי(א) הָיְתָה הָאָרֶץ(ב) מִישׁוֹר כַּבִּקְעָה,(ג) וְהָיוּ הַמַּיִם מְכַסִּים עַל פְּנֵי כָל הָאָרֶץ. וּכְשֶׁיָצָא הַדִּבּוּר(ד) מִפִּי הַגְבוּרָה יִקָּווּ הַמַּיִם [בראשית א, ט],(ה) עָלוּ מִקְּצוֹת הָאָרֶץ הֶהָרִים וְהַגְּבָעוֹת(ו) וְנִתְפַּזְּרוּ עַל פְּנֵי כָל הָאָרֶץ,(ז) וְנַעֲשִׂים עֲמָקִים עַל תּוֹכָה שֶׁל אֶרֶץ, וְנִתְגַּלְגְּלוּ הַמַּיִם וְנִקְווּ לַעֲמָקִים, שֶׁנֶּאֱמַר [שם י] וּלְמִקְוֵה הַמַּיִם קָרָא יַמִּים.(ח) מִיָּד נִתְגָּאוּ הַמַּיִם וְעָלוּ לְכַסּוֹת הָאָרֶץ כְּבַתְּחִלָּה, עַד שֶׁגָעַר בָּהֶם הַקָּדוֹשׁ בָּרוּךְ הוּא(ט) וּכְבָשָׁן וּנְתָנָן תַּחַת כַּפּוֹת רַגְלָיו(י) וּמְדָדָן בְּשָׁעֳלוֹ,(יא) שֶׁלֹּא לְהוֹסִיף וְלֹא לִגְרֹעַ.(יב) וְעָשָׂה חוֹל גְּבוּל לַיָּם,(יג) כְּאָדָם שֶׁהוּא עוֹשֶׂה גָדֵר לְכַרְמוֹ.(יד) וּכְשֶׁהֵן עוֹלִין וְרוֹאִין אֶת הַחוֹל לִפְנֵיהֶם חוֹזְרִין לַאֲחוֹרֵיהֶם, שֶׁנֶּאֱמַר [ירמיה ה, כב] הַאוֹתִי לֹא תִירָאוּ נְאֻם ה' אִם מִפָּנַי לֹא תָחִילוּ אֲשֶׁר שַׂמְתִּי חוֹל גְּבוּל לַיָּם:(טו) וְעַד שֶׁלֹּא נִקְווּ הַמַּיִם נִבְרְאוּ הַמְּאוֹרוֹת וּתְהוֹמוֹת.(טז) וְאֵלוּ הֵן הַתְּהוֹמוֹת, אֵלּוּ שֶׁמִּתַּחַת לָאָרֶץ,(יז) שֶׁהָאָרֶץ עַל הַתְּהוֹמוֹת נִרְקַעַת, כָּאֳנִיָּה(יח) שֶׁהִיא צָפָה בְּלֶב יָם, כָּךְ הָאָרֶץ מְרֻקַּעַת עַל הַמַּיִם, שֶׁנֶּאֱמַר [תהלים קלו, ו] לְרוֹקַע הָאָרֶץ עַל הַמָּיִם.(יט) וּפֶתַח לְגַן עֵדֶן פָּתַח בָּהּ,(כ) וְהוֹצִיא מִמֶּנָּה נְטָעִים עַל פְּנֵי כָל הָאָרֶץ, כָּל מִינֵי עֵץ אִילָן עוֹשֶׂה פְּרִי לְמִינוֹ,(כא) וְכָל מִין עֵשֶׂב וָדֶשֶׁא נָטַע מֵהֶם,(כב) וּבָהֶם זְרָעָם עַל הָאָרֶץ, שֶׁנֶּאֱמַר [בראשית א, יא] אֲשֶׁר זַרְעוֹ בוֹ עַל הָאָרֶץ וְגוֹ'.(כג) וְעָרַךְ שֻׁלְחָן לַבְּרִיּוֹת עַד שֶׁלֹּא נִבְרָא הָעוֹלָם,(כד) שֶׁנֶּאֱמַר [תהלים כג, ה] תַּעֲרֹךְ לְפָנַי שֻׁלְחָן.(כה) וְכָל הַמַּעְיָנוֹת עוֹלִין מִן הַתְּהוֹמוֹת לְהַשְׁקוֹת כָּל הַבְּרִיּוֹת: רַבִּי יְהוֹשֻׁע אוֹמֵר(כו) עָבְיָהּ שֶׁל אֶרֶץ מַהֲלַךְ שִׁשִּׁים שָׁנָה. וּתְהוֹם אֶחָד שֶׁהוּא(כז) אֵצֶל גֵּיהִנָּם נוֹבֵעַ וּמוֹצִיא מַיִם חַמִּים(כח) תַּעֲנוּג לִבְנֵי אָדָם: אָמַר רַבִּי יְהוּדָה,(כט) פַּעַם אַחַת בְּכָל חֹדֶשׁ וָחֹדֶשׁ(ל) סִלוֹנוֹת עוֹלִין מִן הַתְּהוֹמוֹת וּמַשְׁקִין אֶת כָּל פְּנֵי הָאֲדָמָה, שֶׁנֶּאֱמַר [בראשית ב, ו] וְאֵד יַעֲלֶה מִן הָאָרֶץ וְהִשְׁקָה אֶת כָּל פְּנֵי הָאֲדָמָה.(לא) הֶעָבִים מַשְׁמִיעִין אֶת קוֹל צִנּוֹרוֹתֵיהֶן לַיַּמִּים, וְהַיַּמִּים מַשְׁמִיעִים אֶת קוֹל צִנּוֹרוֹתֵיהֶם לַתְּהוֹמוֹת,(לב) וּתְהוֹם קוֹרֵא אֶל תְּהוֹם לְהַעֲלוֹת מַיִם וְלִתֵּן לֶעָבִים, שֶׁנֶּאֱמַר [תהלים מב, ח] תְּהוֹם אֶל תְּהוֹם קוֹרֵא לְקוֹל צִנּוֹרֶיךָ.(לג) וְהֶעָבִים שׁוֹאֲבִין מַיִם מִן הַתְּהוֹמוֹת, שֶׁנֶּאֱמַר [שם קלה, ז] מַעֲלֶה נְשִׂאִים מִקְצֵה הָאָרֶץ.(לד) וּבְמָקוֹם שֶׁיִפְקֹד לָהֶם הַקָּדוֹשׁ בָּרוּךְ הוּא לְהַעֲלוֹת וּלְהַמְטִיר שָׁם מַגְשִׁימִים.(לה) מִיָּד הָאָרֶץ מִתְעַבֶּרֶת וּצְמֵחָה,(לו) כְּאִשָּׁה אַלְמָנָה שֶׁהִיא מְעֻבֶּרֶת מִזְּנוּת. אֲבָל כְּשֶׁיִּרְצֶה הַקָּדוֹשׁ בָּרוּךְ הוּא(לז) לְבָרֵךְ צִמְחָהּ שֶׁל אֶרֶץ(לח) וְלִתֵּן צֵידָן שֶׁל בְּרִיּוֹת, פּוֹתֵחַ אוֹצְרוֹת הַטּוֹב שֶׁבַּשָּׁמַיִם וּמַמְטִיר עַל הָאָרֶץ,(לט) שֶׁהֵן מַיִם זְכָרִים, וּמִיָּד הָאָרֶץ(מ) מִתְעַבֶּרֶת כַּכַּלָּה שֶׁהִיא מִתְעַבֶּרֶת מִבַּעְלָהּ הָרִאשׁוֹן.(מא) וּצְמֵחָה זֶרַע שֶׁל בְּרָכָה, שֶׁנֶאֱמַר [דברים כח, יב] יִפְתַּח ה' לְךָ אֶת אוֹצָרוֹ הַטּוֹב(מב) אֶת הַשָּׁמַיִם.(מג) וּכְתִיב [ישעיה סב, ה] כִּי יִבְעַל בָּחוּר בְּתוּלָה יִבְעָלוּךְ בָּנָיִךְ, וּכְתִיב [שם נה, י] כִּי כַּאֲשֶׁר יֵרֵד הַגֶּשֶׁם וְהַשֶּׁלֶג(מד) מִן הַשָּׁמַיִם וְגוֹ'. בַּחֲמִישִׁי הִשְׁרִיץ מִן הַמַּיִם(א) כָּל מִין עוֹף,(ב) זְכָרִים וּנְקֵבוֹת טְהוֹרִים וּטְמֵאִים.(ג) בִּשְׁנֵי סִימָנִים הֵן מִטַּהֲרִין, בְּזֶפֶק וּבְקֻרְקְבָן נִקְלָף.(ד) רַבִּי אֱלִיעֶזֶר אוֹמֵר בְּאֶצְבַּע יְתֵרָה. וּשְׁנֵי מִינֵי עוֹפוֹת(ה) נִבְחֲרוּ לְקָרְבַּן וְעוֹלָה, אֵלּוּ הֵן, תּוֹרִים וּבְנֵי יוֹנָה. בַּחֲמִישִׁי הִשְׁרִיץ מִן הַמַּיִם כָּל מִין דָּגִים, זְכָרִים וּנְקֵבוֹת, טְמֵאִים וּטְהוֹרִים. בִּשְׁנֵי סִימָנִים הֵם מִטַּהֲרִים,(ו) בִּסְנַפִּיר וְקַשְׂקֶשֶׂת. וְאִם לָאו הֵם טְמֵאִים. בַּחֲמִישִׁי הִשְׁרִיץ מִן הַמַּיִם כָּל מִינֵי חֲגָבִים, זְכָרִים וּנְקֵבוֹת, טְמֵאִים וּטְהוֹרִים.(ז) בִּשְׁנֵי סִימָנִים הֵן מִטַּהֲרִים, בִּכְרָעַיִם אֲרֻכִּים, שֶׁבָּהֵן מְקַפְּצִין עַל הָאָרֶץ,(ח) וּבִכְנָפַיִם מְכַסִּין אֶת כָּל הַגּוּף. וְאֵלּוּ שֶׁשָׁרְצוּ מִן הַמַּיִם(ט) דָּגִים וַחֲגָבִים נֶאֱכָלִין שֶׁלֹּא בִשְׁחִיטָה,(י) אֲבָל הָעוֹף אֵינוֹ נֶאֱכָל אֶלָּא בִשְׁחִיטָה.(יא) אֵלּוּ שֶׁנִּבְרְאוּ מִן הַמַּיִם דָּמָן נִשְׁפָּךְ כַּמַּיִם, וְאֵלּוּ שֶׁנִּבְרְאוּ מִן הָאָרֶץ דָּמָן לִכָּסוֹת בֶּעָפָר:(יב) רַבִּי אֱלִיעֶזֶר אוֹמֵר, לֹא עַל הַמַּיִם בִּלְבָד אָמַר שֶׁיִשְׁרְצוּ, אֶלָּא אַף עַל אֻמּוֹת הָעוֹלָם(יג) שֶׁנִּמְשְׁלוּ כַמַּיִם, שֶׁנֶּאֱמַר [ישעיה יז, יב] וּשְׁאוֹן לְאֻמִּים כִּשְׁאוֹן מַיִם כַּבִּירִים יִשָּׁאוּן. וּכְשֵׁם שֶׁשָׁרְצוּ הַמַּיִם בַּיּוֹם הַהוּא, כָּךְ עֲתִידִים אֻמּוֹת הָעוֹלָם לִשְׁרֹץ בָּעוֹלָם(יד) וְנִלְחָמִים אֵלּוּ עִם אֵלּוּ לְהַשְׁחִית,(טו) שֶׁנֶּאֱמַר [דברי הימים ב, טו, ו] וְכֻתְּתוּ גוֹי בְגוֹי וְעִיר בְּעִיר כִּי אֱלֹהִים הֲמָמָם בְּכָל צָרָה.(טז) וּכְתִיב אַחֲרָיו תְּשׁוּעַת יִשְׂרָאֵל, שֶׁנֶּאֱמַר [שם ז] וְאַתֶּם חִזְקוּ וְאַל יִרְפּוּ יְדֵיכֶם: כָּל הַנְּהָרוֹת כְּשֵׁם שֶׁהֵם מְהַלְּכִין עַל הָאָרֶץ(יז) הֵם טוֹבִים וּבְרוּכִים וּמְתוּקִים(יח) וְיֵשׁ מֵהֶן הֲנָיָה לָעוֹלָם,(יט) נִכְנְסוּ לַיָּם הֵם מְאוּרִים רָעִים וּמְרוֹרִים וְאֵין מֵהֶם הֲנָיָה לָעוֹלָם,(כ) כָּךְ יִשְׂרָאֵל שֶׁבּוֹטְחִים בְּצֵל יוֹצְרָם,(כא) הֵם בְּרוּכִים וּמְתוּקִים וְיֵשׁ מֵהֶם הֲנָיָה לָעוֹלָם, וּבִזְמַן שֶׁהֵם סָרִים מֵאַחֲרֵי יוֹצְרָם(כב) וּבָטְחוּ בְּחֻקּוֹת הַגּוֹיִם, הֵם נֶאָרִים וְרָעִים וְאֵין מֵהֶם הֲנָיָה לָעוֹלָם.(כג) וּכְשֵׁם שֶׁמֵּימֵי נְהָרוֹת מַאֲכָל לַיָּם, כָּךְ הֵם מַאֲכָל לְאִשּׁוֹ שֶׁל גֵּיהִנָּם.(כד) וְכָל הַגְּשָׁמִים יוֹרְדִים בַּיָּם, זֶרַע הֵם(כה) לַאֲשֶׁר בַּיָּם,(כו) וּמֵהֶם הַדָּגִים מִזְדַּוְּגִים:(כז) בַּחֲמִישִׁי נֶהֶפְכוּ כָּל הַמַּיִם לְדָם בְּמִצְרָיִם.(כח) בַּחֲמִישִׁי יָצְאוּ אֲבוֹתֵינוּ מִמִּצְרָיִם.(כט) בַּחֲמִישִׁי עָמְדוּ מֵימֵי הַיַּרְדֵּן לִפְנֵי אֲרוֹן בְּרִית ה'.(ל) בַּחֲמִישִׁי סָתַם חִזְקִיָּהוּ אֶת הַמַּעְיָנוֹת שֶׁהָיוּ בִירוּשָׁלָיִם, שֶׁנֶּאֱמַר [דברי הימים ב, לב, ל] וְהוּא יְחִזְקִיָּהוּ סָתַם אֶת מוֹצָא מֵימֵי גִיחוֹן הָעֶלְיוֹן. בַּחֲמִישִׁי(לא) הִשְׁרִיץ מִן הַמַּיִם לִוְיָתָן נָחָשׁ בָּרִיחַ,(לב) מְדוֹרוֹ בַּמַּיִם הַתַּחְתּוֹנִים.(לג) וּבֵין שְׁנֵי סְנַפִּירָיו הַבְּרִיחַ הַתִּיכוֹן שֶׁל אֶרֶץ עוֹמֵד.(לד) וְכָל הַתַּנִּינִים הַגְּדוֹלִים שֶׁבַּיָּם(לה) מְזוֹנוֹ שֶׁל לִוְיָתָן.(לו) וְהַקָּדוֹשׁ בָּרוּךְ הוּא שׂוֹחֵק עִמּוֹ יוֹם יוֹם.(לז) וְהוּא פוֹתֵחַ פִּיו, וְהַתַּנִּין הַגָּדוֹל שֶׁבָּא יוֹמוֹ לְהֵאָכֵל, הוּא בּוֹרֵחַ וְנָס וְנִכְנָס לְתוֹךְ פִּיו שֶׁל לִוְיָתָן.(לח) וְהַקָּדוֹשׁ בָּרוּךְ הוּא שׂוֹחֵק בּוֹ,(לט) שֶׁנֶּאֱמַר [תהלים קד, כו] לִוְיָתָן זֶה יָצַרְתָּ לְשַׂחֶק בּוֹ: רַבִּי מֵאִיר אוֹמֵר,(מ) אֵלּוּ שֶׁנִּבְרְאוּ מִן הָאָרֶץ, פָּרִים וְרָבִים בָּאָרֶץ, וְאֵלּוּ(מא) שֶׁשָּׁרְצוּ מִן הַמַּיִם פָּרִים וְרָבִים בַּמַּיִם, חוּץ מִכָּל מִין עוֹף כָּנָף, שֶׁבְּרִיָתוֹ מִן הַמַּיִם וּפָרֶה וְרָבֶה בָּאָרֶץ,(מב) שֶׁנֶּאֱמַר [בראשית א, כב] וְהָעוֹף יִרֶב בָּאָרֶץ.(מג) וְאֵלּוּ שֶׁשָּׁרְצוּ מִן הַמַּיִם, פָּרִים וְרָבִים בְּבֵיצִים, וְאֵלּוּ שֶׁנִּבְרְאוּ מִן הָאָרֶץ, פָּרִים וְרָבִים בְּוָלָד.

(א) הַקִּנְאָה וְהַתַּאֲוָה וְהַכָּבוֹד מוֹצִיאִין אֶת הָאָדָם מִן הָעוֹלָם. אָמְרוּ מַלְאֲכֵי הַשָּׁרֵת לִפְנֵי הַקָּדוֹשׁ בָּרוּךְ הוּא, רִבּוֹן כָּל הָעוֹלָמִים(ב) מָה אָדָם וַתֵּדָעֵהוּ [תהלים קמד, ג], אָדָם לַהֶבֶל דָּמָה [שם ד],(ג) אֵין עַל עָפָר מָשְׁלוֹ [איוב מא, כה]. אָמַר לָהֶם(ד) מַה שֶׁאַתֶּם מְקַלְּסִין אוֹתִי בָּעֶלְיוֹנִים הוּא מְיַחֵד אוֹתִי בַּתַּחְתּוֹנִים. וְלֹא עוֹד(ה) אֶלָּא יְכוֹלִין אַתֶּם לַעֲמֹד וְלִקְרֹא שֵׁמוֹת לְכָל הַבְּרִיּוֹת,(ו) עָמְדוּ וְלֹא יָכֹלוּ. מִיָּד עָמַד אָדָם וְקָרָא שֵׁמוֹת לְכָל הַבְּרִיּוֹת, שֶׁנֶּאֱמַר [בראשית ב, כ] וַיִקְרָא הָאָדָם שֵׁמוֹת לְכָל הַבְּהֵמָה וְגוֹ'. וְכֵיוָן שֶׁרָאוּ מַלְאֲכֵי הַשָּׁרֵת אָמְרוּ(ז) אִם אֵין אָנוּ בָּאִים בְּעֵצָה עַל אָדָם שֶׁיֶּחֱטָא לִפְנֵי בוֹרְאוֹ, אֵין אָנוּ יְכוֹלִין בּוֹ.(ח) וְהָיָה סַמָּאֵל הַשַׂר הַגָּדוֹל שֶׁבַּשָּׁמַיִם,(ט) וְחַיּוֹת וּשְׂרָפִים מִשֵּׁשׁ כְּנָפַיִם, וְסַמָּאֵל(י) מִשְׁתֵּים עֶשְׂרֵה כְּנָפַיִם. לָקַח(יא) אֶת הַכַּת שֶׁלּוֹ וְיָרַד,(יב) וְרָאָה כָּל הַבְּרִיּוֹת שֶׁבָּרָא הַקָּדוֹשׁ בָּרוּךְ הוּא וְלֹא מָצָא(יג) חָכָם לְהָרַע כַּנָּחָשׁ, שֶׁנֶּאֱמַר [שם ג, א] וְהַנָּחָשׁ הָיָה עָרוּם מִכֹּל חַיַּת הַשָדֶה.(יד) וְהָיָה דְמוּתוֹ כְּמִין גָּמָל, וְעָלָה וְרָכַב עָלָיו.(טו) וְהָיְתָה הַתּוֹרָה צוֹוַחַת וְאוֹמֶרֶת, סַמָּאֵל עַכְשָׁו נִבְרָא הָעוֹלָם וְעֵת לִמְרֹד בַּמָּקוֹם,(טז) כָּעֵת בַּמָּרוֹם תַּמְרִיא [איוב לט, יח]. רִבּוֹן הָעוֹלָמִים(יז) תִּשְׂחַק לַסּוּס וּלְרֹכְבוֹ.(יח) מָשָׁל לְמָה הַדָּבָר דּוֹמֶה, לְאָדָם שֶׁיֶּשׁ בּוֹ רוּחַ רָעָה, וְכָל הַמַּעֲשִׂים שֶׁהוּא עוֹשֶׂה(יט) מִדַּעְתּוֹ הוּא עוֹשֶׂה, וְכָל הַדְּבָרִים שֶׁהוּא מְדַבֵּר מִדַּעְתּוֹ הוּא מְדַבֵּר, וַהֲלֹא אֵינוֹ עוֹשֶׂה אֶלָּא מִדַּעַת רוּחַ רָעָה שֶׁיֵּשׁ עָלָיו,(כ) כָּךְ הַנָּחָשׁ כָּל מַעֲשָׂיו שֶׁעָשָׂה וְכָל דְּבָרָיו שֶׁדִּבֵּר לֹא דִבֵּר וְלֹא עָשָׂה אֶלָּא מִדַּעְתּוֹ שֶׁל סַמָּאֵל,(כא) וְעָלָיו הַכָּתוּב אוֹמֵר [משלי יד, לב] בְּרָעָתוֹ יִדָּחֶה רָשָׁע.(כב) מָשָׁל לְמֶלֶךְ שֶׁנָּשָׂא אִשָּׁה, וְהִשְׁלִיטָהּ בְּכָל מַה שֶּׁיֵשׁ לוֹ בָּאֲבָנִים טוֹבוֹת וּמַרְגָּלִיּוֹת, אָמַר לָהּ כָּל מַה שֶּׁיֶשׁ לִי יְהֵא בְּיָדֵךְ חוּץ(כג) מֵחָבִית זוֹ שֶׁהִיא מְלֵאָה עַקְרַבִּים. נִכְנַס אֶצְלָהּ(כד) זָקֵן אֶחָד(כה) וְשָׁאַל מִמֶּנָּה חֹמֶץ, אָמַר לָהּ מָה הַמֶּלֶךְ נוֹהֵג לָךְ, אָמְרָה לוֹ כָּל מַה שֶּׁיֶשׁ לוֹ נָתַן לִי וְהִנִּיחַ בְּיָדִי, חוּץ מֵחָבִית זוֹ שֶׁהִיא מְלֵאָה עַקְרַבִּים, אָמַר לָה וַהֲלֹא(כו) כָּל קֻמִּיָּא שֶׁל מֶלֶךְ הֲרֵי הִיא בֶּחָבִית זוֹ, וְלֹא אָמַר לָךְ כָּךְ אֶלָּא שֶׁהוּא מְבַקֵּשׁ לִשָא אִשָּׁה אַחֶרֶת וְלִתְּנָם לָהּ. כָּךְ,(כז) הַמֶּלֶךְ זֶה אָדָם, הָאִשָּׁה זוֹ חַוָּה, זָקֵן שׁוֹאֵל חֹמֶץ זֶה הַנָּחָשׁ,(כח) וַעֲלֵיהֶם הוּא אוֹמֵר [תהלים לו, יג] שָׁם נָפְלוּ פֹּעֲלֵי אָוֶן. דָּן נָחָשׁ דִּין בֵּינוֹ לְבֵין עַצְמוֹ וְאָמַר(כט) אִם אֲנִי אוֹמֵר לָאָדָם, יוֹדֵעַ אֲנִי שֶׁאֵינוֹ שׁוֹמֵעַ לִי,(ל) שֶׁהָאִישׁ קָשֶׁה לְעוֹלָם לְהוֹצִיאוֹ מִדַּעְתּוֹ. אֶלָּא הֲרֵינִי אוֹמֵר לָאִשָּׁה(לא) שֶׁדַּעְתָּה קַלָּה עָלֶיהָ שֶׁאֲנִי יוֹדֵעַ שֶׁהִיא שׁוֹמַעַת לִי, שֶׁהַנָּשִׁים נִשְׁמָעוֹת לְכָל הַבְּרִיּוֹת,(לב) שֶׁנֶּאֱמַר [משלי ט, יג] פְּתַיּוּת וּבַל יָדְעָה מָה. הָלַךְ הַנָּחָשׁ וְאָמַר לָאִשָּׁה, אֱמֶת(לג) שֶׁאַף אַתֶּם(לד) מְצֻוִּים עַל פֵּרוֹת הָאִילָן הַזֶּה, אָמְרָה לוֹ הֵן, שֶׁנֶּאֱמַר [בראשית ג, ג] וּמִפְּרִי הָעֵץ אֲשֶׁר בְּתוֹךְ הַגָּן,(לה) מִתּוֹךְ דְּבָרֶיהָ מָצָא פֶּתַח לְהִכָּנֵס בּוֹ, אָמַר לָהּ אֵין צִוּוּי זֶה אֶלָּא עַיִן רָעָה, שֶׁבְּשָׁעָה שֶׁאַתֶּם אוֹכְלִין מִמֶּנּוּ תִּהְיוּ כֵּאלֹהִים.(לו) מָה הוּא עוֹשֶׂה, בּוֹרֵא עוֹלָמוֹת וּמַחֲרִיב עוֹלָמוֹת, כָך אַתֶּם יְכוֹלִין לִבְרוֹא עוֹלָמוֹת וּלְהַחֲרִיב עוֹלָמוֹת. מָה הוּא מֵמִית וּמְחַיֶּה אַף אַתֶּם יְכוֹלִין לְהָמִית וּלְהַחֲיוֹת, שֶׁנֶּאֱמַר כִּי יֹדֵעַ אֱלֹהִים כִּי בְּיוֹם אֲכָלְכֶם מִמֶּנּוּ וְנִפְקְחוּ עֵינֵיכֶם [שם ה]. הָלַךְ הַנָּחָשׁ וְנָגַע בָּאִילָן,(לז) צָוַח וְאָמַר רָשָׁע אַל תִּגַּע בִּי,(לח) שֶׁנֶּאֱמַר [תהלים לו, יב - יג] אַל תְּבוֹאֵנִי רֶגֶל גַּאֲוָה וְגוֹ' שָׁם נָפְלוּ פֹּעֲלֵי אָוֶן. הָלַךְ הַנָּחָשׁ וְאָמַר לָאִשָּׁה(לט) הֲרֵינִי נָגַעְתִּי בָּאִילָן וְלֹא מַתִּי, אַף אַתְּ גְּעִי בּוֹ וְלֹא תָמוּתִי. הָלְכָה הָאִשָּׁה וְנָגְעָה בָּאִילָן,(מ) וְרָאֲתָה מַלְאַךְ הַמָוֶת שֶׁבָּא כְּנֶגְדָּהּ, אָמְרָה(מא) אוּלַי עַכְשָׁו אֲנִי מֵתָה, וְהַקָּדוֹשׁ בָּרוּךְ הוּא עוֹשֶׂה לוֹ אִשָּׁה אַחֶרֶת וְנוֹתְנָהּ לָאָדָם, אֶלָּא הֲרֵינִי גוֹרֶמֶת לוֹ(מב) שֶׁיֹּאכַל עִמִּי, אִם נָמוּת שְׁנֵינוּ נָמוּת, וְאִם נִחְיֶה שְׁנֵינוּ נִחְיֶה. וְלָקְחָה וְאָכְלָה מִפֵּרוֹת הָאִילָן,(מג) וְנָתְנָה מִפֵּרוֹתָיו גַּם לְבַעְלָהּ שֶׁיֹּאכַל עִמָּהּ, שֶׁנֶּאֱמַר [בראשית ג, ו] וַתִּקַּח מִפִּרְיוֹ וַתֹּאכַל וַתִּתֵּן גַּם לְאִישָׁהּ. כֵּיוָן שֶׁאָכַל אָדָם מִפֵּרוֹת הָאִילָן,(מד) רָאָה אֶת עַצְמוֹ עָרֹם וְנִתְפַּקְּחוּ עֵינָיו(מה) וְקָהוּ שִׁנָּיו, אָמַר לָהּ(מו) מַה הוּא זֶה שֶׁהֶאֱכַלְתָּנִי שֶׁנִּתְפַּקְּחוּ עֵינַי וְקָהוּ שִׁנָּי, עַל דַּעְתִּי כְּשֵׁם שֶׁקָּהוּ שִׁנַּי כָּךְ יִקְהוּ שִׁנֵּי כָל הַדּוֹרוֹת.

(א) גְּמִילוּת חֲסָדִים לַאֲבֵלִים מִנַּיִן אָנוּ לְמֵדִים, מֵהַקָּדוֹשׁ בָּרוּךְ הוּא, שֶׁהוּא בְעַצְמוֹ(ב) גָּמַל חֶסֶד לְמֹשֶׁה עַבְדּוֹ(ג) וּקְבָרוֹ בְּיָדוֹ. וְאִלְמָלֵא הַדָּבָר כָּתוּב אִי אֶפְשָׁר לְאָמְרוֹ, שֶׁנֶּאֱמַר [דברים לד, ו] וַיִּקְבֹּר אֹתוֹ בַגַּי. רַבָּן גַּמְלִיאֵל בְּנוֹ שֶׁל רַבִּי יְהוּדָה הַנָּשִׂיא אוֹמֵר, לֹא לְמֹשֶׁה לְבַדּוֹ גָּמַל הַקָּדוֹשׁ בָּרוּךְ הוּא חֶסֶד,(ד) אֶלָּא אַף לְאַהֲרֹן, כְּשֶׁעָלוּ לְהֹר הָהָר הָיוּ מִדַּיְנִין כָּל שִׁבְטֵי יִשְׂרָאֵל וְאוֹמְרִים, מֹשֶׁה וְאֶלְעָזָר הִנִּיחוּ אֶת אַהֲרֹן בְּהֹר הָהָר וְיָרְדוּ לָהֶם,(ה) וְלֹא הֶאֱמִינוּ שֶׁגָּוַע(ו) וְלִגְמֹל לוֹ חֶסֶד. מֶה עָשָׂה הַקָּדוֹשׁ בָּרוּךְ הוּא, נָטַל אֲרוֹנוֹ שֶׁל אַהֲרֹן וְהֶעֱבִיר אוֹתוֹ(ז) מֵעַל הַמַּחֲנֶה, וְרָאוּ כָּל יִשְׂרָאֵל אֲרוֹנוֹ שֶׁל אַהֲרֹן פּוֹרֵחַ וְטָס בָּאֲוִיר, הֶאֱמִינוּ כִּי גָוַע אַהֲרֹן וְגָמְלוּ לוֹ חֶסֶד, שֶׁנֶּאֱמַר [במדבר כ, כט] וַיִּרְאוּ כָּל הָעֵדָה כִּי גָוַע אַהֲרֹן וְגוֹ'. לְמֹשֶׁה לֹא גָמְלוּ חֶסֶד אֶלָּא הָאֲנָשִׁים לְבָד, שֶׁנֶּאֱמַר [דברים לד, ח] וַיִבְכּוּ בְנֵי יִשְׂרָאֵל אֶת מֹשֶׁה. לְאַהֲרֹן גָּמְלוּ הָאֲנָשִׁים(ח) וְהַנָּשִׁים חֶסֶד, שֶׁנֶאֱמַר [במדבר כ, כט] וַיִּבְכּוּ אֶת אַהֲרֹן שְׁלֹשִׁים יוֹם וְגוֹ'. לָמָּה, שֶׁהָיָה(ט) אוֹהֵב שָׁלוֹם וְרוֹדֵף שָׁלוֹם, וְעוֹבֵר בְּמַחֲנֵה יִשְׂרָאֵל בְּכָל יוֹם וְנוֹתֵן שָׁלוֹם בֵּין אִישׁ לְאִשְׁתּוֹ וּבֵין אִישׁ לְרֵעֵהוּ, וּלְפִיכָךְ גָּמְלוּ לוֹ חֶסֶד כָּל יִשְׂרָאֵל.(י) וּפָסוּק זֶה לֹא זֶה הוּא מְקוֹמוֹ, אֶלָּא בְּמוֹסֵרָה, שָׁם מֵת אַהֲרֹן וַיִּקָּבֵר שָׁם, וְהֶעֱלָה עָלָיו הַכָּתוּב כְּאִלּוּ מֵת שָׁם וְנִקְבַּר שָׁם: רַבִּי יוֹסֵי אוֹמֵר,(יא) שִׁבְעַת יְמֵי הָאֵבֶל מִנַּיִן אָנוּ לְמֵדִין, מִיַּעֲקֹב אָבִינוּ שֶׁעָשָׂה לוֹ יוֹסֵף שִׁבְעַת יְמֵי הָאֵבֶל, שֶׁנֶּאֱמַר [בראשית נ, י] וַיַּעַשׂ לְאָבִיו אֵבֶל שִׁבְעַת יָמִים. וְעָלוּ כָּל גְּדוֹלֵי הַמַּלְכֻיוֹת עִמּוֹ,(יב) שֶׁנֶּאֱמַר [שם ט] וַיַּעַל עִמּוֹ גַּם רֶכֶב גַּם פָּרָשִׁים.(יג) אָמַר לָהֶם הַקָּדוֹשׁ בָּרוּךְ הוּא, אַתֶּם(יד) גְּמַלְתֶּם חֶסֶד עִם יַעֲקֹב עַבְדִּי, אַף אֲנִי אֶתֵּן לָכֶם וְלִבְנֵיכֶם שָׂכָר טוֹב בָּעוֹלָם הַזֶּה. כְּשֶׁמֵּתוּ הַמִּצְרִיִּים בַּיָּם(טו) לֹא טָבְעוּ בַּיָּם וְזָכוּ לְהִקָּבֵר שֶׁנֶּאֱמַר [שמות טו, יב] נָטִיתָ יְמִינְךָ וְגוֹ': גְּמִילוּת חֲסָדִים לַאֲבֵלִים מִנַּיִן אָנוּ לְמֵדִין, מֵאִיזֶבֶל,(טז) שֶׁהָיָה בֵּיתָהּ סָמוּךְ לַשּׁוּק,(יז) וְכָל חָתָן שֶׁהָיָה עוֹבֵר הָיְתָה יוֹצֵאת מִבֵּיתָהּ וּמְצַלְצֶלֶת בְּכַפֶּיהָ וּמְקַלֶּסֶת בְּפִיהָ(יח) וּמְהַלֶּכֶת עֶשֶׂר צְעָדוֹת.(יט) וְכָל מֵת שֶׁהָיָה עוֹבֵר בַּשּׁוּק, הָיְתָה יוֹצֵאת מִבֵּיתָהּ וּמְמַחָה בְּכַפֶּיהָ וּמְקוֹנֶנֶת בְּפִיהָ(כ) וּמְהַלֶּכֶת אַחֲרָיו. וְנִבָּא עָלֶיהָ אֵלִיָּהוּ זָכוּר לַטּוֹב וְאָמַר [מ"ב ט, לו] בְּחֵלֶק יִזְרְעֶאל יֹאכְלוּ הַכְּלָבִים אֶת בְּשַׂר אִיזָבֶל.(כא) וְהָאֵבָרִים שֶׁהָיוּ גוֹמְלִין חֶסֶד לֹא שָׁלְטוּ עֲלֵיהֶם הַכְּלָבִים, שֶׁנֶּאֱמַר [שם לה] וַיֵּלְכוּ לְקָבְרָהּ וְלֹא מָצְאוּ בָהּ כִּי אִם הַגֻּלְגֹּלֶת וְהָרַגְלַיִם וְכַפּוֹת הַיָּדָיִם: גְּמִילוּת חֲסָדִים לַאֲבֵלִים מִנַּיִן אָנוּ לְמֵדִין, מֵאַנְשֵׁי יָבֵשׁ גִּלְעָד, כְּשֶׁנֶּהֱרַג שָׁאוּל וּבָנָיו(כב) אָמְרוּ אַנְשֵׁי יָבֵשׁ גִּלְעָד,(כג) הָאִישׁ שֶׁהִצִּילָנוּ מֵחֶרְפַּת בְּנֵי עַמּוֹן אֵין אָנוּ חַיָּבִין לִגְמֹל לוֹ חָסֶד,(כד) וְעָמְדוּ כָּל הַגִּבּוֹרִים שֶׁלָּהֶם(כה) וְהָלְכוּ לָהֶם כָּל הַלַּיְלָה לְחוֹמַת בֵּית שֵׁן,(כו) שֶׁנֶּאֱמַר [דברי הימים א, י, יב] וַיָּקוּמוּ כָּל אִישׁ חַיִל וַיִשְׂאוּ אֶת גּוּפַת שָׁאוּל וְגוֹ'. לַאֲבֵלִים(כז) מְנַחֲמִים בַּלֶּחֶם וּבַיַּיִן, שֶׁנֶּאֱמַר [משלי לא, ו] וְיַיִן לְמָרֵי נָפֶשׁ. אַנְשֵׁי יָבֵשׁ גִּלְעָד(כח) הָיוּ גּוֹמְלִים חֶסֶד עִם שָׁאוּל וְעִם בָּנָיו(כט) בְּצוֹם וּבְכִי וּמִסְפֵּד,(ל) שֶׁנֶּאֱמַר [ש"א לא, יג] וַיָּצֻמוּ שִׁבְעַת יָמִים. אָמַר לָהֶם הַקָּדוֹשׁ בָּרוּךְ הוּא,(לא) אֲנִי אֶתֵּן שְׂכַרְכֶם לֶעָתִיד לָבֹא, כְּשֶׁאֲנִי עָתִיד לְקַבֵּץ אֶת יִשְׂרָאֵל מֵאַרְבַּע פִּנּוֹת הָעוֹלָם,(לב) רִאשׁוֹן הוּא מְקַבֵּץ חֲצִי שֵׁבֶט מְנַשֶּׁה, שֶׁנֶּאֱמַר [תהלים ס, ט] לִי גִלְעָד וְלִי מְנַשֶּׁה.(לג) וְאַחַר כָּךְ מְקַבֵּץ אֶפְרַיִם וִיהוּדָה, שֶׁנֶּאֱמַר [שם] אֶפְרַיִם מָעוֹז רֹאשִׁי [וְאַחַר כָּךְ] יְהוּדָה מְחֹקְקִי.(לד) לִי גִלְעָד, זֶה אַחְאָב שֶׁמֵּת בְּרָמוֹת גִּלְעָד. וְלִי מְנַשֶּׁה, כְּמַשְׁמָעוֹ. אֶפְרַיִם מָעוֹז רֹאשִׁי, זֶה יָרָבְעָם בֶּן נְבָט. יְהוּדָה מְחֹקְקִי, זֶה אֲחִיתֹפֶל.(לה) מוֹאָב סִיר רַחְצִי [שם י], זֶה גֵּחֲזִי. עַל אֱדוֹם אַשְׁלִיךְ נַעֲלִי [שם], זֶה דּוֹאֵג הָאֲדוֹמִי. עָלַי פְּלֶשֶׁת הִתְרוֹעָעִי [שם], אָמַר לָהֶם הַקָּדוֹשׁ בָּרוּךְ הוּא,(לו) עָלַי לְפַלֵּשׁ עֲלֵיהֶם זְכוּת וְלַעֲשׂוֹתָם רֵעִים אֵלּוּ לָאֵלּוּ: רַבִּי פִּנְחָס אוֹמֵר,(לז) לְאַחַר שָׁנָה שֶׁנֶּהֱרַג שָׁאוּל וּבָנָיו, בָּא רָעָב שָׁלֹשׁ שָׁנִים בִּימֵי דָוִד(לח) שָׁנָה אַחַר שָׁנָה, שֶׁנֶּאֱמַר וַיְהִי רָעָב בִּימֵי דָוִד שָׁלֹשׁ שָׁנִים [ש"ב כא, א]. בְּשָׁנָה רִאשׁוֹנָה עָלוּ יִשְׂרָאֵל לָרֶגֶל, אָמַר לָהֶם דָּוִד צְאוּ וּרְאוּ שֶׁמָּא יֵשׁ בָּכֶם בְּנֵי אָדָם עוֹבְדֵי עֲבוֹדָה זָרָה. שֶׁבַּעֲוֹן עֲבוֹדָה זָרָה הַגְּשָׁמִים נֶעְצָרִים, שֶׁנֶּאֱמַר [דברים יא, טז] הִשָּׁמְרוּ לָכֶם(לט) פֶּן יִפְתֶּה לְבַבְכֶם, יָצְאוּ וּבָדְקוּ וְלֹא מָצָאוּ. בְּשָׁנָה שְׁנִיָּה עָלוּ יִשְׂרָאֵל לָרֶגֶל, אָמַר לָהֶם דָּוִד צְאוּ וּרְאוּ שֶׁמָּא יֵשׁ בָּכֶם מְגַלֶּה עֲרָיוֹת. שֶׁבַּעֲוֹן גִּלּוּי עֲרָיוֹת הַגְּשָׁמִים נֶעְצָרִים, שֶׁנֶּאֱמַר [ירמיה ג, ב] וַתַּחֲנִיפִי אֶרֶץ בִּזְנוּתַיִךְ וּבְרָעָתֵךְ. מַה כְּתִיב שָׁם [שם ג] וַיִּמָּנְעוּ רְבִבִים(מ) וּמַלְקוֹשׁ לוֹא הָיָה. בָּדְקוּ וְלֹא מָצָאוּ.(מא) בְּשָׁנָה שְׁלִישִׁית עָלוּ יִשְׂרָאֵל לָרֶגֶל, אָמַר לָהֶם דָּוִד צְאוּ וּרְאוּ שֶׁמָּא יֵשׁ בָּכֶם שׁוֹפְכֵי דָמִים. שֶׁבַּעֲוֹן שְׁפִיכוּת דָּמִים הַגְּשָׁמִים נֶעְצָרִין,(מב) שֶׁנֶּאֱמַר [במדבר לה, לג] וְלֹא תַחֲנִיפוּ אֶת הָאָרֶץ, בָּדְקוּ וְלֹא מָצָאוּ. אָמַר לָהֶם דָּוִד(מג) מִכָּאן וְאֵילַךְ אֵין הַדָּבָר תָּלוּי אֶלָּא בִּי. מִיָּד עָמַד דָּוִד לִפְנֵי הַקָּדוֹשׁ בָּרוּךְ הוּא וְהִתְפַּלֵּל, וְאָמַר לוֹ עַל שָׁאוּל.(מד) אָמַר דָּוִד רִבּוֹנוֹ שֶׁל עוֹלָם, אֵינִי שָׁאוּל, שֶׁבְּיָמַי לֹא נֶעֶשְׂתָה עֲבוֹדָה זָרָה בְּיִשְׂרָאֵל. וְאֵינִי שָׁאוּל שֶׁנִּמְשַׁח בְּשֶׁמֶן הַמִּשְׁחָה. וְאֵינִי שָׁאוּל שֶׁחָלַק עִם שְׁמוּאֵל הַנָּבִיא, אָמַר לוֹ אַתָּה בָּאָרֶץ(מה) וְהוּא בְּחוּצָה לָאָרֶץ. מִיָּד עָמַד דָּוִד(מו) וְלָקַח אֶת כָּל חַכְמֵי יִשְׂרָאֵל וּגְדוֹלֵיהֶם(מז) וְעָבְרוּ אֶת הַיַּרְדֵּן וּבָאוּ לְיָבֵשׁ גִּלְעָד, וּמָצְאוּ אֶת עַצְמוֹת שָׁאוּל וִיהוֹנָתָן בְּנוֹ(מח) שֶׁלֹּא שָׁלְטָה בָּהֶם רִמָּה.(מט) וְעָלָיו נֶאֱמַר [תהלים לד, כא] שֹׁמֵר כָּל עַצְמוֹתָיו אַחַת מֵהֵנָּה לֹא נִשְׁבָּרָה. וְלָקַח אֶת עַצְמוֹתָם וּנְתָנָם בָּאָרוֹן וְעָבְרוּ אֶת הַיַּרְדֵּן,(נ) שֶׁנֶּאֱמַר [ש"ב כא, יד] וַיִּקְבְּרוּ אֶת עַצְמוֹת שָׁאוּל וִיהוֹנָתָן בְּנוֹ וַיַּעֲשׂוּ כֹּל אֲשֶׁר צִוָּה הַמֶּלֶךְ. מַה צִּוָּה הַמֶּלֶךְ, לִהְיוֹת מַעֲבִירִין אֶת אֲרוֹנוֹ שֶׁל שָׁאוּל בְּכָל גְּבוּל יִשְׂרָאֵל בְּכָל שֵׁבֶט וָשֵׁבֶט,(נא) וְהָיָה הַגְּבוּל שֶׁנִּכְנַס בּוֹ אֲרוֹנוֹ שֶׁל שָׁאוּל,(נב) הָיוּ יוֹצְאִין הֵם וּבְנֵיהֶם וּבְנוֹתֵיהֶם וְגוֹמְלִים חֶסֶד, כְּדֵי שֶׁיֵּצְאוּ יְדֵי חוֹבָתָן בִּגְמִילוּת חֲסָדִים,(נג) עַד שֶׁבָּא לִגְבוּל יִשְׂרָאֵל לְאֶרֶץ בִּנְיָמִין,(נד) שֶׁנֶּאֱמַר וַיִּקְבְּרוּ אוֹתוֹ בִּגְבוּל נַחֲלָתוֹ. כֵּיוָן שֶׁרָאָה הַקָּדוֹשׁ בָּרוּךְ הוּא שֶׁגָּמְלוּ לוֹ חֶסֶד כָּל יִשְׂרָאֵל, מִיָּד נִתְמַלֵּא רַחֲמִים וְנָתַן מָטָר עַל הָאָרֶץ, שֶׁנֶּאֱמַר [שם] וַיֵּעָתֵר אֱלֹהִים לָאָרֶץ אַחֲרֵי כֵן: רַבִּי נְתַנְאֵל אוֹמֵר,(נה) שְׁלֹשׁ מֵאוֹת שָׁנָה(נו) עַד שֶׁלֹּא נוֹלַד יֹאשִׁיָּהוּ נִקְרָא שְׁמוֹ, שֶׁנֶּאֱמַר [מ"א יג, ב] הִנֵּה בֵן נוֹלָד לְבֵית דָּוִד יֹאשִׁיָּהוּ שְׁמוֹ. וּבֶן שְׁמֹנֶה שָׁנָה מָלַךְ.(נז) וְכִי נַעַר בֶּן שְׁמוֹנֶה שָׁנָה מַה הִיא דַּעְתּוֹ, אֶלָּא(נח) מָאַס אֶת הָאֱלִילִים(נט) וְשִׁבֵּר אֶת הַמַּצֵּבוֹת וְכָרַת אֶת הָאֲשֵׁרִים(ס) וְגִדַּע אֶת הַפְּסִילִים,(סא) וְהָיְתָה זְכוּתוֹ מַבְהֶקֶת לִפְנֵי הַקָּדוֹשׁ בָּרוּךְ הוּא וְכִסֵּא הַכָּבוֹד.(סב) וּבִשְׁבִיל הָרָעָה שֶׁהָיוּ עוֹשִׂין בַּסֵּתֶר נֶאֱסַף הַצַּדִּיק. וְנֶאֶסְפוּ(סג) כָּל אַנְשֵׁי יְהוּדָה וִירוּשָׁלַיִם וְיִרְמְיָה הַנָּבִיא לִגְמֹל חֶסֶד עִם יֹאשִׁיָּהוּ, שֶׁנֶּאֱמַר [דברי הימים ב, לה, כה] וַיְקוֹנֵן יִרְמְיָהוּ עַל יֹאשִׁיָּהוּ וְגוֹ'. רַבִּי מֵאִיר אוֹמֵר, הַשָּׁרִים [שם] אֵלּוּ הַלְוִיִּם הָעוֹמְדִים עַל דּוּכָנָם בְּשִׁיר.(סד) וְהַשָּׁרוֹת [שם] אֵלּוּ נְשׁוֹתֵיהֶם.(סה) רַבִּי שִׁמְעוֹן אוֹמֵר, לֹא עַל הַלְוִיִּם וּנְשֵׁיהֶם אָמְרוּ(סו) אֶלָּא עַל כָּל הַנָּשִׁים הַחֲכָמוֹת, שֶׁנֶּאֱמַר [ירמיה ט, טז] כֹּה אָמַר ה' צְבָאוֹת הִתְבּוֹנְנוּ וְקִרְאוּ לַמְקוֹנְנוֹת וּתְבוֹאֶינָה.(סז) וְכֵן קִבְּלוּ עֲלֵיהֶם כָּל יִשְׂרָאֵל לִהְיוֹת גּוֹמְלִין חֶסֶד עִם כָּל חַכְמֵי יִשְׂרָאֵל(סח) וּגְדוֹלֵיהֶן, [שֶׁנֶאֱמַר [דברי הימים ב, לה, כה]] וַיִּתְּנוּם לְחֹק עַל יִשְׂרָאֵל:(סט) רָאָה שְׁלֹמֹה שֶׁמִּדַּת גְּמִילוּת חֲסָדִים גְּדוֹלָה לִפְנֵי הַמָּקוֹם, וּכְשֶׁבָּנָה בֵּית הַמִּקְדָּשׁ(ע) בָּנָה שְׁנֵי שְׁעָרִים, אֶחָד לַחֲתָנִים וְאֶחָד לַאֲבֵלִים וְלַמְנֻדִּים. וְהָיוּ יִשְׂרָאֵל הוֹלְכִין בַּשַּׁבָּתוֹת וְיוֹשְׁבִין בֵּין שְׁנֵי שְׁעָרִים הַלָּלוּ, וְהַנִּכְנָס בְּשַׁעַר חֲתָנִים הָיוּ יוֹדְעִין שֶׁהוּא חָתָן, וְהָיוּ אוֹמְרִים לוֹ, הַשּׁוֹכֵן בַּבַּיִת הַזֶּה יְשַׂמֵּחֲךָ בְּבָנִים וּבְבָנוֹת.(עא) וְהַנִּכְנָס בְּשַׁעַר אֲבֵלִים וְהָיָה שְׂפָמוֹ מְכֻסֶּה אָז הָיוּ יוֹדְעִין שֶׁהוּא אָבֵל, וְהָיוּ אוֹמְרִים לוֹ, הַשּׁוֹכֵן בַּבַּיִת הַזֶּה יְנַחֶמְךָ. וְהַנִּכְנָס בְּשַׁעַר הָאֲבֵלִים(עב) וְלֹא הָיָה שְׂפָמוֹ מְכֻסֶּה, אָז הָיוּ יוֹדְעִין שֶׁהוּא מְנֻדֶּה, וְהָיוּ אוֹמְרִים לוֹ הַשּׁוֹכֵן בַּבַּיִת הַזֶּה יְנַחֶמְךָ,(עג) וְיִתֵּן בִּלְבָבְךָ שֶׁתִּשְׁמַע לַחֲבֵרֶיךָ וִיקָרְבוּךָ, כְּדֵי שֶׁיֵּצְאוּ כָּל יִשְׂרָאֵל יְדֵי חוֹבָתָן בִּגְמִילוּת חֲסָדִים.(עד) מִשֶּׁחָרַב בֵּית הַמִּקְדָּשׁ הִתְקִינוּ שֶׁיִהְיוּ חֲתָנִים וַאֲבֵלִים הוֹלְכִין לְבָתֵּי כְנֵסִיּוֹת וּלְבָתֵּי מִדְרָשׁוֹת, וְאַנְשֵׁי הַמָּקוֹם רוֹאִין אֶת הֶחָתָן וּשְׂמֵחִים עִמּוֹ, וְרוֹאִין אֶת הָאָבֵל וְיוֹשְׁבִים עִמּוֹ עַל הָאָרֶץ,(עה) כְּדֵי שֶׁיֵּשְׁבוּ וְיֵצְאוּ יִשְׂרָאֵל יְדֵי חוֹבָתָן בִּגְמִילוּת חֲסָדִים.(עו) וַעֲלֵיהֶם הוּא אוֹמֵר בָּרוּךְ אַתָּה ה' גּוֹמֵל שָׂכָר טוֹב לְגוֹמְלֵי חֲסָדִים.

כְּתִיב [בראשית ג, ג] וּמִפְּרִי הָעֵץ אֲשֶׁר בְּתוֹךְ הַגָּן, תָּנָא רַבִּי זְעִירָא אוֹמֵר, מִפְּרִי הָעֵץ,(א) אֵין הָעֵץ הַזֶּה אֶלָּא אָדָם שֶׁנִּמְשַׁל כָּעֵץ, שֶׁנֶּאֱמַר [דברים כ, יט] כִּי הָאָדָם עֵץ הַשָׂדֶה וְגוֹ'. אֲשֶׁר בְּתוֹךְ הַגָּן, אֵין בְּתוֹךְ הַגָּן אֶלָּא לָשׁוֹן נְקִיָּה,(ב) מַה שֶׁבְּתוֹךְ הַגּוּף, אֲשֶׁר בְּתוֹךְ הַגָּן, אֲשֶׁר בְּתוֹךְ הָאִשָּׁה. וְאֵין גַּן אֶלָּא הָאִשָּׁה שֶׁנִּמְשְׁלָה לְגַּן, שֶׁנֶּאֱמַר [שה"ש ד, יב] גַּן נָעוּל אֲחֹתִי כַלָּה, מָה(ג) הַגִּנָּה הַזּוֹ כָּל מַה שֶׁנִּזְרְעָה הִיא צוֹמַחַת וּמוֹצִיאָה,(ד) כָּךְ הָאִשָּׁה הַזֹּאת כָּל מַה שֶׁנִּזְרְעָה(ה) הָרָה וְיֹלֶדֶת מִבַּעְלָהּ. בָּא אֵלֶיהָ(ו) (ו)רוֹכֵב(ת) נָחָשׁ וְעִבְּרָה אֶת קַיִן, וְאַחַר כָּךְ עִבְּרָה אֶת הֶבֶל, שֶׁנֶּאֱמַר [בראשית ד, א] וְהָאָדָם יָדַע אֶת חַוָּה אִשְׁתּוֹ.(ז) מַהוּ יָדַע, שֶׁהָיְתָה מְעֻבֶּרֶת(ח) וְרָאֲתָה דְמוּתוֹ שֶׁלֹּא הָיָה מִן הַתַּחְתּוֹנִים אֶלָּא מִן הָעֶלְיוֹנִים, וְהִבִּיטָה וְאָמְרָה קָנִיתִי אִישׁ אֶת ה' [שם].(ט) רַבִּי יִשְׁמָעֵאל אוֹמֵר, מִשֵּׁם עָלוּ וְנִתְיַחֲסוּ כָּל דּוֹרוֹת הַצַּדִּיקִים, וּמִקַּיִן עָלוּ וְנִתְיַחֲסוּ כָּל דּוֹרוֹת הָרְשָׁעִים הַמּוֹרְדִים וְהַפּוֹשְׁעִים בַּמָּרוֹם וְאוֹמְרִים אֵין אָנוּ צְרִיכִין לְטִפַּת גְּשָׁמֶיךָ, שֶׁנֶּאֱמַר [איוב כא, יד] וַיֹּאמְרוּ לָאֵל סוּר מִמֶּנּוּ: רַבִּי מִיאֲשָׁא אוֹמֵר,(י) נוֹלַד קַיִן וּתְאוֹמָתוֹ עִמּוֹ, נוֹלַד הֶבֶל וּתְאוֹמָתוֹ עִמּוֹ. אָמַר לוֹ רַבִּי יִשְׁמָעֵאל,(יא) וַהֲלֹא כְּבָר נֶאֱמַר [ויקרא כ, יז] וְאִישׁ אֲשֶׁר יִקַּח אֶת אֲחֹתוֹ בַּת אָבִיו. אָמַר לוֹ,(יב) מִתּוֹךְ הַדְּבָרִים הָאֵלֶּה תֵּדַע לְךָ שֶׁלֹּא הָיוּ נָשִׁים אֲחֵרוֹת בָּעוֹלָם שֶׁיִּשְׂאוּ לָהֶן, וְהִתִּירָן לָהֶם, וְעַל זֶה נֶאֱמַר [תהלים פט, ג] כִּי אָמַרְתִּי עוֹלָם חֶסֶד יִבָּנֶה, בְּחֶסֶד נִבְרָא הָעוֹלָם(יג) עַד שֶׁלֹּא נִתְּנָה תוֹרָה. רַבִּי יוֹסֵי אוֹמֵר,(יד) קַיִן וְהֶבֶל תְּאוֹמִים הָיוּ, שֶׁנֶּאֱמַר [בראשית ד, א] וַתַּהַר וַתֵּלֶד אֶת קַיִן.(טו) בְּהַהִיא שַׁעְתָּא אוֹסִיפַת לְמֵילַד, שֶׁנֶּאֱמַר [שם ב] וַתֹּסֶף לָלֶדֶת.(טז) וְהָיָה קַיִן אוֹהֵב לַעֲבֹד אֲדָמָה לִזְרֹעַ, וְהֶבֶל הָיָה אוֹהֵב לִרְעוֹת צֹאן,(יז) זֶה נוֹתֵן מִמְּלַאכְתּוֹ מַאֲכָל לָזֶה וְזֶה נוֹתֵן מִמְּלַאכְתּוֹ מַאֲכָל לָזֶה.(יח) הִגִּיעַ לֵיל יוֹם טוֹב שֶׁל פֶּסַח, אָמַר לָהֶם אָדָם לְבָנָיו, בְּלֵיל זֶה עֲתִידִין יִשְׂרָאֵל לְהַקְרִיב קָרְבְּנוֹת פְּסָחִים, הַקְרִיבוּ גַם אַתֶּם לִפְנֵי בּוֹרַאֲכֶם. הֵבִיא קַיִן(יט) מוֹתַר מַאֲכָלוֹ קְלָיּוֹת זֶרַע פִּשְׁתָּן, וְהֵבִיא הֶבֶל מִבְּכוֹרוֹת צֹאנוֹ וּמֵחֶלְבֵּהֶן(כ) כְּבָשִׂים שֶׁלֹּא נִגְזְזוּ לַצֶּמֶר.(כא) וְנִתְעַב מִנְחַת קַיִן וְנִרְצֵית מִנְחַת הֶבֶל, שֶׁנֶּאֱמַר [שם ד] וַיִּשַׁע ה' אֶל הֶבֶל וְאֶל מִנְחָתוֹ. רַבִּי יְהוֹשֻׁעַ בֶּן קָרְחָה אוֹמֵר, אָמַר הַקָּדוֹשׁ בָּרוּךְ הוּא(כב) אַל יִתְעָרְבוּ מִנְחַת קַיִן וְהֶבֶל לְעוֹלָם, שֶׁמָּא חַס וְשָׁלוֹם יִתְעָרְבוּ בְּאֶרֶג בֶּגֶד, שֶׁנֶּאֱמַר [דברים כב, יא] לֹא תִלְבַּשׁ שַׁעַטְנֵז וְגוֹ'.(כג) אֲפִלוּ הִיא מְרוּכֶּבֶת לֹא יַעֲלֶה עָלֶיךָ, שֶׁנֶּאֱמַר [ויקרא יט, יט] וּבֶגֶד כִּלְאַיִם שַׁעַטְנֵז לֹא יַעֲלֶה עָלֶיךָ: רַבִּי צָדוֹק אוֹמֵר,(כד) נִכְנְסָה קִנְאָה וְשִׂנְאָה גְדוֹלָה בְּלִבּוֹ שֶׁל קַיִן, עַל שֶׁנִּרְצֵית מִנְחָתוֹ שֶׁל הֶבֶל.(כה) וְלֹא עוֹד אֶלָּא שֶׁהָיְתָה אִשְׁתּוֹ תְאוֹמָתוֹ יָפָה בַּנָּשִׁים, אָמַר אֲנִי אֶהֱרֹג אֶת הֶבֶל אָחִי וְאֶקַּח אֶת אִשְׁתּוֹ, שֶׁנֶּאֱמַר [בראשית ד, ח] וַיֹּאמֶר קַיִן אֶל הֶבֶל אָחִיו וַיְהִי בִּהְיוֹתָם בַּשָׂדֶה. וְאֵין בַּשָׂדֶה אֶלָּא הָאִשָּׁה(כו) שֶׁנִּמְשְׁלָה כַּשָׂדֶה, שֶׁנֶּאֱמַר [דברים כ, יט] כִּי הָאָדָם עֵץ הַשָׂדֶה.(כז) לָקַח אֶבֶן וְטִבְּעָהּ בְּמִצְחוֹ שֶׁל הֶבֶל וַהֲרָגוֹ, שֶׁנֶּאֱמַר [בראשית שם] וַיָּקָם קַיִן אֶל הֶבֶל אָחִיו וַיַּהַרְגֵהוּ. רַבִּי יוֹחָנָן אָמַר,(כח) לֹא הָיָה יוֹדֵע קַיִן שְׁהַנִּסְתָּרוֹת גְּלוּיוֹת לִפְנֵי הַמָּקוֹם. מֶה עָשָׂה,(כט) נָטַל אֶת נִבְלָתוֹ שֶׁל אָחִיו וְחָפַר וְטָמַן אוֹתוֹ בָּאָרֶץ, אָמַר לוֹ הַקָּדוֹשׁ בָּרוּךְ הוּא, אֵי הֶבֶל אָחִיךָ, אָמַר לְפָנָיו, רִבּוֹנוֹ שֶׁל עוֹלָם(ל) שׁוֹמֵר שָׂדֶה וָכֶרֶם שַׂמְתָּנִי, שׁוֹמֵר אָחִי לֹא שַׂמְתָּנִי, שֶׁנֶּאֱמַר [שם ט] הֲשֹׁמֵר אָחִי אָנֹכִי.(לא) אָמַר לוֹ הַקָּדוֹשׁ בָּרוּךְ הוּא, הֲרָצַחְתָּ גַם שָׁמַעְתִּי, קוֹל דְּמֵי אָחִיךָ צוֹעֲקִים אֵלָי.(לב) כֵּיוָן שֶׁשָּׁמַע קַיִן הַדָּבָר(לג) נִבְהַל. וְאֵרְרוֹ(לד) שֶׁיִּהְיֶה נוֹדֵד בָּאָרֶץ,(לה) עַל שְׁפִיכוּת דָּמִים וְעַל מָוֶת רָע. אָמַר קַיִן לְפָנָיו רִבּוֹן הָעוֹלָמִים,(לו) גָּדוֹל עֲוֹנִי מִנְּשׂוֹא(לז) שֶׁאֵין בּוֹ כַּפָּרָה, וְנֶחְשַׁב לוֹ הַדָּבָר הַזֶּה כִּתְשׁוּבָה, שֶׁנֶּאֱמַר [שם יג] גָּדוֹל עֲוֹנִי מִנְּשׂוֹא.(לח) וְלֹא עוֹד אֶלָּא יַעֲמֹד אֶחָד בָּאָרֶץ וְיַזְכִּיר אֶת שִׁמְךָ הַגָּדוֹל וְיַהַרְגֵנִי. מֶה עָשָׂה הַקָּדוֹשׁ בָּרוּךְ הוּא, נָטַל(לט) אוֹת אַחַת מֵעֶשְׂרִים וּשְׁתַּיִם אוֹתִיּוֹת שֶׁבַּתּוֹרָה וְכָתַב(מ) עַל זְרוֹעוֹ שֶׁל קַיִן שֶׁלֹּא יֵהָרֵג, שֶׁנֶּאֱמַר [שם טו] וַיָּשֶׂם ה' לְקַיִן אוֹת.(מא) הַכֶּלֶב שֶׁהָיָה מְשַׁמֵּר צֹאנוֹ שֶׁל הֶבֶל, הוּא(מב) הָיָה שׁוֹמְרוֹ(מג) מִחַיַּת הַשָׂדֶה וּמֵעוֹף הַשָּׁמָיִם. וְהָיוּ אָדָם וְעֶזְרוֹ(מד) יוֹשְׁבִים וּבוֹכִים וּמִתְאַבְּלִים עָלָיו, וְלֹא הָיוּ יוֹדְעִים מַה לַעֲשׂוֹת לְהֶבֶל,(מה) שֶׁלֹּא הָיוּ נְהוּגִים בִּקְבוּרָה.(מו) בָּא עוֹרֵב אֶחָד שֶׁמֵּת לוֹ אֶחָד מֵחֲבֵרָיו, לָקַח אוֹתוֹ וְחָפַר בָּאָרֶץ וּטְמָנוֹ לְעֵינֵיהֶם, אָמַר אָדָם, כָּעוֹרֵב אֲנִי עוֹשֶׂה, מִיָּד לָקַח נִבְלָתוֹ שֶׁל הֶבֶל וְחָפַר בָּאָרֶץ וּטְמָנָהּ. וְשִׁלֵּם הַקָּדוֹשׁ בָּרוּךְ הוּא שָׂכָר טוֹב לָעוֹרְבִים בָּעוֹלָם הַזֶּה. וּמַה שָׂכָר נָתַן לָהֶם, כְּשֶׁהֵן מוֹלִידִין אֶת בְּנֵיהֶם(מז) רוֹאִין אוֹתָן לְבָנִים וּבוֹרְחִים מִפְּנֵיהֶם(מח) וּסְבוּרִים שֶׁהֵם בְּנֵי נָחָשׁ, וְהַקָּדוֹשׁ בָּרוּךְ הוּא(מט) נוֹתֵן לָהֶם מְזוֹנָם(נ) בְּלֹא חִיסוּר. וְלֹא עוֹד אֶלָּא שֶׁהֵם(נא) קוֹרְאִים לִתֵּן מָטָר עַל הָאָרֶץ, וְהַקָּדוֹשׁ בָּרוּךְ הוּא עוֹנֶה אוֹתָן,(נב) שֶׁנֶּאֱמַר [תהלים קמז, ט] נוֹתֵן לִבְהֵמָה לַחְמָהּ לִבְנֵי עֹרֵב אֲשֶׁר יִקְרָאוּ.

(א) יְרִידָה שְׁלִישִׁית, שֶׁיָּרַד הַקָּדוֹשׁ בָּרוּךְ הוּא לִסְדוֹם, שֶׁנֶּאֱמַר [בראשית יח, כא] אֵרְדָה נָּא וְאֶרְאֶה. אָמַר הַקָּדוֹשׁ בָּרוּךְ הוּא,(ב) דָּבָר גָּדוֹל שֶׁאֲנִי עָתִיד לַעֲשׂוֹת בְּעוֹלָמִי אֲנִי מַגִּיד לְאַבְרָהָם אוֹהֲבִי, שֶׁנֶּאֱמַר [שם יז] וַה' אָמָר הַמְכַסֶּה אֲנִי מֵאַבְרָהָם וְגוֹ'. רַבִּי חֲנִינָא אוֹמֵר,(ג) נִגְלָה הַקָּדוֹשׁ בָּרוּךְ הוּא וּשְׁלשָׁה מַלְאָכִים עַל אַבְרָהָם אָבִינוּ, שֶׁנֶּאֱמַר [שם ב] וַיִּשָׂא עֵינָיו וַיַּרְא וְהִנֵּה שְׁלשָׁה אֲנָשִׁים. אֶחָד הִתְחִיל מְבַשֵׂר אוֹתוֹ(ד) עַל הֵרָיוֹן בֶּטֶן שֶׁל שָׂרָה אִשְׁתּוֹ, שֶׁנֶּאֱמַר [שם יד] לַמּוֹעֵד אָשׁוּב אֵלֶיךָ.(ה) וְאֶחָד הִגִּיד לוֹ מַעֲשֵׂה סְדוֹם וַעֲמוֹרָה, שֶׁנֶּאֱמַר [שם כ] וַיֹּאמֶר ה' זַעֲקַת סְדוֹם וַעֲמוֹרָה כִּי רָבָּה. מִכָּאן אָמְרוּ כָּל מִי שֶׁהוּא מְדַבֵּר עִם חֲבֵרוֹ דָּבָר שֶׁהוּא גְּנַאי לוֹ,(ו) מַתְחִיל לוֹ בִּדְבָרִים טוֹבִים וְגוֹמֵר בְּדָבָר שֶׁאֵינוֹ לְרָצוֹן לוֹ. מִנַּיִן אָנוּ לְמֵדִים, מֵהַקָּדוֹשׁ בָּרוּךְ הוּא, כְּשֶׁנִּגְלָה עַל אַבְרָהָם אָבִינוּ הִתְחִיל מְבַשְׂרוֹ עַל הַהֵרָיוֹן שֶׁל שָׂרָה אִשְׁתּוֹ,(ז) וְאַחַר כָּךְ הִגִּיד לוֹ מַעֲשֵׂה סְדוֹם.(ח) הִתְחִיל(ט) מְבַקֵּשׁ וּמִתְחַנֵּן לְפָנָיו(י) עַל לוֹט בֶּן אָחִיו, אָמַר לְפָנָיו רִבּוֹן כָּל הָעוֹלָמִים, כְּמוֹת הָרָשָׁע כֵּן יָמוּת הַצַּדִּיק שֶׁנֶּאֱמַר [שם יח, כג] הַאַף תִּסְפֶּה צַדִּיק עִם רָשָׁע.(יא) אָמַר לוֹ הַקָּדוֹשׁ בָּרוּךְ הוּא, אַבְרָהָם, חַיֶּיךָ בִּזְכוּת חֲמִשִּׁים צַדִּיקִים אֶסְלַח לִסְדוֹם, שֶׁנֶּאֱמַר [שם כו] אִם אֶמְצָא בִסְדוֹם חֲמִשִּׁים צַדִּיקִים(יב) וְאֶסְלַח לְכָל עֲוֹנוֹתָם. מִכָּאן אָמְרוּ(יג) אִם יֵשׁ חֲמִשִּׁים צַדִּיקִים בָּעוֹלָם, בְּצִדְקָתָם הָעוֹלָם עוֹמֵד. הִתְחִיל מְבַקֵּשׁ וּמִתְחַנֵּן לְפָנָיו, עַד שֶׁהִגִּיעַ לַעֲשָׂרָה. מִכָּאן אָמְרוּ חֲכָמִים, אִם יֵשׁ(יד) עֲשָׂרָה צַדִּיקִים בְּמָקוֹם בִּזְכוּתָן הַמָּקוֹם נִצּוֹל, שֶׁנֶּאֱמַר [שם לב] וַיֹּאמֶר לֹא אַשְׁחִית בַּעֲבוּר הָעֲשָׂרָה: רַבִּי זְעִירָא אוֹמֵר,(טו) עֲשִׁירֵי עוֹלָם הָיוּ אַנְשֵׁי סְדוֹם(טז) מִן הָאָרֶץ הַטּוֹבָה וְהַשְּׁמֵנָה שֶׁהָיוּ יוֹשְׁבִין עָלֶיהָ,(יז) שֶׁכָּל צֹרֶךְ שֶׁבָּעוֹלָם הָיָה יוֹצֵא מִמֶּנָּה, שֶׁנֶּאֱמַר [איוב כח, ו] וְעַפְרֹת זָהָב לוֹ. בְּשָׁעָה שֶׁהָיָה אָדָם יוֹצֵא וּמְבַקֵּשׁ לִקַּח לוֹ יָרָק, הָיָה אוֹמֵר לְעַבְדּוֹ צֵא וְקַח לִי בְּאִסָּר יָרָק, הוּא(יח) הוֹלֵךְ לִקַּח לוֹ יָרָק(יט) וּמוֹצֵא תַחְתָּיו זָהָב, שֶׁנֶּאֱמַר וְעַפְרֹת זָהָב לוֹ. וְכֶסֶף הָיָה יוֹצֵא מִמֶּנָּה, דִּכְתִיב כִּי יֵשׁ לַכֶּסֶף מוֹצָא [שם א].(כ) אֲבָנִים טוֹבוֹת וּמַרְגָּלִיוֹת הָיוּ יוֹצְאִים מִמֶּנָּה, שֶׁנֶּאֱמַר [שם ו] מְקוֹם סַפִּיר אֲבָנֶיהָ. לֶחֶם הָיָה יוֹצֵא מִמֶּנָּה, דִּכְתִיב [שם ה] אֶרֶץ מִמֶּנָּה יֵצֵא לָחֶם. וְלֹא בָטְחוּ(כא) בְּצֵל יוֹצְרָם(כב) אֶלָּא בְּרֹב עָשְׁרָם,(כג) שֶׁהָעֹשֶׁר דּוֹחֶה [נ"א: מהם] מִבְּעָלָיו יִרְאַת שָׁמַיִם(כד) הַבּוֹטְחִים עַל חֵילָם. רַבִּי יְהוֹשֻׁעַ בֶּן קָרְחָה אוֹמֵר,(כה) לֹא חָסוּ עַל כְּבוֹד קוֹנָם לִשְׁבֹּר אֹכֶל(כו) לָאוֹרֵחַ וְלַגֵּר, אֶלָּא(כז) פָּסְקוּ אֶת כָּל הָאִילָנוֹת לְמַעְלָה מִפֵּרוֹתֵיהֶן, שֶׁלֹּא יְהֵא הֲנָאָה מֵהֶם אֲפִלּוּ לְעוֹף הַשָּׁמַיִם, שֶׁנֶּאֱמַר [איוב כח, ז] נָתִיב לֹא יְדָעוֹ עָיִט. רַבִּי נְתַנְאֵל אוֹמֵר,(כח) הֵקִימוּ עֲלֵיהֶם שׁוֹפְטִים שׁוֹפְטֵי שֶׁקֶר, וְכָל אוֹרֵחַ וְגֵר שֶׁהָיָה נִכְנַס בִּסְדוֹם הָיוּ עוֹשְׁקִים בְּדִינָם הַמְעֻוָּת וּמוֹצִיאִין אוֹתָם עֲרֻמִּים,(כט) שֶׁנֶּאֱמַר [יחזקאל כב, כט] וְאֶת הַגֵּר עָשְׁקוּ בְּלֹא מִשְׁפָּט.(ל) וּשְׂבֵעִים בְּכָל תְּבוּאַת הָאָרֶץ.(לא) וְהָיוּ יוֹשְׁבִין לָבֶטַח וְשַׁאֲנָן וְשַׁלְוָה בְּלִי פַחַד מִלְחָמָה מִכָּל סְבִיבוֹתֵיהֶם.(לב) שְׂבֵעִים בְּכָל טוּב.(לג) וְלֹא הֶחֱזִיקוּ יָדָם בְּפַת לֶחֶם לֹא לֶעָנִי וְלֹא לָאֶבְיוֹן, שֶׁנֶּאֱמַר [שם טז, מט] הִנֵּה זֶה הָיָה עֲוֹן סְדֹם אֲחוֹתֵךְ: רַבִּי יְהוּדָה אוֹמֵר, הִכְרִיזוּ בִסְדוֹם וְאָמְרוּ, כָּל מִי שֶׁהוּא מַחֲזִיק יָדוֹ בְּפַת לֶחֶם לֶעָנִי לַגֵּר וְלָאֶבְיוֹן(לד) יִשָׂרֵף בָּאֵשׁ.(לה) פְּלוֹטִית בִּתּוֹ שֶׁל לוֹט הָיְתָה(לו) נְשׂוּאָה(לז) לְאֶחָד מִגְּדוֹלֵי סְדוֹם, רָאֲתָה(לח) עָנִי אֶחָד מְדֻקְדָּק(לט) בִּרְחוֹב הָעִיר,(מ) וְעָגְמָה נַפְשָׁהּ עָלָיו. מֶה הָיְתָה עוֹשָׂה, בְּכָל יוֹם כְּשֶׁהָיְתָה יוֹצֵאת לִשְׁאֹב מַיִם הָיְתָה נוֹתֶנֶת(מא) בַּכַּד שֶׁלָּהּ מִכָּל מְזוֹן בֵּיתָהּ, וּמְכַלְכֶּלֶת אוֹתוֹ הֶעָנִי. אָמְרוּ אַנְשֵׁי סְדוֹם, הֶעָנִי הַזֶּה מֵאַיִן הוּא חָי,(מב) עַד שֶׁיָּדְעוּ הַדָּבָר וְהוֹצִיאוּ אוֹתָהּ לְהִשָׂרֵף, אָמְרָה אֱלֹהֵי עוֹלָם(מג) עֲשֵׂה מִשְׁפָּטִי וְדִינִי מֵאַנְשֵׁי סְדוֹם,(מד) וְעָלְתָה זַעֲקָתָהּ לִפְנֵי כִּסֵּא הַכָּבוֹד, אָמַר הַקָּדוֹשׁ בָּרוּךְ הוּא, אֵרְדָה נָּא וְאֶרְאֶה הַכְּצַעֲקָתָהּ הַבָּאָה אֵלַי [בראשית יח, כא],(מה) אִם כְּצַעֲקַת(מו) הַנַּעֲרָה הַזֹּאת עָשׂוּ אַנְשֵׁי סְדוֹם,(מז) אֶהֱפֹךְ יְסוֹדוֹתֶיהָ שֶׁלָּהּ לְמַעְלָה וּפָנֶיהָ לְמַטָּה, שֶׁנֶּאֱמַר הַכְּצַעֲקָתָהּ, הַכְּצַעֲקָתָם אֵין כְּתִיב אֶלָּא הַכְּצַעֲקָתָהּ.(מח) וְכֵן הוּא אוֹמֵר [משלי יג, כ] הוֹלֵךְ אֶת חֲכָמִים יֶחְכָּם, לְמָה הוּא דוֹמֶה, לְמִי שֶׁנִּכְנַס לְבֵית הַמְרַקְּחִים, אַף עַל פִּי שֶׁלֹּא לָקַח וְלֹא נָתַן כְּלוּם, מִכָּל מָקוֹם לָקַח רֵיחַ טוֹב וְהוֹצִיא עִמּוֹ. כָּךְ כָּל מִי שֶׁהוּא הוֹלֵךְ עִם הַצַּדִּיקִים לוֹקֵחַ מִדַּרְכֵיהֶם וּמִמַּעֲשֵׂיהֶם הַטּוֹבִים. וְרֹעֶה כְסִילִים יֵרוֹעַ [שם], לְמָה הוּא דוֹמֶה, לְאָדָם שֶׁהוּא נִכְנַס לְבֵית הַבֻּרְסְקִי, אַף עַל פִּי שֶׁלֹּא נָתַן וְלֹא לָקַח כְּלוּם, מִכָּל מָקוֹם רֵיחַ רָע קָלַט וְהוֹצִיא עִמּוֹ, כָּךְ מִי שֶׁהוּא הוֹלֵךְ עִם הָרְשָׁעִים לוֹקֵחַ מִדַּרְכֵיהֶם וּמִמַּעֲשֵׂיהֶם הָרָעִים, שֶׁנֶּאֱמַר וְרֹעֶה כְסִילִים יֵרוֹעַ. דָּבָר אַחֵר הוֹלֵךְ אֶת חֲכָמִים יֶחְכָּם,(מט) זֶה לוֹט, שֶׁהָיָה מְהַלֵּךְ עִם אַבְרָהָם אָבִינוּ וְלָמַד מִמַּעֲשָׂיו וּמִדְּרָכָיו הַטּוֹבִים. אָמְרוּ מֶה עָשָׂה אַבְרָהָם אָבִינוּ, עָשָׂה לוֹ(נ) בַּיִת קֳבֵל(נא) לְחָרָן, וְכָל מִי שֶׁהוּא נִכְנָס וְיוֹצֵא לְחָרָן הָיָה מְקַבְּלוֹ וְנוֹתֵן לוֹ מַאֲכָלוֹ וּמַשְׁקֵהוּ,(נב) וְאוֹמֵר לוֹ אֶחָד הוּא הָאֱלֹהִים בָּעוֹלָם.(נג) וּכְשֶׁבָּא לוֹט לִסְדוֹם כָּךְ הָיָה עוֹשֶׂה. וְכֵיוָן שֶׁהִכְרִיזוּ אַנְשֵׁי סְדוֹם וְאָמְרוּ כָּל מִי שֶׁהוּא מַחֲזִיק יָדוֹ בְּפַת לֶחֶם לֶעָנִי וּלְאֶבְיוֹן יִשָׂרֵף בָּאֵשׁ,(נד) הָיָה מִתְיָרֵא מִלַּעֲשׂוֹת בַּיּוֹם וְעָשָׂה בַּלָּיְלָה. רָאָה שְׁנֵי הַמַּלְאָכִים,(נה) וְהָיָה סָבוּר שֶׁהֵן אוֹרְחֵי הָאָרֶץ(נו) וְרָץ לִקְרָאתָם, שֶׁנֶּאֱמַר [בראשית יט, א] וַיָּבֹאוּ שְׁנֵי הַמַּלְאָכִים סְדֹמָה וְגוֹ'. וְלָמָּה הָיָה יוֹשֵׁב לוֹט(נז) בְּשַׁעַר סְדוֹם, אֶלָּא שֶׁאוֹתוֹ הַיּוֹם(נח) מִנוּהוּ עֲלֵיהֶם.(נט) הֱבִיאָם וְאָמַר לָהֶם(ס) בֹּאוּ וְלִינוּ בְּתוֹךְ בֵּיתִי וְאִכְלוּ וּשְׁתוּ(סא) וְתֵלְכוּ לְדַרְכֵיכֶם בְּשָׁלוֹם.(סב) וְלֹא קִבְּלוּ הָאֲנָשִׁים עֲלֵיהֶם,(סג) וְהֶחֱזִיק בְּיָדָם שֶׁלֹּא בִרְצוֹנָם וְהִכְנִיס אוֹתָם בְּתוֹךְ בֵּיתוֹ, שֶׁנֶּאֱמַר [שם ג] וַיִּפְצַר בָּם מְאֹד.(סד) רָאָה אוֹתָם נַעַר אֶחָד מֵאַנְשֵׁי סְדוֹם וְיָצָא וְהִגִּיד לָהֶם,(סה) וְנִתְקַבְּצוּ אֵלָיו כֻּלָּם אֶל פֶּתַח הַבַּיִת(סו) לַעֲשׂוֹת כְּמִנְהָגָם. אָמְרוּ אֶל לוֹט אַיֵּה הָאֲנָשִׁים אֲשֶׁר בָּאוּ אֵלֶיךָ הַלָּיְלָה הוֹצִיאֵם אֵלֵינוּ וְנֵדְעָה אֹתָם [שם ה], רוֹצִים אָנוּ לַעֲשׂוֹת כְּמִנְהָגֵנוּ. מֶה עָשָׂה לוֹט,(סז) כְּשֵׁם שֶׁנָּתַן משֶׁה רַבֵּנוּ נַפְשׁוֹ עַל יִשְׂרָאֵל,(סח) כָּךְ נָתַן לוֹט נַפְשׁוֹ עֲלֵיהֶם,(סט) לָקַח שְׁתֵּי בְּנוֹתָיו וְהוֹצִיא לָהֶם תַּחַת שְׁנֵי הַמַּלְאָכִים, שֶׁנֶּאֱמַר [שם, ח] הִנֵּה נָא לִי שְׁתֵּי בָנוֹת,(ע) וְלֹא קִבְּלוּ הָאֲנָשִׁים. מֶה עָשׂוּ לָהֶם הַמַּלְאָכִים,(עא) הִכּוּ אוֹתָם בָּעִוָּרוֹן עַד שֶׁעָלָה עַמּוּד הַשָּׁחַר. כֵּיוָן שֶׁעָלָה עַמּוּד הַשַּׁחַר, הַכֹּל בְּמִדָּה,(עב) כְּשֵׁם שֶׁהֶחֱזִיק לָהֶם בְּיָדָם שֶׁלֹּא כִרְצוֹנָם וְהִכְנִיסָם בְּתוֹךְ בֵּיתוֹ, כָּךְ הֶחֱזִיקוּ בְיָדוֹ וּבְיַד אִשְׁתּוֹ וּבְיַד שְׁתֵּי בְנוֹתָיו וְהוֹצִיאוּם בַּחוּץ.(עג) וְאָמַר לָהֶם אַל תַּבִּיטוּ אַחֲרֵיכֶם,(עד) שֶׁהֲרֵי יָרַד שְׁכִינָתוֹ שֶׁל הַקָּדוֹשׁ בָּרוּךְ הוּא שֶׁיַּמְטִיר עַל סְדוֹם וְעַל עֲמוֹרָה גָּפְרִית וָאֵשׁ.(עה) עִדִּית אִשְׁתּוֹ שֶׁל לוֹט(עו) נִכְמְרוּ רַחֲמֶיהָ עַל בְּנוֹתֶיהָ(עז) הַנְשׂוּאוֹת בִּסְדוֹם, וְהִבִּיטָה אַחֲרֶיהָ לִרְאוֹת אִם הָיוּ הוֹלְכוֹת אַחֲרֶיהָ אִם לָאו.(עח) וְרָאֲתָה אַחֲרֶיהָ הַשְּׁכִינָה,(עט) וְנַעֲשִׂית נְצִיב מֶלַח,(פ) וַעֲדַיִן הִיא עוֹמֶדֶת, כָּל הַיּוֹם לוֹחֲכִים אוֹתָהּ הַשְּׁוָרִים(פא) וְיוֹרֶדֶת עַל רַגְלֶיהָ,(פב) וּבַבֹּקֶר צוֹמַחַת, שֶׁנֶּאֱמַר [שם כו] וַתַּבֵּט אִשְׁתּוֹ מֵאַחֲרָיו וַתְּהִי נְצִיב מֶלַח.

נִסָּיוֹן הַשְּׁמִינִי, וַיְהִי אַבְרָם בֶּן תִּשְׁעִים שָׁנָה וְתֵשַׁע שָׁנִים [בראשית יז, א]. אָמַר לוֹ הַקָּדוֹשׁ בָּרוּךְ הוּא, הִתְהַלֵּךְ לְפָנַי וֶהְיֵה תָמִים [שם א]. אָמַר לוֹ הַקָּדוֹשׁ בָּרוּךְ הוּא, עַד עַכְשָׁו לֹא הָיִיתָ תָּמִים לְפָנַי, אֶלָּא מֹל בְּשַׂר עָרְלָתְךָ מֵעָלֶיךָ וְהִתְהַלֵּךְ לְפָנַי וֶהְיֵה תָמִים, שֶׁהָעָרְלָה טְמֵאָה הִיא מִכָּל הַטֻּמְאוֹת, שֶׁנֶּאֱמַר [ישעיה נב, א] לֹא יוֹסִיף יָבֹא בָךְ עוֹד עָרֵל וְטָמֵא. שֶׁהָעָרְלָה מוּם הִיא מִכָּל מוּמִים. מֹל אֶת בְּשַׂר עָרְלָתְךָ, וְהִתְהַלֵּךְ לְפָנַי וֶהְיֵה תָמִים. רַבָּן גַּמְלִיאֵל אוֹמֵר,(א) שָׁלַח וְקָרָא לְשֵׁם בֶּן נֹחַ, וּמָל אֶת בְּשַׂר עָרְלָתוֹ(ב) וּבְשַׂר עָרְלַת יִשְׁמָעֵאל בְּנוֹ, שֶׁנֶּאֱמַר [שם כו] בְּעֶצֶם הַיּוֹם הַזֶּה נִמּוֹל אַבְרָהָם וְיִשְׁמָעֵאל בְּנוֹ. מַהוּ(ג) בְּעֶצֶם הַיּוֹם הַזֶּה, בִּגְבוּרַת הַשֶּׁמֶשׁ בַּחֲצִי הַיּוֹם.(ד) וְלֹא עוֹד אֶלָּא אַתְיָא עֶצֶם עֶצֶם מִיּוֹם הַכִּפּוּרִים. מַה לְּהַלָּן וְכָל מְלָאכָה לֹא תַעֲשׂוּ בְּעֶצֶם הַיּוֹם הַזֶּה כִּי יוֹם כִּפֻּרִים הוּא [ויקרא כג, כח],(ה) שֶׁבְּיוֹם הַכִּפּוּרִים נִמּוֹל אַבְרָהָם. וּכָל שָׁנָה וְשָׁנָה הַקָּדוֹשׁ בָּרוּךְ הוּא(ו) רוֹאֶה דַּם הַבְּרִית שֶׁל מִילָה שֶׁל אַבְרָהָם אָבִינוּ וּמְכַפֵּר עַל כָּל עֲוֹנוֹתֵינוּ,(ז) שֶׁנֶּאֱמַר [שם טז, ל] כִּי בַיּוֹם הַזֶּה יְכַפֵּר עֲלֵיכֶם לְטַהֵר אֶתְכֶם וְגוֹ'.(ח) וּבְאוֹתוֹ מָקוֹם שֶׁנִּמּוֹל(ט) וְנִשְׁאַר דָּמוֹ, שָׁם נִבְנָה הַמִּזְבֵּחַ.(י) וּלְכָךְ נֶאֱמַר וְאֶת כָּל דָּמוֹ יִשְׁפֹּךְ אֶל יְסוֹד הַמִּזְבֵּחַ [שם ד, לד],(יא) וָאֹמַר לָךְ בְּדָמַיִךְ חֲיִי וָאֹמַר לָךְ בְּדָמַיִךְ חֲיִי [יחזקאל טז, ו]. רַבִּי חֲנִינָא בֶּן דּוֹסָא אוֹמֵר, כָּל הַנִּמּוֹלִים(יב) בַּיּוֹם הַשְּׁלִישִׁי כְּאֵב הוּא לָהֶם, שֶׁנֶּאֱמַר [בראשית לד, כה] וַיְהִי בַיוֹם הַשְּׁלִישִׁי בִּהְיוֹתָם כֹּאֲבִים.(יג) וּלְכָךְ שָׁנוּ חֲכָמִים מַרְחִיצִין אֶת הַקָּטָן בַּשְּׁלִישִׁי שֶׁלוֹ שֶׁחָל לִהְיוֹת בַּשַּׁבָּת,(יד) וְכָל צָרְכֵי מִילָה עוֹשִׂין בַּשַּׁבָּת. וְכָל עָרֵל לֹא יֹאכַל בּוֹ [שמות יב, מח] וְלֹא יִגַּע בַּמִּקְדָּשׁ, הַפּוֹרֵשׁ מִן הָעָרְלָה כְּפוֹרֵשׁ מִן הַקֶּבֶר: רַבָּן גַּמְלִיאֵל בְּנוֹ שֶׁל רַבִּי יְהוּדָה הַנָּשִׂיא אוֹמֵר, בְּיוֹם שֶׁנִּמּוֹל אַבְרָהָם בַּיּוֹם הַשְּׁלִישִׁי כְּאֵב הָיָה לוֹ מְאֹד(טו) כְּדֵי לְנַסּוֹתוֹ. מֶה עָשָׂה הַקָּדוֹשׁ בָּרוּךְ הוּא,(טז) נָקַב חֹר אֶחָד בַּגֵּיהִנֹּם, וְהִרְתִּיחַ אֶת הַיּוֹם(יז) כְּיוֹמָן שֶׁל רְשָׁעִים,(יח) וְיָצָא וְיָשַׁב לוֹ בְּפֶתַח הָאֹהֶל לְרוּחַ הַיּוֹם, שֶׁנֶּאֱמַר [בראשית יח, א] וְהוּא ישֵׁב פֶּתַח הָאֹהֶל כְּחֹם הַיּוֹם. אָמַר הַקָּדוֹשׁ בָּרוּךְ הוּא לְמַלְאֲכֵי הַשָּׁרֵת, בּוֹאוּ וּנְבַקֵּר אֶת הַחוֹלֶה,(יט) שֶׁמִּדַּת גְּמִילוּת חֲסָדִים גְּדוֹלָה לְפָנָי, מִיָּד יָרְדוּ הַמַּלְאָכִים וּבִקְּרוּ אֶת אַבְרָהָם.(כ) בּוֹא וּרְאֵה כֹּחָהּ שֶׁל מִילָה, שֶׁעַד שֶׁלֹּא נִמּוֹל אַבְרָהָם(כא) הָיָה נוֹפֵל וְאַחַר כָּךְ אֲנִי מְדַבֵּר עִמּוֹ, שֶׁנֶּאֱמַר [שם יז, יז] וַיִּפֹּל אַבְרָהָם עַל פָּנָיו. עַכְשָׁו שֶׁנִּמּוֹל הוּא יוֹשֵׁב וַאֲנִי עוֹמֵד, שֶׁנֶּאֱמַר [שם יח, א] וְהוּא ישֵׁב פֶּתַח הָאֹהֶל וְגוֹ',(כב) וּכְתִיב [שם ב] וַיִּשָׂא עֵינָיו וַיַּרְא וְהִנֵּה שְׁלשָׁה אֲנָשִׁים נִצָּבִים עָלָיו: רַבִּי זְעִירָא אוֹמֵר,(כג) חָמֵשׁ עֲרָלוֹת בָּעוֹלָם, אַרְבַּע בָּאָדָם וְאַחַת בָּאִילָן. אַרְבַּע בָּאָדָם מִנַּיִן, עָרְלַת הָאֹזֶן, שֶׁנֶּאֱמַר [ירמיה ו, י] הִנֵּה עֲרֵלָה אָזְנָם. עָרְלַת שְׂפָתַיִם, שֶׁנֶּאֱמַר [שמות ו, יב] וַאֲנִי עֲרַל שְׂפָתָיִם. עָרְלַת הַלֵּב, שֶׁנֶּאֱמַר [דברים י, טז] וּמַלְתֶּם אֶת עָרְלַת לְבַבְכֶם.(כד) עָרְלַת בָּשָׂר מִנַּיִן, שֶׁנֶּאֱמַר [בראשית יז, יד] וְעָרֵל זָכָר אֲשֶׁר לֹא יִמּוֹל אֶת בְּשַׂר עָרְלָתוֹ. וְאוֹמֵר, כִּי כָל הַגּוֹיִם עֲרֵלִים וְכָל בֵּית יִשְׂרָאֵל עַרְלֵי לֵב [ירמיה ט, כה].(כה) וְעָרְלַת הַלֵּב אֵינָהּ מַנַּחַת לַעֲשׂוֹת רְצוֹן בּוֹרְאוֹ.(כו) וּלֶעָתִיד לָבֹא הַקָּדוֹשׁ בָּרוּךְ הוּא מֵסִיר אֶת עָרְלַת הַלֵּב מֵעֲלֵיהֶן וְאֵינָן מַקְשִׁין אֶת עָרְפָּם לִפְנֵי בוֹרְאָם, שֶׁנֶּאֱמַר [יחזקאל לו, כו] וַהֲסִרֹתִי אֶת לֵב הָאֶבֶן מִבְּשַׂרְכֶם וְנָתַתִּי לָכֶם לֵב בָּשָׂר.(כז) וְאוֹמֵר [דברים י, טז] וּמַלְתֶּם אֶת עָרְלַת לְבַבְכֶם. אַחַת בָּאִילָן, שֶׁנֶּאֱמַר [ויקרא יט, כג] וְכִי תָבֹאוּ אֶל הָאָרֶץ וּנְטַעְתֶּם כָּל עֵץ מַאֲכָל וַעֲרַלְתֶּם עָרְלָתוֹ אֶת פִּרְיוֹ. רַבִּי זְרִיקָא אוֹמֵר,(כח) אֵין עֵץ הָאָמוּר כָּאן אֶלָּא עֵץ הַגָּפֶן. אִם אֵינָן כּוֹרְתִין אֶת עֵץ עָרְלָתוֹ, כָּל הַפֵּרוֹת שֶׁהוּא עוֹשֶׂה עוֹלְלוֹת קְטוּפִין וְלֹא טוֹבִים לַמַּרְאֶה, וְנִפְסָל יֵינוֹ מֵעַל גַּבֵּי הַמִּזְבֵּחַ. וְאִם כּוֹרְתִין עֵץ עָרְלָתוֹ, כָּל פֵּרוֹת שֶׁהוּא עוֹשֶׂה טוֹבִים לַמַּרְאֶה, וְנִבְחָר יֵינָם לְהַקְרִיב עַל גַּבֵּי הַמִּזְבֵּחַ. כָּךְ אָבִינוּ אַבְרָהָם, עַד שֶׁלֹּא נִמּוֹל, הַפְּרִי שֶׁעָשָׂה לֹא הָיָה טוֹב בְּמַעֲשָׂיו וְנִפְסַל מֵעַל גַּבֵּי הַמִּזְבֵּחַ, וּכְשֶּׁנִּמּוֹל הַפְּרִי שֶׁעָשָׂה הָיָה טוֹב בְּמַעֲשָׂיו וְנִבְחַר יֵינוֹ עַל גַּבֵּי הַמִּזְבֵּחַ כְּיֵין נֶסֶךְ, שֶׁנֶּאֱמַר [במדבר טו, ז] וְיַיִן לַנֶּסֶךְ: רַבִּי יוֹחָנָן אוֹמֵר, כָּל מַכְחִישֵׁי הַשֵּׁם הַבָּאִים בְּיִשְׂרָאֵל נִמּוֹלִים בְּטוֹבָתָם וּבִרְצוֹנָם, וּבְיִרְאַת שָׁמַיִם הֵם נִמּוֹלִים, וְאַף עַל פִּי כֵן אֵין מַאֲמִינִים בָּהֶם עַד שִׁבְעָה דוֹרוֹת, שֶׁלֹּא יָשׁוּבוּ הַמַּיִם לְמוֹצָאֵיהֶם. אֲבָל עֲבָדִים נִמּוֹלִים שֶׁלֹּא בְּטוֹבָתָם וְשֶׁלֹּא בִרְצוֹנָם, וְאֵין אֱמוּנָה בַּעֲבָדִים. כָּךְ כָּל עֲבָדִים שֶׁנִּמּוֹלוּ עִם אַבְרָהָם אָבִינוּ, לֹא נִתְקַיְּמוּ לֹא הֵם וְלֹא זַרְעָם בְּיִשְׂרָאֵל. וּמִנַּיִן שֶׁמָּלָן, שֶׁנֶּאֱמַר [בראשית יז, כז] וְכָל אַנְשֵׁי בֵיתוֹ יְלִיד בַּיִת נִמֹּלוּ. וְלָמָּה מָלָן, בִּשְׁבִיל הַטָּהֳרָה, שֶׁלֹּא יְטַמְּאוּ אֶת אֲדוֹנֵיהֶם בְּמַאֲכָלֵיהֶם וּבְמִשְׁתֵּיהֶם. [שֶׁכָּל מִי שֶׁאוֹכֵל עִם הֶעָרֵל, כְּאִלּוּ אֹכֵל עִם הַכֶּלֶב, מָה הַכֶּלֶב שֶׁלֹּא נִמּוֹל כָּךְ הָעֶבֶד שֶׁלֹּא נִמּוֹל. וְכָל הַנּוֹגֵעַ בְּעָרֵל כְּנוֹגֵעַ בַּמֵּת. וְכָל הָרוֹחֵץ עִמּוֹ כְּרוֹחֵץ עִם הַמְּצֹרָע, שֶׁהֵם בְּחַיֵּיהֶם כַּמֵּתִים וּבְמוֹתָם כְּנִבְלַת הַשָׂדֶה. וְאֵין תְּפִלָּתָן נִכְנֶסֶת לִפְנֵי הַקָּדוֹשׁ בָּרוּךְ הוּא, וַעֲלֵיהֶם הוּא אוֹמֵר [תהלים קטו, יז] לֹא הַמֵּתִים יְהַלְלוּ יָהּ. אֲבָל יִשְׂרָאֵל שֶׁהֵן נִמּוֹלִין, תְּפִלָּתָן נִכְנֶסֶת לִפְנֵי הַקָּדוֹשׁ בָּרוּךְ הוּא. וַעֲלֵיהֶם הוּא אוֹמֵר [שם יח] וַאֲנַחְנוּ נְבָרֵךְ יָהּ מֵעַתָּה וְעַד עוֹלָם הַלְלוּיָהּ. רַבִּי אוֹמֵר, יִצְחָק מָל אֶת יַעֲקֹב, וְעֵשָׂו מָאַס בַּמִּילָה כְּשֵׁם שֶׁמָּאַס אֶת הַבְּכוֹרָה, שֶׁנֶּאֱמַר [בראשית כה, לד] וַיִּבֶז עֵשָׂו אֶת הַבְּכֹרָה. וְיַעֲקֹב דָּבַק בִּבְרִית מִילָה וּמָל אֶת בָּנָיו וְאֶת בְּנֵי בָנָיו. וּמִנַּיִן שֶׁהָיוּ בְּנֵי יַעֲקֹב נִמּוֹלִין, שֶׁנֶּאֱמַר [שם לד, כב] אַךְ בְּזֹאת(כט) יֵאֹתוּ לָנוּ הָאֲנָשִׁים וְגוֹ'. מִכָּאן אַתָּה לָמֵד שֶׁבְּנֵי יַעֲקֹב הָיוּ נִמּוֹלִין, וּמָלוּ לִבְנֵיהֶם(ל) וְהִנְחִילוּהָ לְחָק עוֹלָם, עַד שֶׁעָמַד(לא) פַּרְעֹה וְגָזַר עֲלֵיהֶם גְּזֵרוֹת קָשׁוֹת וּמָנַע מֵהֶם בְּרִית מִילָה. וּבְיוֹם שֶׁיָּצְאוּ יִשְׂרָאֵל מִמִּצְרַיִם נִמּוֹלוּ כֻלָּם(לב) מִגָּדוֹל וְעַד קָטָן, שֶׁנֶּאֱמַר [יהושע ה, ה] כִּי מֻלִּים הָיוּ כָּל הָעָם הַיֹּצְאִים וְגוֹ'.(לג) וְהָיוּ לוֹקְחִים דַּם מִילָה וְדַם פֶּסַח, וְהָיוּ נוֹתְנִין(לד) עַל מַשְׁקוֹף בָּתֵּיהֶן, וּכְשֶׁעָבַר הַקָּדוֹשׁ בָּרוּךְ הוּא לִנְגֹּף אֶת מִצְרַיִם(לה) וְרָאָה אֶת דַּם הַבְּרִית וְדַם הַפֶּסַח, נִתְמַלֵּא רַחֲמִים עַל יִשְׂרָאֵל,(לו) שֶׁנֶּאֱמַר [יחזקאל טז, ו] וָאֶעֱבֹר עָלַיִךְ וָאֶרְאֵךְ מִתְבּוֹסֶסֶת בְּדָמָיִךְ וָאֹמַר לָךְ בְּדָמַיִךְ חֲיִי וָאֹמַר לָךְ בְּדָמַיִךְ חֲיִי.(לז) אָמַר רַבִּי אֱלִיעֶזֶר, וְכִי מָה רָאָה הַכָּתוּב לוֹמַר שְׁתֵּי פְעָמִים בְּדָמַיִךְ חֲיִי. אֶלָּא אָמַר הַקָּדוֹשׁ בָּרוּךְ הוּא בִּזְכוּת דַּם בְּרִית מִילָה וְדַם פֶּסַח גָּאַלְתִּי אֶתְכֶם מִמִּצְרַיִם,(לח) וּבִזְכוּתָם אַתֶּם עֲתִידִין לִגָּאֵל בְּסוֹף מַלְכוּת רְבִיעִית, לְכָךְ נֶאֱמַר שְׁתֵּי פְעָמִים בְּדָמַיִךְ חֲיִי. רַבִּי יִשְׁמָעֵאל אוֹמֵר,(לט) לֹא עִכֵּב אַבְרָהָם מִכֹּל אֲשֶׁר צִוָּהוּ. וּכְשֶׁנּוֹלַד יִצְחָק, בֶּן שְׁמוֹנָה יָמִים(מ) הִגִּישׁוֹ לַמִּילָה, שֶׁנֶּאֱמַר [בראשית כא, ד] וַיָּמָל אַבְרָהָם אֶת יִצְחָק בְּנוֹ בֶּן שְׁמֹנַת יָמִים.(מא) וְהִגִּישׁוֹ לְמִנְחָה עַל גַּבֵּי הַמִּזְבֵּחַ,(מב) וְעָשָׂה שִׂמְחָה וּמִשְׁתֶּה. מִכָּאן אָמְרוּ חֲכָמִים(מג) חַיָּב אָדָם לַעֲשׂוֹת שִׂמְחָה וּמִשְׁתֶּה בְּאוֹתוֹ הַיּוֹם שֶׁזָּכָה לָמוּל אֶת בְּנוֹ כְּאַבְרָהָם אָבִינוּ,(מד) שֶׁנֶּאֱמַר [שם ח] וַיַּעַשׂ אַבְרָהָם מִשְׁתֶּה גָדוֹל בְּיוֹם הִגָּמֵל אֶת יִצְחָק:(מה) שְׁלשָׁה עִנּוּיִין הֵן, עִנּוּי הַצּוֹם, וְעִנּוּי בֵּית הָאֲסוּרִין, וְעִנּוּי הַדֶּרֶךְ. עִנּוּי הַצּוֹם מִנַּיִן, שֶׁנֶּאֱמַר [תהלים לה, יג] עִנֵּיתִי בַצּוֹם נַפְשִׁי. עִנּוּי בֵּית הָאֲסוּרִין מִנַּיִן, שֶׁנֶּאֱמַר [שם קה, יח] עִנּוּ בַכֶּבֶל רַגְלוֹ. עִנּוּי הַדֶּרֶךְ מִנַּיִן, שֶׁנֶּאֱמַר [שם קב, כד] עִנָּה בַדֶּרֶךְ כֹּחִי.(מו) וּמֵעִנּוּי הַדֶּרֶךְ לֹא מָלוּ בְּנֵי יִשְׂרָאֵל, שֶׁנֶּאֱמַר [יהושע ה, ז] כִּי לֹא מָלוּ אוֹתָם בַּדָּרֶךְ.(מז) רַבִּי יִשְׁמָעֵאל אוֹמֵר, וְכִי יִשְׂרָאֵל שֶׁלֹא נִמוֹלוּ שָׁמְעוּ קוֹלוֹ שֶׁל הַקָּדוֹשׁ בָּרוּךְ הוּא עַל הַר סִינַי וְנָתַן לָהֶם אֶת הַתּוֹרָה. אֶלָּא חַס וְשָׁלוֹם(מח) נִמּוֹלִין הָיוּ, אֶלָּא(מט) שֶׁלֹּא הָיָה לָהֶם פְּרִיעָה, וְכָל מִי שֶׁמָּל וְלֹא פָרַע אֶת הַמִּילָה כְּאִלּוּ לֹא מָל. וּכְשֶׁבָּאוּ יִשְׂרָאֵל לְאֶרֶץ יִשְׂרָאֵל, אָמַר הַקָּדוֹשׁ בָּרוּךְ הוּא לִיהוֹשֻׁעַ, אִי אַתָּה יוֹדֵעַ שֶׁאֵין יִשְׂרָאֵל נִמּוֹלִין כְּתִקְנָן חֲזֹר(נ) וּמֹל אוֹתָם פַּעַם שְׁנִיָּה, שֶׁנֶּאֱמַר [שם ב] מֹל אֶת בְּנֵי יִשְׂרָאֵל שֵׁנִית. וְקִבֵּץ כָּל הָעֲרָלוֹת עַד שֶׁעָשָׂה אוֹתָם כַּגִּבְעָה, שֶׁנֶּאֱמַר [שם ג] וַיָּמָל אֶת בְּנֵי יִשְׂרָאֵל אֶל גִּבְעַת הָעֲרָלוֹת.(נא) וְהָיוּ יִשְׂרָאֵל לוֹקְחִין אֶת הָעָרְלָה וְאֶת הַדָּם(נב) וּמְכַסִּין אוֹתָן בַּעֲפַר הַמִּדְבָּר. וּכְשֶׁבָּא בִלְעָם הַקּוֹסֵם רָאָה אֶת כָּל הַמִּדְבָּר מָלֵא מֵעָרְלָתָן שֶׁל יִשְׂרָאֵל, אָמַר מִי יוּכַל לַעֲמֹד בִּזְכוּת בְּרִית דַּם הַמִּילָה שֶׁהוּא מְכֻסֶּה בֶּעָפָר, שֶׁנֶּאֱמַר [במדבר כג, י] מִי מָנָה עֲפַר יַעֲקֹב. מִכָּאן אָמְרוּ חֲכָמִים מְכַסִּין אֶת עָרְלַת הַדָּם בַּעֲפַר הָאָרֶץ.(נג) וְלֹא עוֹד אֶלָּא שֶׁנִמְשְׁלוּ לֶעָפָר, שֶׁנֶּאֱמַר [בראשית כח, יד] וְהָיָה זַרְעֲךָ כַּעֲפַר הָאָרֶץ.(נד) וְכָךְ הָיוּ יִשְׂרָאֵל נְהוּגִין לָמוּל,(נה) עַד שֶׁנֶּחְלְקוּ לִשְׁתֵּי מַמְלָכוֹת, וּמַלְכוּת אֶפְרַיִם מָנְעוּ מֵהֶם אֶת הַמִּילָה. וְעָמַד אֵלִיָּהוּ זָכוּר לְטוֹב וְקִנֵּא קִנְאָה גְדוֹלָה, וְנִשְׁבַּע עַל הַשָּׁמַיִם שֶׁלֹּא לְהוֹרִיד טַל וּמָטָר עַל הָאָרֶץ, וְשָׁמְעָה אִיזֶבֶל וּבִקְשָׁה לַהֲרֹג אוֹתוֹ. עָמַד אֵלִיָּהוּ(נו) וְהָיָה מִתְפַּלֵּל לִפְנֵי הַקָּדוֹשׁ בָּרוּךְ הוּא.(נז) אָמַר לוֹ הַקָּדוֹשׁ בָּרוּךְ הוּא, טוֹב אַתָּה מֵאֲבוֹתֶיךָ, עֵשָׂו בִּקֵּשׁ אֶת יַעֲקֹב לְהָרְגוֹ, שֶׁנֶּאֱמַר [שם כז, מא] יִקְרְבוּ יְמֵי אֵבֶל אָבִי וְאַהַרְגָה אֶת יַעֲקֹב אָחִי. וּבָרַח [נ"א: יעקב ברח] מִלְּפָנָיו וְנִמְלַט,(נח) שֶׁנֶּאֱמַר [הושע יב, יג] וַיִּבְרַח יַעֲקֹב שְׂדֵה אֲרָם. פַּרְעֹה בִקֵּשׁ לַהֲרֹג אֶת משֶׁה, [נ"א: משה ברח] וּבָרַח מִלְּפָנָיו וְנִמְלַט, שֶׁנֶּאֱמַר [שמות ב, טו] וַיִּבְרַח משֶׁה מִפְּנֵי פַרְעֹה. שָׁאוּל בִּקֵּשׁ לַהֲרֹג אֶת דָּוִד, וּבָרַח [נ"א: דוד ברח] מִלְּפָנָיו וְנִמְלַט, שֶׁנֶּאֱמַר [ש"א יט, י] וְדָוִד בָּרַח וַיִּמָּלֵט. לְלַמֶּדְךָ שֶׁכָּל מִי שֶׁהוּא בוֹרֵחַ נִמְלָט. עָמַד אֵלִיָּהוּ(נט) וּבָרַח מֵאֶרֶץ יִשְׂרָאֵל וְנִמְלַט, שֶׁנֶּאֱמַר [מ"א יט, ח] וַיָּקָם וַיֹּאכַל וַיִּשְׁתֶּה. נִגְלָה עָלָיו הַקָּדוֹשׁ בָּרוּךְ הוּא(ס) וְאָמַר לוֹ מַה לְּךָ פֹה אֵלִיָּהוּ [שם ט], אָמַר לוֹ קַנֹּא קִנֵּאתִי [שם י]. אָמַר לוֹ הַקָּדוֹשׁ בָּרוּךְ הוּא,(סא) לְעוֹלָם אַתָּה מְקַנֵּא,(סב) קִנֵּאתָ בַּשִּׁטִּים עַל(סג) גִּלּוּי עֲרָיוֹת, שֶׁנֶּאֱמַר [במדבר כה, יא] פִּינְחָס בֶּן אֶלְעָזָר וְגוֹ'. וְכָאן אַתָּה מְקַנֵּא.(סד) חַיֶּיךָ, שֶׁאֵין יִשְׂרָאֵל עוֹשִׂין בְּרִית מִילָה עַד שֶׁאַתָּה רוֹאֶה בְּעֵינֶיךָ.(סה) מִכָּאן הִתְקִינוּ חֲכָמִים שֶׁיִּהְיוּ עוֹשִׂין(סו) מוֹשַׁב כָּבוֹד לְמַלְאַךְ הַבְּרִית,(סז) שֶׁנֶּאֱמַר [מלאכי ג, א] וּמַלְאַךְ הַבְּרִית אֲשֶׁר אַתֶּם חֲפֵצִים הִנֵּה בָא וְגוֹ'. אֱלֹהֵי יִשְׂרָאֵל יָחִישׁ וְיָבִיא(סח) בְחַיֵּינוּ מָשִׁיחַ לְנַחֲמֵנוּ,(סט) וִיחַדֵּשׁ לְבָבֵנוּ, שֶׁנֶּאֱמַר, וְהֵשִׁיב לֵב אָבוֹת עַל בָּנִים וְגוֹ' [שם כד].

כְּתִיב [בראשית כו, יב] וַיִּזְרַע יִצְחָק בָּאָרֶץ הַהִוא, רַבִּי אֱלִיעֶזֶר אוֹמֵר,(א) וְכִי יִצְחָק זָרַע דָּגָן חַס וְשָׁלוֹם, אֶלָּא(ב) לָקַח אֶת כָּל מַעֲשַׂר מָמוֹנוֹ(ג) וְזָרַע צְדָקָה לַעֲנִיִּים, כְּמָה דְאַתְּ אָמַר זִרְעוּ לָכֶם לִצְדָקָה [הושע י, יב].(ד) וְכָל דָּבָר שֶׁעִשֵׂר הֵבִיא לוֹ הַקָּדוֹשׁ בָּרוּךְ הוּא מֵאָה שְׁעָרִים שֶׁל מָמוֹן וּבֵרֲכוֹ, לְכָךְ נֶאֱמַר [בראשית כו, יב] וַיְבָרְכֵהוּ ה': רַבִּי שִׁמְעוֹן אוֹמֵר,(ה) מִכֹּחַ צְדָקָה הַמֵּתִים עֲתִידִין לִחְיוֹת. וּמִנַּיִן(ו) אָנוּ לְמֵדִין, מֵאֵלִיָּהוּ הַתִּשְׁבִּי(ז) שֶׁהָיָה הוֹלֵךְ מֵהַר אֶל הַר וּמִמְּעָרָה אֶל מְעָרָה,(ח) הָלַךְ לוֹ לְצָרְפַת(ט) וְקִבְּלַתּוּ אִשָּׁה אַלְמָנָה בְּכָבוֹד גָּדוֹל,(י) אִמּוֹ שֶׁל יוֹנָה הָיְתָה,(יא) וּמִפִּתָּהּ וּמִשַּׁמְנָהּ הָיוּ אוֹכְלִים וְשׁוֹתִים(יב) הוּא וְהִיא וּבְנָהּ, שֶׁנֶּאֱמַר [מ"א יז, טו] וַתֹּאכַל הוּא וָהִיא.(יג) אָמַר רַבִּי לֵוִי, הוּא וָהִיא כְּתִיב,(יד) וְהִיא וָהוּא קָרֵינָן, בִּזְכוּת אֵלִיָּהוּ אָכְלוּ.(טו) לְאַחַר יָמִים(טז) חָלָה בֶּן הָאִשָּׁה וָמֵת, שֶׁנֶּאֱמַר [שם יז] וַיְהִי אַחַר הַדְּבָרִים הָאֵלֶּה חָלָה בֶּן הָאִשָּׁה.(יז) אָמְרָה לוֹ הָאִשָּׁה, יֹאמְרוּ כִּי בָאתָ אֵלַי בְּתַשְׁמִישׁ הַמִּטָּה וְהִזְכַּרְתָּ עֲוֹנִי עָלַי וּמֵת בְּנִי,(יח) אֶלָּא טֹל מַה שֶּׁהֵבֵאתָ אֵלַי וְתֶן לִי אֶת בְּנִי. וְעָמַד אֵלִיָּהוּ וְהִתְפַּלֵּל לִפְנֵי הַקָּדוֹשׁ בָּרוּךְ הוּא וְאָמַר, רִבּוֹנוֹ שֶׁל עוֹלָם,(יט) לֹא דַי לִי כָּל הָרָעוֹת שֶׁעָבְרוּ עָלַי וְעַל רֹאשִׁי, אֶלָּא אַף הָאִשָּׁה הַזֹּאת(כ) שֶׁאֲנִי יוֹדֵעַ שֶׁמִּתּוֹךְ צָרַת בְּנָהּ דִּבְּרָה עָלַי(כא) לְעַקְּשֵׁנִי,(כב) עַכְשָׁו יִלְמְדוּ הַדּוֹרוֹת שֶׁיֵּשׁ תְּחִיַּת הַמֵּתִים הָשֵׁב אֶת נֶפֶשׁ הַיָּלֶד,(כג) וְנֶעְתַּר לוֹ, שֶׁנֶּאֱמַר [שם כב] וַיִּשְׁמַע ה' בְּקוֹל אֵלִיָּהוּ וְגוֹ'. רַבִּי יְהוֹשֻׁעַ בֶּן קָרְחָה אוֹמֵר,(כד) וְעַל זֶה אַתְּ תָּמֵהַּ, בֹּא וּרְאֵה מֵאֱלִישָׁע בֶּן שָׁפָט(כה) שֶׁלֹּא הָיְתָה אִשָּׁה יְכוֹלָה לְהִסְתַּכֵּל בְּפָנָיו שֶׁלֹּא תָמוּת,(כו) וְהָיָה הוֹלֵךְ מֵהַר לְהַר וּמִמְּעָרָה לִמְעָרָה, וְהָלַךְ לְשׁוּנֵם(כז) וְקִבְּלַתּוּ אִשָּׁה גְדוֹלָה בְּכָבוֹד גָּדוֹל,(כח) אֲחוֹתָהּ שֶׁל אֲבִישַׁג הַשּׁוּנַמִּית(כט) אִשְׁתּוֹ שֶׁל עִדּוֹא הַנָּבִיא וְקִבְּלַתּוּ בְּכָבוֹד גָּדוֹל. אָמְרָה הָאִשָּׁה לְבַעְלָהּ אִישׁ הָאֱלֹהִים הַזֶּה(ל) קָדוֹשׁ הוּא, אֵין אִשָּׁה יְכוֹלָה לְהִסְתַּכֵּל בְּפָנָיו שֶׁלֹּא תָמוּת, שֶׁנֶּאֱמַר [מ"ב ד, י] נַעֲשֶׂה נָּא עֲלִיַּת קִיר קְטַנָּה, וְכָל פַּעַם שֶׁהוּא עוֹבֵר יָסוּר אֶל הַעֲלִיָּה.(לא) וּבָנוּ וְהֵכִינוּ וְהִתְקִינוּ.(לב) לְאַחַר יָמִים הָיָה עוֹבֵר אֶל שׁוּנֵם, וַיָּסַר אֶל הָעֲלִיָּה, שֶׁנֶּאֱמַר [שם יא] וַיְהִי הַיּוֹם וַיָּבֹא שָׁמָּה.(לג) וְקָרָא לַשּׁוּנַמִּית, שֶׁנֶּאֱמַר [שם טו] וַיִּקְרָא לָהּ וַתַּעֲמֹד בַּפָּתַח.(לד) וְלָמָּה עָמְדָה בַפֶּתַח, שֶׁלֹּא הָיְתָה יְכוֹלָה לְהִסְתַּכֵּל בְּפָנָיו. אָמַר לָהּ לַמּוֹעֵד הַזֶּה כָּעֵת חַיָּה אַתְּ חֹבֶקֶת בֵּן [שם טז].(לה) אָמְרָה לוֹ אֲדוֹנִי זָקֵן מְאֹד וְחָדְלָה מִמֶּנִּי אֹרַח נָשִׁים, וְאִי אֶפְשָׁר לֵעָשׂוֹת הַדָּבָר הַזֶּה, אַל אֲדֹנִי אִישׁ הָאֱלֹהִים אַל תְּכַזֵּב בְּשִׁפְחָתֶךָ [שם]. רַבִּי זְכַרְיָה אוֹמֵר,(לו) רְצוֹן יְרֵאָיו יַּעֲשֶׂה [תהלים קמה, יט], עָשָׂה הַקָּדוֹשׁ בָּרוּךְ הוּא רְצוֹנוֹ שֶׁל נָבִיא, וְהָרָה וְיָלְדָה,(לז) וְגָדַל הַנַּעַר,(לח) וְיָצָא לְפוּגַת נֶפֶשׁ לִרְאוֹת בַּקּוֹצְרִים,(לט) וְקָרָהוּ אָסוֹן, שֶׁנֶּאֱמַר [מ"ב ד, יח] וַיֵּצֵא אֶל אָבִיו אֶל הַקֹּצְרִים וַיָּמֹת [שם כ].(מ) וְהָלְכָה הָאִשָּׁה לְהַר הַכַּרְמֶל וְנָפְלָה עַל פָּנֶיהָ אַרְצָה לִפְנֵי אֱלִישָׁע, אָמְרָה לוֹ(מא) הַלְוַאי שֶׁהָיְתָה כְּלִי רֵיקָם, אֶלָּא שֶׁנִּתְמַלֵּא וְנִשְׁפַּךְ.(מב) אָמַר הַנָּבִיא, כָּל דָּבָר שֶׁהַקָּדוֹשׁ בָּרוּךְ הוּא עוֹשֶׂה הָיָה מַגִּיד לִי חוּץ מִזּוֹ, שֶׁנֶּאֱמַר [שם כז] וַתָּבֹא אֶל אִישׁ הָאֱלֹהִים וַיִּגַּשׁ גֵּחֲזִי לְהָדְפָהּ, אֲחָזָהּ בְּהוֹד יָפְיָהּ,(מג) מְלַמֵּד שֶׁנָּתַן יָדוֹ עַל גַּבֵּי דַדֶּיהָ. וַיֹּאמֶר אִישׁ הָאֱלֹהִים(מד) הַרְפֵּה לָהּ [שם].(מה) לָקַח אֶת הַמִּשְׁעֶנֶת שֶׁבְּיָדוֹ וּנְתָנָהּ לְגֵחֲזִי, אָמַר לוֹ(מו) אַל תְּדַבֵּר בְּפִיךָ כָּל דָּבָר מְאוּמָה(מז) עַד שֶׁאַתָּה הוֹלֵךְ וְנוֹתֵן אֶת הַמִּשְׁעֶנֶת עַל פְּנֵי הַנַּעַר וְיִחְיֶה.(מח) וְגֵחֲזִי הָיָה כִּשְׂחֹק בְּעֵינָיו, וְכָל אָדָם שֶׁהָיָה פוֹגֵעַ הָיָה אוֹמֵר לוֹ הַתַאֲמִין שֶׁהַמִּשְׁעֶנֶת הַזֹּה מְחַיָּה. לְפִיכָךְ לֹא עָלְתָה בְיָדוֹ,(מט) עַד שֶׁהָלַךְ בְּרַגְלָיו(נ) וְנָתַן פָּנָיו עַל פָּנָיו וְעֵינָיו עַל עֵינָיו, וְהִתְחִיל מִתְפַּלֵּל, רִבּוֹנוֹ שֶׁל עוֹלָם(נא) כְּשֵׁם שֶׁעָשִׂיתָ נִסִּים עַל יְדֵי אֵלִיָּהוּ רַבִּי וְהֶחֱיָה אֶת הַמֵּת, כָּךְ עֲשֵׂה עִמִּי וְהַחֲיֵה הַנַּעַר הַזֶּה.(נב) וְנֶעְתַּר לוֹ, שֶׁנֶּאֱמַר [מ"ב ד, לה] וַיְזוֹרֵר הַנַּעַר עַד שֶׁבַע פְּעָמִים: רַבִּי עֲזַרְיָה אוֹמֵר, תֵּדַע לְךָ כֹּחַ הַצְּדָקָה, בֹּא וּרְאֵה(נג) מִן שַׁלּוּם בֶּן תִּקְוָה,(נד) שֶׁהָיָה מִגְּדוֹלֵי הַדּוֹר(נה) וְהָיָה עוֹשֶׂה צְדָקוֹת בְּכָל יוֹם וָיוֹם. וּמֶה הָיָה עוֹשֶׂה, הָיָה מְמַלֵּא אֶת הַחֵמֶת מַיִם(נו) וְהָיָה יוֹשֵׁב עַל פֶּתַח הָעִיר, וְכָל מִי שֶׁהָיָה בָא מִן הַדֶּרֶךְ הָיָה מַשְׁקֶה(נז) וּמֵשִׁיב אֶת נַפְשׁוֹ עָלָיו.(נח) וּבִזְכוּת הַצְּדָקָה שֶׁעָשָׂה(נט) שָׁרְתָה רוּחַ הַקֹּדֶשׁ עַל אִשְׁתּוֹ.(ס) וַיֵּלֶךְ חִלְקִיָּהוּ הַכֹּהֵן וְגוֹ' אֶל חֻלְדָּה הַנְּבִיאָה אֵשֶׁת שַׁלֻּם בֶּן תִּקְוָה [שם כב, יד].(סא) וּלְמַפְרֵעַ הָיָה שְׁמוֹ בֶּן סַחְרָא,(סב) כְּשֵׁם שֶׁאַתָּה אוֹמֵר כִּי טוֹב סַחְרָהּ מִסְּחַר כָּסֶף [משלי ג, יד].(סג) וּכְשֶׁמֵּת בַּעְלָהּ, חָסְרוּ צִדְקוֹת בַּעְלָהּ,(סד) וְיָצְאוּ כָּל יִשְׂרָאֵל לִגְמֹל חֶסֶד עִם שַׁלֻּם בֶּן תִּקְוָה,(סה) וְרָאוּ אֶת הַגְּדוּד שֶׁבָּא עֲלֵיהֶם,(סו) וְהִשְׁלִיכוּ אֶת הָאִישׁ בְּקֶבֶר אֱלִישָׁע.(סז) הָלַךְ הָאִישׁ וְנָגַע בְּעַצְמוֹת אֱלִישָׁע וְחָיָה,(סח) וְאַחַר כָּךְ הוֹלִיד חֲנַמְאֵל בֶּן שַׁלֻּם: רַבִּי חֲנַנְיָה בֶן תְּרַדְיוֹן אוֹמֵר,(סט) תֵּדַע לְךָ כֹּחַ הַצְּדָקָה, בֹּא וּרְאֵה מִשָּׁאוּל בֶּן קִישׁ(ע) שֶׁהֵסִיר אֶת הָאוֹבוֹת וְאֶת הַיִּדְּעוֹנִים,(עא) וְחָזַר וְאָהַב אֲשֶׁר שָׂנֵא, וְהָלַךְ אֶל עֵין דֹּאר(עב) אֶל אִשָּׁה צְפַנְיָה, אִמּוֹ שֶׁל אַבְנֵר, וּשְׁאָלָהּ בָּאוֹב, וְהֶעֶלְתָה לוֹ שְׁמוּאֵל,(עג) וְעָלוּ עִמּוֹ אֲחֵרִים,(עד) וְכִסְבוּרִין שֶׁהִגִּיעַ תְּחִיַּת הַמֵּתִים, וְרָאֲתָה הָאִשָּׁה(עה) וְנִבְהֲלָה מְאֹד, שֶׁנֶּאֱמַר [ש"א כח, יג] וַיֹּאמֶר לָהּ הַמֶּלֶךְ אַל תִּירְאִי.(עו) וְיֵשׁ אוֹמְרִים צַדִּיקִים הַרְבֵּה כַּיּוֹצֵא בוֹ עָלוּ עִמּוֹ בְּאוֹתָהּ שָׁעָה: רַבִּי אֱלִיעֶזֶר אוֹמֵר,(עז) כָּל הַמֵּתִים עוֹמְדִים בִּתְחִיַּת הַמֵּתִים וְעוֹלִין בִּלְבוּשֵׁיהֶן. מִמִּי אַתָּה לָמֵד,(עח) מִזֶּרַע הָאָרֶץ, קַל וָחֹמֶר(עט) מֵחִטָּה, מָה חִטָּה שֶׁנִקְבְּרָה עֲרֻמָּה יוֹצֵאת בְּכַמָּה לְבוּשִׁין, צַדִּיקִים שֶׁנִּקְבְּרוּ בִּלְבוּשֵׁיהֶן עַל אַחַת כַּמָּה וְכַמָּה. וְלֹא עוֹד אֶלָּא מִמִּי אַתָּה לָמֵד,(פ) מֵחֲנַנְיָה מִישָׁאֵל וַעֲזַרְיָה שֶׁיָּרְדוּ לְתוֹךְ כִּבְשַׁן הָאֵשׁ לְבוּשִׁין וְעָלוּ בְּתַכְרִיכֵיהֶן, שֶׁנֶּאֱמַר [דניאל ג, כז] וּשְׂעַר רֵאשְׁהוֹן לָא הִתְחָרַךְ. וְעוֹד מִמִּי אַתָּה לָמֵד(פא) מִשְּׁמוּאֵל הַנָּבִיא, שֶׁעָלָה עוֹטֶה מְעִיל, שֶׁנֶּאֱמַר [ש"א כח, יד] וַתֹּאמֶר אִישׁ זָקֵן עֹלֶה וְהוּא עֹטֶה מְעִיל: רַבִּי נָתָן אוֹמֵר,(פב) כָּל הַנְּבִיאִים נִתְנַבְּאוּ בְּחַיֵּיהֶם,(פג) וּשְׁמוּאֵל נִתְנַבָּא בְחַיָּיו וְאַחַר מוֹתוֹ, שֶׁאָמַר שְׁמוּאֵל לְשָׁאוּל אִם אַתָּה שׁוֹמֵעַ לַעֲצָתִי לִפּוֹל בַּחֶרֶב תְּהֵא מִיתָתְךָ כַּפָּרָה עָלֶיךָ,(פד) וִיהֵא גוֹרָלְךָ עִמִּי בְּמָקוֹם שֶׁאֲנִי שָׁרוּי. וְשָׁמַע לַעֲצָתוֹ וְנָפַל בַּחֶרֶב הוּא(פה) וְכָל בֵּיתוֹ, שֶׁנֶּאֱמַר [שם לא, ו] וַיָּמָת שָׁאוּל וּשְׁלשֶׁת בָּנָיו וְגוֹ'. לָמָּה, שֶׁיְּהֵא חֶלְקוֹ עִם שְׁמוּאֵל הַנָּבִיא לֶעָתִיד לָבוֹא, שֶׁנֶּאֱמַר [שם כח, יט] וּמָחָר אַתָּה וּבָנֶיךָ עִמִּי.(פו) מַאי עִמִּי, אָמַר רַבִּי יוֹחָנָן, עִמִּי בִּמְחִיצָתִי: הִלֵּל הַזָּקֵן אוֹמֵר, אָמַר לוֹ שְׁמוּאֵל לְשָׁאוּל לֹא דַיְּךָ(פז) שֶׁלֹּא שָׁמַעְתָּ בְקוֹלוֹ וְלֹא עָשִׂיתָ חֲרוֹן אַפּוֹ בַּעֲמָלֵק,(פח) אֶלָּא גַּם בָּאתָה לִשְׁאֹל לְךָ בָּאוֹב. אוֹי לָרוֹעֶה אוֹי לְצֹאנוֹ, שֶׁבִּשְׁבִילְךָ נָתַן הַקָּדוֹשׁ בָּרוּךְ הוּא אֶת יִשְׂרָאֵל בְּיַד פְּלִשְׁתִּים:(צ) רַבִּי יוֹחָנָן אוֹמֵר,(צא) עָלוּ יִשְׂרָאֵל לְבָבֶל וְלֹא הִנִּיחוּ אֶת מַעֲשֵׂיהֶם הָרָעִים. אַחְאָב בֶּן קוֹלָיָה וְצִדְקִיָּה בֶּן מַעֲשֵׂיָה(צב) נַעֲשׂוּ רוֹפְאֵי שֶׁקֶר, וְהָיוּ מְרַפְּאִין אֶת נְשֵׁי כַּשְׂדִּים וּבָאִים עֲלֵיהֶם בְּתַשְׁמִישׁ הַמִּטָּה,(צד) שָׁמַע הַמֶּלֶךְ וְאָמַר לְשָׂרְפָן. אָמְרוּ שְׁנֵיהֶם(צה) יְהוֹשֻׁעַ בֶּן יְהוֹצָדָק אִישׁ צַדִּיק הוּא, יָבֹא עִמָּנוּ וּבִזְכוּתוֹ אָנוּ נִצֹּלִין, אָמְרוּ לוֹ אֲדוֹנֵינוּ הַמֶּלֶךְ הָאִישׁ הַזֶּה הָיָה עִמָּנוּ בְּכָל דָּבָר. צִוָּה הַמֶּלֶךְ לִשְׂרֹף אֶת שְׁלָשְׁתָּן בָּאֵשׁ,(צז) וְיָרַד הַמַּלְאָךְ מִיכָאֵל וְהִצִּיל אֶת יְהוֹשֻׁעַ מִשְׂרֵפַת הָאֵשׁ(צח) וְהֶעֱלָהוּ לִפְנֵי כִּסֵּא הַכָּבוֹד, שֶׁנֶּאֱמַר [זכריה ג, א] וַיַּרְאֵנִי אֶת יְהוֹשֻׁעַ הַכֹּהֵן הַגָּדוֹל. וּשְׁנֵיהֶם נִשְׂרְפוּ בָּאֵשׁ, שֶׁנֶּאֱמַר [ירמיה כט, כב] וְלֻקַּח מֵהֶם קְלָלָה אֲשֶׁר קָלָם. אֲשֶׁר שְׂרָפָם אֵין כְּתִיב כָּאן(צט) אֶלָּא אֲשֶׁר קָלָם [שם], הָא לָמַדְנוּ(ק) שֶׁנִּתְחָרְכוּ שַׂעֲרוֹתָיו בַּעֲוֹנוֹתֵיהֶם, שֶׁנֶּאֱמַר [זכריה ג, ב] הֲלוֹא זֶה אוּד מֻצָּל מֵאֵשׁ: רַבִּי יְהוּדָה אוֹמֵר,(קא) כְּשֶׁהֵבִיא נְבוּכַדְנֶצַּר עֲלִילוֹת דְּבָרִים עַל יִשְׂרָאֵל לְהָרְגָם, הֶעֱמִיד צֶלֶם בְּבִקְעַת דּוּרָא(קב) וְהוֹצִיא כָרוּז וְאָמַר, כָּל מִי שֶׁלֹּא יִשְׁתַּחֲוֶה לַצֶּלֶם הַזֶה יִשָׂרֵף בָּאֵשׁ. וְיִשְׂרָאֵל(קג) לֹא בָטְחוּ בְּצֵל יוֹצְרָם,(קד) וּבָאוּ הֵם וּנְשֵׁיהֶם וּבְנֵיהֶם(קה) וְהִשְׁתַּחֲווּ לַצֶּלֶם,(קו) חוּץ מִדָּנִיֵּאל חֲנַנְיָה מִישָׁאֵל וַעֲזַרְיָה, שֶׁהָיוּ קוֹרְאִים בְּשֵׁם אֱלֹהֵיהֶם(קז) וְהָיָה גְּנַאי לָהֶם לְשָׂרְפוֹ בָאֵשׁ, שֶׁנֶּאֱמַר [דניאל ד, ה] וְעַד אָחֳרֵין עַל קֳדָמַי דָּנִיֵּאל וְגוֹ'. וְנָטְלוּ לַחֲנַנְיָה מִישָׁאֵל וַעֲזַרְיָה וְנָתְנוּ אוֹתָן בְּתוֹךְ כִּבְשַׁן הָאֵשׁ,(קח) וְיָרַד גַּבְרִיאֵל וְהִצִּילָן מִכִּבְשַׁן הָאֵשׁ. אָמַר לָהֶם הַמֶּלֶךְ [נ"א: נבוכדנצאר] הֱיִיתֶם יוֹדְעִים(קט) שֶׁיֵּשׁ לָכֶם אֱלוֹהַּ מְשֵׁיזֵב וּמַצִּיל, לָמָּה הִנַּחְתֶּם אֱלֹהֵיכֶם וְהִשְׁתַּחֲוִיתֶם לַצֶּלֶם שֶׁאֵין בּוֹ כֹחַ לְהַצִּיל, אֶלָּא כְּדֶרֶךְ שֶׁעֲשִׂיתֶם בְּאַרְצְכֶם וְהֶחֱרַבְתֶּם אוֹתָהּ,(קי) כָּךְ אַתֶּם מְבַקְשִׁים לַעֲשׂוֹת בְּאַרְצִי לְהַחֲרִיב אוֹתָהּ. וְצִוָּה הַמֶּלֶךְ [נ"א: נבוכדנצאר] וְנֶהֶרְגוּ כֻּלָּם. וּמִנַּיִן שֶׁהָיוּ כֻּלָּם הֲרוּגֵי חֶרֶב, שֶׁנֶּאֱמַר [יחזקאל לז, ט] וַיֹּאמֶר אֵלַי(קיא) הִנָּבֵא וגו' בַּהֲרוּגִים הָאֵלֶּה וְיִחְיוּ: רַבִּי פִינְחָס אוֹמֵר,(קיב) לְאַחַר עֶשְׂרִים שָׁנָה שֶׁנֶּהֶרְגוּ כֻלָּם בְּבָבֶל,(קיג) שָׁרְתָה רוּחַ הַקֹּדֶשׁ עַל יְחֶזְקֵאל, וְהוֹצִיאוֹ לְבִקְעַת דּוּרָא(קיד) וְהֶרְאָהוּ עֲצָמוֹת יְבֵשׁוֹת הַרְבֵּה מְאֹד.(קטו) אָמַר לוֹ בֶּן אָדָם מָה אַתָּה רוֹאֶה, אָמַר לוֹ אֲנִי רוֹאֶה כָּאן עֲצָמוֹת יְבֵשׁוֹת,(קטז) אָמַר לוֹ יֶשׁ בִּי כֹּחַ לְהַחֲיוֹת אוֹתָן, וְהָיָה לוֹ לוֹמַר [נ"א: לא אמר הנביא] רִבּוֹנוֹ שֶׁל עוֹלָם יֵשׁ בְּךָ כֹּחַ לַעֲשׂוֹת יוֹתֵר מֵאֵלּוּ, אֶלָּא אָמַר ה' אֱלֹהִים אַתָּה יָדַעְתָּ [שם ג], כְּאִלּוּ לֹא הֶאֱמִין,(קיז) לְפִיכָךְ לֹא נִקְבְּרוּ עַצְמוֹתָיו בְּאֶרֶץ טְהוֹרָה [נ"א: ישראל], שֶׁנֶּאֱמַר [עמוס ז, יז] וְאַתָּה עַל אֲדָמָה טְמֵאָה תִּקָּבֵר. אָמַר לוֹ הִנָּבֵא עַל הָעֲצָמוֹת הָאֵלֶּה [יחזקאל לז, ד], אָמַר לְפָנָיו רִבּוֹנוֹ שֶׁל עוֹלָם,(קיח) מָה הַנְּבוּאָה מְבִיאָה עֲלֵיהֶם,(קיט) בָּשָׂר וְגִידִים וַעֲצָמוֹת אוֹ כָּל בָּשָׂר וְעֶצֶם שֶׁאָכְלוּ מֵהֶם בְּהֵמָה וְחַיָּה וָעוֹף וּמֵתוּ בְּאֶרֶץ אַחֶרֶת. מִיָּד(קכ) הִשְׁמִיעַ הַקָּדוֹשׁ בָּרוּךְ הוּא קוֹלוֹ וְרָעֲשָׁה הָאָרֶץ, שֶׁנֶּאֱמַר [שם ז] וַיְהִי קוֹל כְּהִנָּבְאִי וְהִנֵּה רַעַשׁ וְגוֹ',(קכא) וְהָיְתָה הָאָרֶץ הַהִיא מְקָרֶבֶת עֶצֶם אֶל עַצְמוֹ. רַבִּי יְהוֹשֻׁעַ בֶּן קָרְחָה אוֹמֵר,(קכב) יָרַד עֲלֵיהֶם תְּחִיַּת טַל מִן הַשָּׁמַיִם(קכג) שֶׁהוּא נוֹבֵעַ וּמוֹצִיא מַיִם, כָּךְ הָיוּ נוֹבְעִים(קכד) וְעוֹלִים עֲלֵיהֶם בָּשָׂר וַעֲצָמוֹת וְגִידִים, שֶׁנֶּאֱמַר [שם לז, ח] וְרָאִיתִי וְהִנֵּה עֲלֵיהֶם גִּידִים, אָמַר לוֹ הִנָּבֵא אֶל הָרוּחַ [שם ט]. בְּאוֹתָהּ שָׁעָה(קכה) יָצְאוּ אַרְבַּע רוּחוֹת הַשָּׁמַיִם וּפָתְחוּ אֶת אוֹצְרוֹת הַנְּשָׁמוֹת,(קכו) וְהֶחֱזִיר כָּל רוּחַ וָרוּחַ לְגוּף בְּשַׂר אִישׁ,(קכז) שֶׁנֶּאֱמַר [שם י] וְהִנַּבֵּאתִי כַּאֲשֶׁר צִוָּנִי. כְּתִיב הָכָא חַיִל גָּדוֹל מְאֹד וְגוֹ' [שם],(קכח) וּכְתִיב וּבְנֵי יִשְׂרָאֵל פָּרוּ וְגוֹ' [שמות א, ז], מַה לְּהַלָּן שִׁשִּׁים רִבּוֹא אַף כָּאן שִׁשִּׁים רִבּוֹא.(קכט) וְכֻלָּם עָמְדוּ עַל רַגְלֵיהֶם(קל) חוּץ מֵאִישׁ אֶחָד. אָמַר הַנָּבִיא רִבּוֹנוֹ שֶׁל עוֹלָם מַה טִּיבוֹ שֶׁל זֶה, אָמַר לוֹ בַּנֶּשֶׁךְ נָתַן [יחזקאל יח, יג]. בְּאוֹתָהּ שָׁעָה(קלא) הָיוּ יִשְׂרָאֵל בּוֹכִים וְאוֹמְרִים,(קלב) הָיִינוּ מְקַוִּים לַעֲמֹד עִם כָּל יִשְׂרָאֵל בִּתְחִיַּת הַמֵּתִים עַכְשָׁו אָבְדָה תִקְוָתֵנוּ [שם לז, יא]. בְּאוֹתָהּ שָׁעָה אָמַר הַקָּדוֹשׁ בָּרוּךְ הוּא לַנָּבִיא לֵךְ אֱמֹר לָהֶם,(קלג) חַי אָנִי שֶׁאַעֲמִיד אֶתְכֶם בִּתְחִיַּת הַמֵּתִים לֶעָתִיד לָבוֹא, וּמְקַבֵּץ אֶת כָּל יִשְׂרָאֵל לְקִבּוּץ גָּלֻיּוֹת לְאֶרֶץ יִשְׂרָאֵל, שֶׁנֶּאֱמַר [שם יב] הִנֵּה אֲנִי פֹתֵחַ אֶת קִבְרוֹתֵיכֶם.

כַּאֲשֶׁר יָנוּס אִישׁ מִפְּנֵי הָאֲרִי וּפְגָעוֹ הַדֹּב [עמוס ה, יט]. הָאֲרִי זֶה לָבָן, שֶׁרָדַף אַחֲרֵי יַעֲקֹב(א) לַחֲטֹף אֶת נַפְשׁוֹ.(ב) הַדֹּב [נ"א: אחיו], זֶה עֵשָׂו, שֶׁעָמַד עַל הַדֶּרֶךְ כְּדֹב שַׁכּוּל בָּא לְהָמִית אֵם עַל בָּנִים. וְהָאֲרִי יֶשׁ לוֹ בּשֶׁת פָּנִים וְהַדֹּב אֵין לוֹ בּשֶׁת פָּנִים. עָמַד יַעֲקֹב(ה) וְהָיָה מִתְפַּלֵּל לִפְנֵי הַקָּדוֹשׁ בָּרוּךְ הוּא, וְאָמַר רִבּוֹנוֹ שֶׁל עוֹלָם, לֹא כָךְ אָמַרְתָּ לִי שׁוּב אֶל אֶרֶץ אֲבוֹתֶיךָ וּלְמוֹלַדְתֶּךָ וְאֶהְיֶה עִמָּךְ [בראשית לא, ג], וַהֲרֵי עֵשָׂו אָחִי עַתָּה בָּא לְהָרְגֵנִי וְאֵינוֹ מִתְיָרֵא מִמְּךָ, וַאֲנִי מִתְיָרֵא מִמֶּנוּ. מִכָּאן אָמְרוּ אַל תִּתְיָרֵא מִפְּנֵי אִישׁ שׁוֹטֵר וּמוֹשֵׁל, אֶלָּא מִפְּנֵי אָדָם שֶׁאֵין בּוֹ יִרְאַת שָׁמָיִם. וְעָמַד לוֹ עַל הַדֶּרֶךְ כְּדֹב שַׁכּוּל לְהָמִית אֵם עַל בָּנִים. מֶה עָשָׂה הַקָּדוֹשׁ בָּרוּךְ הוּא,(ט) שָׁלַח לוֹ מַלְאָךְ(י) לְהַצִּילוֹ וּלְהוֹשִׁיעוֹ מִיַּד עֵשָׂו, וְנִרְאָה לוֹ כְּאִישׁ, שֶׁנֶּאֱמַר [שם לב, כה] וַיֵּאָבֵק אִישׁ עִמּוֹ עַד עֲלוֹת הַשָּׁחַר.(יא) וְכֵיוָן שֶׁעָלָה עַמּוּד הַשַּׁחַר אָמַר לוֹ הַמַּלְאָךְ שַׁלְּחֵנִי, שֶׁהִגִּיעָה עֵת עֲמִידָתִי לְשָׁרֵת וּלְזַמֵר לִפְנֵי הַקָּדוֹשׁ בָּרוּךְ הוּא, וְלֹא רָצָה יַעֲקֹב לְשַׁלְּחוֹ. מֶה עָשָׂה הַמַּלְאָךְ,(יב) הִתְחִיל לְשׁוֹרֵר וּלְזַמֵּר מִן הָאָרֶץ. וּכְשֶׁשָּׁמְעוּ הָעֶלְיוֹנִים אֶת קוֹל הַמַּלְאָךְ שֶׁהָיָה מְשׁוֹרֵר וּמְזַמֵּר מִן הָאָרֶץ, אָמְרוּ(יג) בִּשְׁבִיל כְּבוֹד הַצַּדִּיק אָנוּ שׁוֹמְעִים אֶת קוֹל הַמַּלְאָךְ מִן הָאָרֶץ,(יד) עָלָיו הַכָּתוּב אוֹמֵר מִכְּנַף הָאָרֶץ זְמִרֹת שָׁמַעְנוּ צְבִי לַצַּדִּיק וְגוֹ' [ישעיה כד, טז].(טו) וְעוֹד אָמַר לוֹ הַמַּלְאָךְ שַׁלְּחֵנִי, אָמַר לוֹ יַעֲקֹב אֵינִי מְשַׁלֵּחֲךָ עַד שֶׁתְּבָרְכֵנִי, שֶׁנֶּאֱמַר [בראשית לב, כז] לֹא אֲשַׁלֵּחֲךָ כִּי אִם בֵּרַכְתָּנִי. וּבֵרְכוֹ, שֶׁנֶּאֱמַר [שם לא] וַיְבָרֶךְ אֹתוֹ שָׁם. וְעוֹד אָמַר לוֹ שַׁלְּחֵנִי, אָמַר לוֹ אֵינִי מְשַׁלֵּחֲךָ(טז) עַד שֶׁתֹּאמַר לִי מַה שְּׁמֶךָ. וְקָרָא שְׁמוֹ יִשְׂרָאֵל כִּשְׁמוֹ, שֶׁנִקְרָא שְׁמוֹ יִשְׂרָאֵל.(יז) רָצָה יַעֲקֹב לְהִתְגַּבֵּר עִם הַמַּלְאָךְ, וְהִפִּילוֹ לָאָרֶץ. מֶה עָשָׂה הַמַּלְאָךְ,(יח) אָחַז בְּגִיד הַנָּשֶׁה שֶׁעַל כַּף הַיָּרֵךְ, וְנָשָׁה גִיד הַנָּשֶׁה שֶׁלּוֹ(כ) וְנַעֲשָׂה כְּחֵלֶב הַמֵּת.(כא) לְפִיכָךְ אָסוּר לִבְנֵי יִשְׂרָאֵל לֶאֱכֹל גִּיד הַנָּשֶׁה שֶׁעַל כַּף הַיָּרֵךְ, שֶׁנֶּאֱמַר [שם לג] עַל כֵּן לֹא יֹאכְלוּ בְנֵי יִשְׂרָאֵל אֶת גִּיד הַנָּשֶׁה אֲשֶׁר עַל כַּף הַיָּרֵךְ וְגוֹ':(כב) רָצָה יַעֲקֹב לַעֲבֹר אֶת מַעֲבַר יַבֹּק וּלְהִתְעַכֵּב שָׁם, אָמַר לוֹ הַמַּלְאָךְ,(כג) לֹא כָךְ אָמַרְתָּ לִי עַשֵׂר אֲעַשְׂרֶנּוּ לָךְ [שם כח, כב]. מֶה עָשָׂה יַעֲקֹב,(כד) לָקַח אֶת כָּל מִקְנֵה קִנְיָנוֹ שֶׁהֵבִיא מִפַּדַּן אֲרָם,(כה) וְהָיוּ חֲמֵשֶׁת אֲלָפִים וַחֲמֵשׁ מֵאוֹת צֹאן. וְעוֹד אָמַר הַמַּלְאָךְ לְיַעֲקֹב, וַהֲלֹא יֶשׁ לְךָ בָּנִים וְלֹא עִשַׂרְתָּ אוֹתָם לִי, מֶה עָשָׂה יַעֲקֹב,(כו) הִפְרִישׁ אַרְבָּעָה בְּכוֹרוֹת לְאַרְבַּע אִמָּהוֹת, וְנִשְׁתַּיְּרוּ שְׁמוֹנָה. הִתְחִיל מִשִּׁמְעוֹן וְגָמַר בְּבִנְיָמִין שֶׁבִּמְעֵי אִמּוֹ,(כז) וְעוֹד הִתְחִיל מִשִּׁמְעוֹן וְגָמַר בְּלֵוִי, וְעָלָה לֵוִי מַעֲשֵׂר קֹדֶשׁ לַה', שֶׁנֶּאֱמַר [ויקרא כז, לב] הָעֲשִׂירִי יִהְיֶה קֹּדֶשׁ לַה'.(כח) רַבִּי יִשְׁמָעֵאל אוֹמֵר, כָּל הַבְּכוֹרוֹת כְּשֶׁהֵן בִּשְׁמִירַת הָעַיִן חַיָּבִין לְהִתְעַשֵׂר מַעֲשֵׂר, וְלֹא עִשֵׂר יַעֲקֹב(כט) אֶלָּא לְמַפְרֵעַ. הִתְחִיל מִבִּנְיָמִין שֶׁבִּמְעֵי אִמּוֹ, וְעָלָה לֵוִי קֹדֶשׁ לַה',(ל) וְעָלָיו הַכָּתוּב אוֹמֵר הָעֲשִׂירִי יִהְיֶה קֹּדֶשׁ לַה'. יָרַד מִיכָאֵל הַמַּלְאָךְ וְנָטַל אֶת לֵוִי וְהֶעֱלָהוּ לִפְנֵי הַקָּדוֹשׁ בָּרוּךְ הוּא, אָמַר לְפָנָיו רִבּוֹנוֹ שֶׁל עוֹלָם(לא) זֶה גוֹרָלְךָ(לב) וְחֵלֶק מַעֲשֶׂרְךָ [נ"א: מעשיך].(לג) וּפָשַׁט יַד יְמִינוֹ וּבֵרְכוֹ שֶׁיִּהְיוּ(לד) בְּנֵי לֵוִי מְשָׁרְתִים לְפָנָיו בָּאָרֶץ כְּמַלְאֲכֵי הַשָּׁרֵת בַּשָּׁמָיִם. אָמַר מִיכָאֵל לִפְנֵי הַקָּדוֹשׁ בָּרוּךְ הוּא, רִבּוֹן כָּל הָעוֹלָמִים,(לה) הַמְשָׁרְתִים לַמֶּלֶךְ אֵינוֹ נוֹתֵן לָהֶם טֶרֶף מְזוֹנָם, לְפִיכָךְ נָתַן לָהֶם לִבְנֵי לֵוִי(לו) כָּל קֹדֶשׁ שֶׁעָלָה לִשְׁמוֹ, שֶׁנֶּאֱמַר [דברים יח, א] אִשֵּׁי ה' וְנַחֲלָתוֹ יֹאכֵלוּן:(לז) כְּשֶׁעָבַר יַעֲקֹב לָבֹא בְּאֶרֶץ כְּנַעַן, בָּא אֵלָיו עֵשָׂו [נ"א: אחיו] מֵהַר שֵׂעִיר בְּזַעַף אַף וְזָמַם לְהָרְגוֹ, שֶׁנֶּאֱמַר [תהלים לז, יב] זֹמֵם רָשָׁע לַצַּדִּיק וְחֹרֵק עָלָיו שִׁנָּיו. אָמַר עֵשָׂו אֵינִי הוֹרֵג אֶת יַעֲקֹב אָחִי בְּחִצִּים וּבְקֶשֶׁת, אֶלָּא בְּפִי אֲנִי הוֹרְגוֹ וּמוֹצֵץ אֶת דָּמוֹ, שֶׁנֶּאֱמַר [בראשית לג, ד] וַיָּרָץ עֵשָׂו לִקְרָאתוֹ וַיְחַבְּקֵהוּ וַיִּפֹּל עַל צַוָּארָיו וַיִּשָּׁקֵהוּ וַיִּבְכּוּ. אַל תְּהִי קוֹרֵא וַיִּשָּׁקֵהוּ אֶלָּא וַיִּשָּׁכֵהוּ. וְנַעֲשָׂה צַוָּארוֹ שֶׁל יַעֲקֹב כְּעֶצֶם הַשֵּׁן, וְעָלָיו הַכָּתוּב אוֹמֵר [שה"ש ז, ה] צַוָּארֵךְ כְּמִגְדַּל הַשֵּׁן, וְקָהוּ שִׁנָּיו שֶׁל עֵשָׂו. וְכֵיוָן שֶׁרָאָה שֶׁלֹּא עָלְתָה בְיָדוֹ תַּאֲוָתוֹ, הִתְחִיל כּוֹעֵס וְחוֹרֵק בְּשִׁנָּיו, שֶׁנֶּאֱמַר [תהלים קיב, י] רָשָׁע יִרְאֶה וְכָעָס שִׁנָּיו יַחֲרֹק וְנָמָס. וְלָקַח יַעֲקֹב אֶת מַעְשַׂר קִנְיָנוֹ שֶׁהֵבִיא מִפַּדַּן אֲרָם,(מד) וְשָׁלַח בְּיַד עֲבָדָיו וְנָתַן לְעֵשָׂו,(מה) אָמַר לָהֶם אִמְרוּ לוֹ כֹּה אָמַר עַבְדְּךָ יַעֲקֹב [בראשית לב, ה]. אָמַר לוֹ הַקָּדוֹשׁ בָּרוּךְ הוּא, יַעֲקֹב, עָשִׂיתָ אֶת הַקֹּדֶשׁ חֹל. אָמַר לְפָנָיו רִבּוֹן כָּל הָעוֹלָמִים אֲנִי מַחֲנִיף לָרָשָׁע בִּשְׁבִיל שֶׁלֹּא יַהֲרְגֵנִי. מִכָּאן אָמְרוּ חֲכָמִים מַחֲנִיפִין אֶת הָרְשָׁעִים בָּעוֹלָם הַזֶּה מִפְּנֵי דַרְכֵי שָׁלוֹם. אָמַר לוֹ עֵשָׂו, אָחִי יֶשׁ לִי רָב. וּבִשְׁבִיל שֶׁחָלַק כָּבוֹד יַעֲקֹב לְעֵשָׂו, לְפִיכָךְ חָלְקוּ בְּנֵי יַעֲקֹב כָּבוֹד לִבְנֵי עֵשָׂו בִּלְשׁוֹן רַב, שֶׁנֶּאֱמַר [דברים ב, ג] רַב לָכֶם סֹב אֶת הָהָר הַזֶּה. אָמַר לוֹ הַקָּדוֹשׁ בָּרוּךְ הוּא, לֹא דַיְּךָ שֶׁעָשִׂיתָ אֶת עַצְמְךָ קֹדֶשׁ חֹל, אֶלָּא שֶׁאָמַרְתִּי וְרַב יַעֲבֹד צָעִיר [בראשית כה, כג], וְאַתָּה אָמַרְתָּ עַבְדְּךָ יַעֲקֹב. חַיֶּיךָ כִּדְבָרֶיךָ יִהְיֶה, הוּא יִמְשֹׁל עָלֶיךָ בָּעוֹלָם הַזֶּה, וְאַתָּה תִּהְיֶה מוֹשֵׁל עָלָיו בָּעוֹלָם הַבָּא. וּלְפִיכָךְ אָמַר לוֹ יַעֲקֹב יַעֲבָר נָא אֲדֹנִי לִפְנֵי עַבְדוֹ וַאֲנִי אֶתְנַהֲלָה לְאִטִּי וְגוֹ' [שם לג, יד]. מִכָּאן אַתָּה לָמֵד שֶׁאֵין בְּנֵי עֵשָׂו נוֹפְלִין עַד שֶׁיָּבֹא שָׂרִיד מִיַּעֲקֹב וִיקַצֵּץ רַגְלֵיהֶם שֶׁל בְּנֵי עֵשָׂו מֵהַר שֵׂעִיר, שֶׁנֶּאֱמַר [במדבר כד, יט] וְיֵרְדְּ מִיַּעֲקֹב וְגוֹ', וּכְתִיב [עובדיה א, יח] וְלֹא יִהְיֶה שָׂרִיד לְבֵית עֵשָׂו כִּי ה' דִּבֵּר.

(א) יְרִידָה הַשִּׁשִּׁית, שֶׁיָּרַד לְסִינַי, שֶׁנֶּאֱמַר [שמות יט, כ] וַיֵּרֶד ה' עַל הַר סִינַי.(ב) בְּשִׁשָּׁה בְּסִיוָן(ג) נִגְלָה הַקָּדוֹשׁ בָּרוּךְ הוּא עַל יִשְׂרָאֵל בְּהַר סִינַי,(ד) וּמִמְּקוֹמוֹ נִתְלַשׁ הַר סִינַי,(ה) וְנִפְתְּחוּ הַשָּׁמַיִם(ו) וְנִכְנַס רֹאשׁ הָהָר בַּשָּׁמַיִם,(ז) וַעֲרָפֶל מְכַסֶּה אֶת הָהָר.(ח) הַקָּדוֹשׁ בָּרוּךְ הוּא יוֹשֵׁב עַל כִּסְאוֹ(ט) וְרַגְלָיו עוֹמְדוֹת עַל הָעֲרָפֶל, שֶׁנֶּאֱמַר [ש"ב כב, י] וַיֵּט שָׁמַיִם וַיֵּרַד וַעֲרָפֶל תַּחַת רַגְלָיו. רַבִּי טַרְפוֹן אוֹמֵר, זָרַח הַקָּדוֹשׁ בָּרוּךְ הוּא מֵהַר שֵׂעִיר וְנִגְלָה עַל בְּנֵי עֵשָׂו, שֶׁנֶּאֱמַר [דברים לג, ב] וַיֹּאמַר ה' מִסִּינַי בָּא וְזָרַח מִשֵׂעִיר לָמוֹ. וְאֵין שֵׂעִיר אֶלָּא בְּנֵי עֵשָׂו, שֶׁנֶּאֱמַר [בראשית לו, ח] וַיֵּשֶׁב עֵשָׂו בְּהַר שֵׂעִיר. אָמַר לָהֶם הַקָּדוֹשׁ בָּרוּךְ הוּא, מְקַבְּלִים אַתֶּם אֶת הַתּוֹרָה, אָמְרוּ לוֹ מַה כְּתִיב בָּהּ, אָמַר לָהֶם לֹא תִרְצָח [שמות כ, יג]. אָמְרוּ לוֹ לֵךְ מֵעִמָּנוּ, כִּי אֵין אָנוּ יְכוֹלִים לַעֲזֹב אֶת הַבְּרָכָה שֶׁבֵּרַךְ יִצְחָק אֶת עֵשָׂו אָבִינוּ שֶׁאָמַר לוֹ וְעַל חַרְבְּךָ תִחְיֶה [בראשית כז, מ]. וּמִשָּׁם חָזַר וְנִגְלָה עַל בְּנֵי יִשְׁמָעֵאל, שֶׁנֶּאֱמַר [דברים לג, ב] הוֹפִיעַ מֵהַר פָּארָן, אָמַר לָהֶם מְקַבְּלִים אַתֶּם אֶת הַתּוֹרָה, אָמְרוּ לוֹ מַה כְּתִיב בָּהּ, אָמַר לָהֶם לֹא תִגְנֹב [שמות כ, טו], אָמְרוּ לוֹ אֵין אָנוּ יְכוֹלִים לַעֲזֹב אֶת הַדָּבָר שֶׁעָשׂוּ אֲבוֹתֵינוּ, שֶׁגָּנְבוּ אֶת יוֹסֵף וְהוֹרִידוּהוּ לְמִצְרַיִם, שֶׁנֶּאֱמַר [בראשית מ, טו] כִּי גֻנֹּב גֻּנַּבְתִּי מֵאֶרֶץ הָעִבְרִים.(יג) וּמִשָּׁם שָׁלַח לְכָל אֻמּוֹת הָעוֹלָם, [נ"א: לשאר האומות] וְאָמַר לָהֶם מְקַבְּלִים אַתֶּם אֶת הַתּוֹרָה, אָמְרוּ לוֹ מַה כְּתִיב בָּהּ, אָמַר לָהֶם לֹא יִהְיֶה לְךָ אֱלֹהִים אֲחֵרִים עַל פָּנָי [שמות כ, ג],(יד) אָמְרוּ לוֹ(טו) אֵין אָנוּ יְכוֹלִין לְהַנִּיחַ דַּת אֲבוֹתֵינוּ שֶׁעָבְדוּ אֶת הָאֱלִילִים, אֵין אָנוּ חֲפֵצִים בַּתּוֹרָה, אֶלָּא(טז) תֵּן תּוֹרָתְךָ לְעַמֶּךָ, שֶׁנֶּאֱמַר [תהלים כט, יא] ה' עֹז לְעַמּוֹ יִתֵּן ה' יְבָרֵךְ אֶת עַמּוֹ בַשָּׁלוֹם. וּמִשָּׁם חָזַר וְנִגְלָה עַל בְּנֵי יִשְׂרָאֵל, שֶׁנֶּאֱמַר [דברים לג, ב] וְאָתָה מֵרִבְבֹת קֹדֶשׁ. וְאֵין רִבְבוֹת אֶלָּא יִשְׂרָאֵל, שֶׁנֶּאֱמַר [במדבר י, לו] וּבְנֻחֹה יֹאמַר שׁוּבָה ה' רִבְבוֹת אַלְפֵי יִשְׂרָאֵל.(יז) וְעִמּוֹ(יח) אַלְפֵי שִׁנְאָן וְרֶכֶב רִבּוֹתַיִם מַלְאֲכֵי קֹדֶשׁ.(יט) וִימִינוֹ אוֹחֶזֶת אֶת הַתּוֹרָה, שֶׁנֶּאֱמַר [דברים לג, ב] מִימִינוֹ אֵשׁ דָּת לָמוֹ. מִכָּאן אַתָּה לָמֵד(כ) שֶׁדִּבְרֵי תוֹרָה כְּגַחֲלֵי אֵשׁ.(כא) וְנָתַן לָהֶם בִּלְשׁוֹן חִבָּה, שֶׁנֶּאֱמַר [שה"ש ב, ו] שְׂמֹאלוֹ תַּחַת לְרֹאשִׁי וְגוֹ'.(כב) בִּלְשׁוֹן שְׁבוּעָה, שֶׁנֶּאֱמַר נִשְׁבַּע ה' בִּימִינוֹ וּבִזְרוֹעַ עֻזּוֹ, וְאֵין יְמִינוֹ אֶלָּא שְׁבוּעָה, שֶׁנֶּאֱמַר [ישעיה סב, ח] נִשְׁבַּע ה' בִּימִינוֹ: רַבִּי אֱלִיעֶזֶר אוֹמֵר,(כג) מִיּוֹם שֶׁיָּצְאוּ יִשְׂרָאֵל מִמִּצְרַיִם, הָיוּ נוֹסְעִים וְחוֹנִים בְּמַחֲלֹקֶת [נ"א: בחלקלות],(כד) שֶׁנֶּאֱמַר [שמות יט, ב] וַיִסְעוּ וַיַּחֲנוּ.(כה) עַד שֶׁבָּאוּ כֻּלָּם לְהַר סִינַי(כו) וְחָנוּ כְּנֶגֶד הָהָר, שֶׁנֶּאֱמַר [שם] וַיִּחַן שָׁם יִשְׂרָאֵל נֶגֶד הָהָר. אָמַר לָהֶם מְקַבְּלִים אַתֶּם אֶת הַתּוֹרָה, אָמְרוּ כֻלָּם(כז) פֶּה אֶחָד,(כח) עַד שֶׁלֹּא נִשְׁמַע אֶת הַתּוֹרָה אָנוּ(כט) שׁוֹמְרִים וְעוֹשִׂים כָּל הַמִּצְוֹת הַכְּתוּבִים בַּתּוֹרָה, שֶׁנֶּאֱמַר [שם כד, ג](כט) וַיֹּאמְרוּ כָּל הַדְּבָרִים אֲשֶׁר דִּבֶּר ה' נַעֲשֶׂה וְנִשְׁמָע: רַבִּי אֶלְעָזָר הַמּוֹדָעִי אוֹמֵר,(ל) מִיּוֹם שֶׁנִּבְרְאוּ שָׁמַיִם וָאָרֶץ(לא) נִקְרָא שֵׁם הָהָר חוֹרֵב,(לב) וּכְשֶׁנִּגְלָה הַקָּדוֹשׁ בָּרוּךְ הוּא עַל מֹשֶׁה מִתּוֹךְ הַסְּנֶה, עַל שֵׁם הַסְּנֶה(לג) נִקְרָא סִינַי, הוּא חוֹרֵב. וּמִנַּיִן שֶׁקִּבְּלוּ עֲלֵיהֶם אֶת הַתּוֹרָה בְּהַר חוֹרֵב,(לד) שֶׁנֶּאֱמַר [דברים ד, י] יוֹם אֲשֶׁר עָמַדְתָּ לִפְנֵי ה' אֱלֹהֶיךָ בְּחֹרֵב: רַבִּי פִּינְחָס אוֹמֵר,(לה) עֶרֶב שַׁבָּת עָמְדוּ יִשְׂרָאֵל בְּהַר סִינַי(לו) עֲרוּכִים, הָאֲנָשִׁים לְבַד וְהַנָּשִׁים לְבָד. אָמַר לוֹ הַקָּדוֹשׁ בָּרוּךְ הוּא לְמֹשֶׁה לֵךְ אֱמֹר לִבְנוֹת יִשְׂרָאֵל, אִם רוֹצוֹת לְקַבֵּל אֶת הַתּוֹרָה,(לז) שֶׁדַּרְכָּם שֶׁל אֲנָשִׁים הוֹלְכִים אַחַר דַּעְתָּן שֶׁל נָשִׁים, שֶׁנֶּאֱמַר [שמות יט, ג] כֹּה תֹאמַר לְבֵית יַעֲקֹב אֵלּוּ הַנָּשִׁים. וְתַגֵּיד לִבְנֵי יִשְׂרָאֵל [שם] אֵלּוּ הָאֲנָשִׁים.(לח) וְעָנוּ כֻלָּם פֶּה אֶחָד, כֹּל אֲשֶׁר דִּבֶּר ה' נַעֲשֶׂה וְנִשְׁמָע [שם כד, ז].(לט) וְאוֹמֵר וְשָׁרִים כְּחֹלְלִים כָּל מַעְיָנַי בָּךְ [תהלים פז, ז]: רַבִּי חֲכִינַאי אוֹמֵר, בַּחֹדֶשׁ הַשְּׁלִישִׁי [שמות יט, א],(מ) הַיּוֹם כָּפוּל כַּלַּיְלָה,(מא) וְיָשְׁנוּ יִשְׂרָאֵל עַד שְׁתֵּי שָׁעוֹת בַּיּוֹם.(מב) שֶׁשְּׁנַת [נ"א: ששעת] יוֹם בַּעֲצֶרֶת עֲרֵבָה [נ"א: אֲרוּכָּה] וְהַלַּיְלָה קָצָר. וְיָצָא מֹשֶׁה לְמַחֲנֵה יִשְׂרָאֵל(מג) וְהָיָה מְעוֹרֵר אוֹתָם מִשְּׁנָתָם, אָמַר לָהֶם עִמְדוּ מִשְּׁנַתְכֶם,(מד) כְּבָר בָּא הֶחָתָן וּמְבַקֵּשׁ אֶת הַכַּלָּה לְהַכְנִיסָהּ לַחֻפָּה וּמַמְתִּין לָהּ, כְּדֵי לִתֵּן לָהֶם אֶת הַתּוֹרָה. בָּא הַשּׁוֹשְׁבִין וְהוֹצִיא אֶת הַכַּלָּה, כְּאָדָם שֶׁעוֹשֶׂה שׁוֹשְׁבִינוּת לַחֲבֵרוֹ, שֶׁנֶּאֱמַר [שם יז] וַיּוֹצֵא מֹשֶׁה אֶת הָעָם לִקְרַאת הָאֱלֹהִים וְגוֹ'.(מה) וְהֶחָתָן יוֹצֵא לִקְרַאת הַכַּלָּה לִתֵּן לָהֶם אֶת הַתּוֹרָה, שֶׁנֶּאֱמַר [תהלים סח, ח] אֱלֹהִים בְּצֵאתְךָ לִפְנֵי עַמֶּךָ:(מו) רַבִּי יְהוֹשֻׁעַ בֶּן קָרְחָה אוֹמֵר, מֹשֶׁה הָיוּ רַגְלָיו עוֹמְדוֹת בָּהָר וְכֻלּוֹ בְּתוֹךְ הַשָּׁמַיִם,(מז) כְּטֶנְדָּס שֶׁהִיא פְּרוּשָׂה וּבְנֵי אָדָם יוֹשְׁבִים בְּתוֹכָהּ, וְרַגְלֵיהֶם עוֹמְדוֹת בָּאָרֶץ וְכֻלָּם בְּתוֹךְ הָאֹהֶל, כָּךְ הָיָה מֹשֶׁה רַגְלָיו עוֹמְדוֹת בָּהָר וְכֻלּוֹ בַּשָּׁמַיִם,(מח) צוֹפֶה וּמַבִּיט כָּל מַה שֶּׁבַּשָּׁמָיִם.(מט) וְהַקָּדוֹשׁ בָּרוּךְ הוּא מְדַבֵּר עִמּוֹ כְּאָדָם שֶׁהוּא מְדַבֵּר עִם חֲבֵרוֹ, שֶׁנֶּאֱמַר [שמות לג, יא] וְדִבֶּר ה' אֶל מֹשֶׁה פָּנִים אֶל פָּנִים. אָמַר לוֹ הַקָּדוֹשׁ בָּרוּךְ הוּא, לֵךְ וְקַדֵּשׁ אֶת יִשְׂרָאֵל שְׁנֵי יָמִים, שֶׁנֶּאֱמַר [שם יט, י] וַיֹּאמֶר ה' אֶל מֹשֶׁה לֵךְ אֶל הָעָם וְקִדַּשְׁתָּם הַיּוֹם וּמָחָר.(נ) וְכִי מֶה הָיְתָה קְדֻשָּׁתָן שֶׁל יִשְׂרָאֵל בַּמִּדְבָּר, עַרְלֵי לֵב לֹא הָיוּ בְּתוֹכָם, מָן הָיָה יוֹרֵד לָהֶם מִן הַשָּׁמַיִם, מַיִם הָיוּ שׁוֹתִים מִן הַבְּאֵר, עַנְנֵי כָבוֹד הָיוּ מַקִּיפִים עֲלֵיהֶם. אֶלָּא עַל תַּשְׁמִישׁ הַמִּטָּה [נ"א: ההפרשה מנשותיהם] שֶׁלֹּא יְשַׁמְּשׁוּ. וּמֹשֶׁה דָּן דִּין בֵּינוֹ לְבֵין עַצְמוֹ,(נא) אָמַר מֹשֶׁה יֵלֵךְ אָדָם אֵצֶל אִשְׁתּוֹ(נב) וְנִמְצְאוּ מִתְעַכְּבִין מִלְּקַבֵּל אֶת הַתּוֹרָה, מֶה עָשָׂה, הוֹסִיף לָהֶם יוֹם אֶחָד מִשֶּׁלּוֹ, שֶׁאִם יֵלֵךְ אָדָם אֵצֶל אִשְׁתּוֹ נִמְצְאוּ טְהוֹרִים שְׁנֵי יָמִים, לְפִיכָךְ הוֹסִיף לָהֶם יוֹם אֶחָד מִשֶּׁלּוֹ. אָמַר לוֹ הַקָּדוֹשׁ בָּרוּךְ הוּא, מֹשֶׁה, הַרְבֵּה נְפָשׁוֹת מִיִשְׂרָאֵל הָיוּ בָּאוֹת בְּאוֹתוֹ הַלַּיְלָה,(נד) אֶלָּא מַה שֶּׁעָשִׂיתָ יָפֶה עָשִׂיתָ, וְהוֹדָה לוֹ הַקָּדוֹשׁ בָּרוּךְ הוּא,(נה) שֶׁנֶּאֱמַר [שמות יט, יא] וְהָיוּ נְכֹנִים לַיּוֹם הַשְּׁלִישִׁי.(נו) אָמַר הַקָּדוֹשׁ בָּרוּךְ הוּא יֵרֵד מֹשֶׁה אֶל הַמַּחֲנֶה, וְאַחַר כָּךְ אֲנִי מַשְׁמִיעַ תּוֹרָתִי לְיִשְׂרָאֵל,(נז) שֶׁלֹּא יְהֵא אוֹמֵר מֹשֶׁה הָיָה מְדַבֵּר עִמָּנוּ מִתּוֹךְ הֶעָנָן. אָמַר לוֹ רֵד הָעֵד בָּעָם [שם כא].(נח) מֹשֶׁה הָיָה רוֹצֶה לִהְיוֹת שָׁם,(נט) אָמַר לוֹ כְּבָר הַעִידוֹתִי בָּעָם, אָמַר לוֹ קְרָא אֶל אַהֲרֹן. יָרַד אֶל הַמַּחֲנֶה לִקְרֹא אֶת אַהֲרֹן, וְהַקָּדוֹשׁ בָּרוּךְ הוּא הִשְׁמִיעַ אֶת תּוֹרָתוֹ לְיִשְׂרָאֵל, שֶׁנֶּאֱמַר [שם כד] וַיֹּאמֶר אֵלָיו ה' לֶךְ רֵד, וַיֵּרֶד מֹשֶׁה אֶל הָעָם וַיֹּאמֶר אֲלֵהֶם [שם כה].(ס) מַה כְּתִיב בַּתְרֵהּ, וַיְדַבֵּר אֱלֹהִים אֶת כָּל הַדְּבָרִים הָאֵלֶּה לֵאמֹר אָנֹכִי ה' אֱלֹהֶיךָ אֲשֶׁר הוֹצֵאתִיךָ [שם כ, א - ב]. יָצָא קוֹל רִאשׁוֹן,(סא) וְהַשָּׁמַיִם וְהָאָרֶץ רָעֲשׁוּ מִמֶּנּוּ,(סב) וְהַיַּמִּים וְהַנְּהָרוֹת בָּרְחוּ,(סג) הֶהָרִים וְהַגְּבָעוֹת נִתְמוֹטְטוּ,(סד) וְכָל הָאִילָנוֹת כָּרְעוּ.(סה) וְהַמֵּתִים שֶׁבִּשְׁאוֹל חָיוּ וְעָמְדוּ עַל רַגְלֵיהֶם, שֶׁנֶּאֱמַר [דברים כט, יד] כִּי אֶת אֲשֶׁר יֶשְׁנוֹ פֹּה עִמָּנוּ עֹמֵד הַיּוֹם.(סו) וְכָל הָעֲתִידִים לְהִבָּרְאוֹת עַד סוֹף כָּל הַדּוֹרוֹת, שָׁם עָמְדוּ עִמָּהֶם בְּהַר סִינַי, שֶׁנֶּאֱמַר [שם] וְאֵת אֲשֶׁר אֵינֶנּוּ פֹּה עִמָּנוּ הַיּוֹם.(סז) וְיִשְׂרָאֵל שֶׁהֵם חַיִּים נָפְלוּ עַל פְּנֵיהֶם(סח) וָמֵתוּ,(סט) וְיָצָא קוֹל שֵׁנִי וְחָיוּ וְעָמְדוּ עַל רַגְלֵיהֶם. וְאָמְרוּ לְמֹשֶׁה, מֹשֶׁה רַבֵּנוּ, אֵין אָנוּ יְכוֹלִים לִשְׁמֹעַ קוֹלוֹ שֶׁל הַקָּדוֹשׁ בָּרוּךְ הוּא,(ע) וּמַתְנוּ כְּשֵׁם שֶׁמַּתְנוּ,(עא) שֶׁנֶּאֱמַר [שה"ש ה, ו] נַפְשִׁי יָצְאָה בְדַבְּרוֹ.(עב) וּכְתִיב [שמות כ, טז] וַיֹּאמְרוּ אֶל מֹשֶׁה דַּבֶּר אַתָּה עִמָּנוּ וְנִשְׁמָעָה.(עג) וְשָׁמַע הַקָּדוֹשׁ בָּרוּךְ הוּא קוֹלָן שֶׁל יִשְׂרָאֵל וְעָרַב לוֹ,(עד) וְשָׁלַח לְמִיכָאֵל וּלְגַבְרִיאֵל וְאָחֲזוּ בְּיָדָיו שֶׁל מֹשֶׁה(עה) שֶׁלֹּא כִרְצוֹנוֹ וְהִגִּישׁוּהוּ אֶל הָעֲרָפֶל, שֶׁנֶּאֱמַר [שם יח] וּמֹשֶׁה נִגַּשׁ(עו) אֶל הָעֲרָפֶל. נוֹגֵשׁ אֵין כְּתִיב כָּאן אֶלָּא נִגַּשׁ.(עז) וּשְׁאָר כָּל הַדְּבָרִים דִּבֵּר עִם מֹשֶׁה.(עח) וְעָלָיו הַכָּתוּב אוֹמֵר כְּצִנַּת שֶׁלֶג בְּיוֹם קָצִיר צִיר נֶאֱמָן לְשֹׁלְחָיו [משלי כה, יג]: וַיְהִי כְּשָׁמְעֲכֶם אֶת הַקּוֹל(עט) מִתּוֹךְ הַחֹשֶׁךְ [דברים ה, כ], וְלָמָּה הִשְׁמִיעַ הַקָּדוֹשׁ בָּרוּךְ הוּא אֶת קוֹלוֹ מִתּוֹךְ הָאֵשׁ וְהַחֹשֶׁךְ וְלֹא מִתּוֹךְ הָאוֹר. לְמָה הַדָּבָר דּוֹמֶה,(פ) לְמֶלֶךְ שֶׁהָיָה לוֹ אַסְטֵרְלוּגוּס, וְהָיָה מַשִׂיא בְּנוֹ אִשָּׁה, וְתָלָה בְּחֻפַּת בְּנוֹ(פא) פַּרְכִּיוֹת שְׁחוֹרוֹת וְלֹא פַּרְכִּיוֹת לְבָנוֹת. אָמְרוּ לוֹ בְּנֵי פַּלְטִין, אֲדוֹנֵנוּ הַמֶּלֶךְ, אֵין אָדָם תּוֹלֶה בְּחֻפַּת בְּנוֹ(פב) אֶלָּא פַּרְכִיוֹת לְבָנוֹת, אָמַר לָהֶם יוֹדֵעַ אֲנִי(פג) שֶׁאֵין בְּנִי מַמְתִּין אֶת אִשְׁתּוֹ אֶלָּא אַרְבָּעִים יוֹם, שֶׁלֹּא יֹאמְרוּ לְמָחָר אַסְטֵרְלוּגוּס לֹא הָיָה יוֹדֵעַ מַה עָתִיד לֵאָרַע לִבְנוֹ. כָּךְ, הַמֶּלֶךְ זֶה הַקָּדוֹשׁ בָּרוּךְ הוּא, וְהַבֵּן שֶׁלּוֹ אֵלּוּ יִשְׂרָאֵל, וְהַכַּלָּה זוֹ הַתּוֹרָה, וְהָיָה הַקָּדוֹשׁ בָּרוּךְ הוּא יוֹדֵעַ שֶׁאֵין יִשְׂרָאֵל מַמְתִּינִין בַּדִּבְּרוֹת אֶלָּא אַרְבָּעִים יוֹם, לְפִיכָךְ הִשְׁמִיעַ לָהֶם הַקָּדוֹשׁ בָּרוּךְ הוּא אֶת קוֹלוֹ מִתּוֹךְ הָאֵשׁ וְהַחֹשֶׁךְ: רַבִּי יְהוּדָה אוֹמֵר, מְדַבֵּר אָדָם עִם חֲבֵרוֹ(פד) הוּא נִרְאֶה וְקוֹלוֹ לֹא נִרְאֶה, וְיִשְׂרָאֵל שָׁמְעוּ קוֹלוֹ שֶׁל הַקָּדוֹשׁ בָּרוּךְ הוּא וְרָאוּ(פה) אֶת הַקּוֹל יוֹצֵא מִפִּי הַגְּבוּרָה וּבְרָקִים וּרְעָמִים, שֶׁנֶּאֱמַר [שמות כ, טו] וְכָל הָעָם רֹאִים אֶת הַקּוֹלֹת וְאֶת הַלַּפִּידִם: כָּל מִצְוֹת שֶׁבַּתּוֹרָה שֵׁשׁ מֵאוֹת וּשְׁלֹשׁ עֶשְׂרֵה מִצְוֹת, וּשְׁתַּיִם שֶׁדִּבֶּר הַקָּדוֹשׁ בָּרוּךְ הוּא,(פו) לְפִיכָךְ נִקְרֵאת תּוֹרָה, וּמִנְיַן תּוֹרָה שֵׁשׁ מֵאוֹת וְאַחַת עֶשְׂרֵה, וּשְׁתַּיִם דִּבֶּר ה',(פז) שֶׁנֶּאֱמַר [תהלים סב, יב] אַחַת דִּבֶּר אֱלֹהִים שְׁתַּיִם זוּ שָׁמָעְתִּי, הֲרֵי שֵׁשׁ מֵאוֹת וּשְׁלשׁ עֶשְׂרֵה: רַבִּי פִּינְחָס אוֹמֵר, כָּל אוֹתוֹ הַדּוֹר שֶׁשָּׁמְעוּ קוֹלוֹ שֶׁל הַקָּדוֹשׁ בָּרוּךְ הוּא בְּהַר סִינַי,(פח) זָכוּ לִהְיוֹת כְּמַלְאֲכֵי הַשָּׁרֵת, לֹא רָאוּ טִפַּת קֶרִי בִּימֵיהֶם, וְלֹא מָשַׁל בָּהֶם כָּל מִינֵי כִנִּים,(צא) וּבְמוֹתָם לֹא שָׁלְטָה בָּהֶם רִמָּה וְתוֹלֵעָה, אַשְׁרֵיהֶם בָּעוֹלָם הַזֶּה(צב) וְאַשְׁרֵיהֶם בָּעוֹלָם הַבָּא.(צג) וַעֲלֵיהֶם הַכָּתוּב אוֹמֵר [שם קמד, טו] אַשְׁרֵי הָעָם שֶׁכָּכָה לּוֹ.

רַבִּי שִׁמְעוֹן בֶּן יוֹחַאי אוֹמֵר,(א) כְּשֶׁנִּגְלָה הַקָּדוֹשׁ בָּרוּךְ הוּא לְמֹשֶׁה מִתּוֹךְ הַסְּנֶה שְׁלָחוֹ לְמִצְרָיִם. אָמַר מֹשֶׁה לִפְנֵי הַקָּדוֹשׁ בָּרוּךְ הוּא, רִבּוֹן כָּל הָעוֹלָמִים הִשָּׁבְעָה לִי(ב) שֶׁכָּל מַה שֶּׁאֲנִי מְבַקֵּשׁ לַעֲשׂוֹת אַתָּה עוֹשֶׂה,(ג) שֶׁלֹּא אֲדַבֵּר דָּבָר לִפְנֵי פַּרְעֹה וְלֹא תַעֲשֶׂה וְיַהַרְגֵנִי. וְנִשְׁבַּע לוֹ שֶׁכָּל מַה שֶּׁהוּא מְבַקֵּשׁ הוּא עוֹשֶׂה,(ד) חוּץ מִשְּׁנֵי דְּבָרִים, מִלְּהִכָּנֵס בָּאָרֶץ, וּמִיּוֹם הַמָּוֶת.(ה) וּמִנַּיִן שֶׁנִּשְׁבַּע לוֹ, שֶׁנֶּאֱמַר [בראשית כב, טז] בִּי נִשְׁבַּעְתִּי נְאֻם ה'. וּכְשֶׁקִּבְּלוּ יִשְׂרָאֵל אֶת הַדִּבְּרוֹת,(ו) לְאַחַר אַרְבָּעִים יוֹם שָׁכְחוּ אֶת אֱלֹהֵיהֶם, וְאָמְרוּ לְאַהֲרֹן, הַמִּצְרִים הָיוּ נוֹשְׂאִין אֶת אֱלֹהֵיהֶם(ח) וְהָיוּ מְשׁוֹרְרִים וּמְזַמְּרִים לְפָנָיו וְרוֹאִין אוֹתוֹ לִפְנֵיהֶם, עֲשֵׂה לָנוּ אֱלֹהִים כֵּאלֹהֵי מִצְרַיִם וְנִרְאֶה אוֹתוֹ לְפָנֵינוּ, שֶׁנֶּאֱמַר [שמות לב, א] וַיֹּאמְרוּ לוֹ קוּם עֲשֵׂה לָנוּ אֱלֹהִים וְגוֹ'. הָלְכוּ לָהֶם(ט) אֵצֶל חֲבֵרָיו שֶׁל מֹשֶׁה, אַהֲרֹן וְחוּר בֶּן אֲחוֹתוֹ. וּמִנַּיִן שֶׁהָיָה חוּר בֶּן אֲחוֹתוֹ, שֶׁנֶּאֱמַר [דברי הימים א, ב, יט] וַתָּמָת עֲזוּבָה וַיִּקַּח לוֹ כָלֵב אֶת אֶפְרָת וַתֵּלֶד לוֹ אֶת חוּר. וְלָמָּה נִקְרָא(י) שְׁמָהּ שֶׁל מִרְיָם אֶפְרָת. בַּת(יא) פָּלָטִינִי בַּת מְלָכִים מִגְּדוֹלֵי הַדּוֹר, שֶׁכָּל נָשִׂיא וְגָדוֹל שֶׁעָמַד בְּיִשְׂרָאֵל נִקְרָא שְׁמוֹ אֶפְרָת,(יב) שֶׁנֶּאֱמַר [מ"א יא, כו] יָרָבְעָם בֶּן נְבָט אֶפְרָתִי, וְדָוִד בֶּן אִישׁ אֶפְרָתִי [ש"א יז, יב]. וְכִי אֶפְרָתִי הָיָה, וַהֲלֹא מִשֵּׁבֶט יְהוּדָה הָיָה. אֶלָּא בֶּן פָּלָטִינִי בֶּן מְלָכִים, מִגְּדוֹלֵי הַדּוֹר הָיָה. וּלְפִי שֶׁהָיָה חוּר מִשֵּׁבֶט יְהוּדָה וּמִגְּדוֹלֵי הַדּוֹר, הִתְחִיל מוֹכִיחַ לְיִשְׂרָאֵל(יג) דְּבָרִים קָשִׁים.(יד) וְהַבְּזוּיִים שֶׁבְּיִשְׂרָאֵל(טו) עָמְדוּ עָלָיו וַהֲרָגוּהוּ.(טז) וְרָאָה אַהֲרֹן לְחוּר(יז) שֶׁנֶּהֱרַג, וּבָנָה מִזְבֵּחַ, שֶׁנֶּאֱמַר [שמות לב, ה] וַיַּרְא אַהֲרֹן, מָה רָאָה, שֶׁנֶּהֱרַג חוּר בֶּן אֲחוֹתוֹ. וּבָנָה מִזְבֵּחַ, שֶׁנֶּאֱמַר [שם ה] וַיִּבֶן מִזְבֵּחַ. דָּן אַהֲרֹן דִּין בֵּינוֹ לְבֵין עַצְמוֹ, אָמַר אִם אֲנִי אוֹמֵר לָהֶם תְּנוּ לִי כֶּסֶף וְזָהָב, מִיָּד הֵם מְבִיאִים, אֶלָּא הֲרֵינִי אוֹמֵר לָהֶם תְּנוּ נִזְמֵי נְשֵׁיכֶם בְּנֵיכֶם וּבְנוֹתֵיכֶם,(יח) וּמִיָּד הַדָּבָר בָּטֵל, שֶׁנֶּאֱמַר [שם ב] וַיֹּאמֶר [אֲלֵהֶם] אַהֲרֹן פָּרְקוּ וְגוֹ'. שָׁמְעוּ הַנָּשִׁים,(יט) וְלֹא רָצוּ, וְלֹא קִבְּלוּ עֲלֵיהֶן לִתֵּן נִזְמֵיהֶן לְבַעֲלֵיהֶן,(כ) אֶלָּא אָמְרוּ לָהֶם(כא) לַעֲשׂוֹת שִׁקוּץ וְתוֹעֵבָה שֶׁאֵין בּוֹ כֹּחַ לְהַצִּיל לֹא נִשְׁמַע לָכֶם. וְנָתַן לָהֶן הַקָּדוֹשׁ בָּרוּךְ הוּא שְׂכָרָן בָּעוֹלָם הַזֶּה,(כב) שֶׁהֵן מְשַׁמְּרוֹת רָאשֵׁי חֳדָשִׁים יוֹתֵר מִן הָאֲנָשִׁים. וְנָתַן לָהֶם שָׂכָר לָעוֹלָם הַבָּא, שֶׁהֵן עֲתִידוֹת לְהִתְחַדֵּשׁ כְּמוֹ רָאשֵׁי חֳדָשִׁים, שֶׁנֶּאֱמַר [תהלים קג, ה] הַמַּשְׂבִּיעַ בַּטּוֹב עֶדְיֵךְ וְגוֹ'. רָאוּ הָאֲנָשִׁים שֶׁלֹּא שָׁמְעוּ הַנָּשִׁים לָהֶם לִתֵּן נִזְמֵיהֶן לְבַעֲלֵיהֶן, מֶה עָשׂוּ, עַד אוֹתָהּ הַשָּׁעָה הָיוּ הַנְּזָמִים בְּאָזְנֵיהֶם,(כג) כְּמַעֲשֵׂה הַמִּצְרִים וּכְמַעֲשֵׂה הָעַרְבִיִים, פָּרְקוּ אֶת נִזְמֵיהֶם אֲשֶׁר בְּאָזְנֵיהֶם וְנָתְנוּ לְאַהֲרֹן, שֶׁנֶּאֱמַר [שמות לב, ג] וַיִּתְפָּרְקוּ כָּל הָעָם אֶת נִזְמֵי הַזָּהָב אֲשֶׁר בְּאָזְנֵיהֶם. אֲשֶׁר בְּאָזְנֵי נְשֵׁיהֶם אֵין כְּתִיב כָּאן, אֶלָּא אֲשֶׁר בְּאָזְנֵיהֶם. וּמָצָא אַהֲרֹן בֵּין הַנְּזָמִים(כד) צִיץ שֶׁל זָהָב, שֶׁכָּתוּב עָלָיו שֵׁם הַקֹּדֶשׁ וְחָרוּת עָלָיו צוּרַת עֵגֶל. וְאוֹתוֹ לְבַד הִשְׁלִיךְ בְּכוּר שֶׁל אֵשׁ, שֶׁנֶּאֱמַר [שם כד] וַיִּתְּנוּ לִי. וָאַשְׁלִיכֵם בָּאֵשׁ אֵין כְּתִיב כָּאן אֶלָּא וָאַשְׁלִכֵהוּ בָאֵשׁ. וַיֵּצֵא הָעֵגֶל הַזֶּה גּוֹעֶה,(כה) וְרָאוּ אוֹתוֹ יִשְׂרָאֵל.(כו) רַבִּי יְהוּדָה אוֹמֵר, סַמָּאֵל [נ"א: שטן] נִכְנַס לְתוֹכוֹ, וְהָיָה גּוֹעֶה לְהַתְעוֹת אֶת יִשְׂרָאֵל,(כז) שֶׁנֶּאֱמַר [ישעיה א, ג] יָדַע שׁוֹר קֹנֵהוּ.(כח) וְרָאוּ אוֹתוֹ כָּל יִשְׂרָאֵל וּנְשָׁקוּהוּ וְהִשְׁתַּחֲווּ לוֹ וַיִּזְבְּחוּ לוֹ: אָמַר הַקָּדוֹשׁ בָּרוּךְ הוּא לְמֹשֶׁה, מֹשֶׁה,(כט) שָׁכְחוּ יִשְׂרָאֵל אֶת כֹּחַ גְּבוּרָתִי שֶׁעָשִׂיתִי עִמָּהֶם בְּמִצְרַיִם וּבְיַם סוּף,(ל) וְעָשׂוּ לָהֶם עֲבוֹדָה זָרָה, שֶׁנֶּאֱמַר [שמות לב, ז] לֶךְ רֵד כִּי שִׁחֵת עַמְּךָ. אָמַר לוֹ לְמֹשֶׁה, לֶךְ רֵד כִּי שִׁחֵת עַמְּךָ,(לא) לֵךְ רֵד מִגְּדֻלָּתֶךָ. אָמַר מֹשֶׁה לְפָנָיו, רִבּוֹנוֹ שֶׁל עוֹלָם, עַד שֶׁלֹּא חָטְאוּ לְפָנֶיךָ(לב) הָיִיתָ קוֹרֵא אוֹתָם עַמִּי, שֶׁנֶּאֱמַר [שם ז, ד] וְהוֹצֵאתִי אֶת צִבְאוֹתַי אֶת עַמִּי. וְעַכְשָׁו שֶׁחָטְאוּ לְפָנֶיךָ אַתָּה אוֹמֵר לִי לֶךְ רֵד כִּי שִׁחֵת עַמְּךָ, עַמְּךָ וְנַחֲלָתְךָ הֵם, שֶׁנֶּאֱמַר [דברים ט, כט] וְהֵם עַמְּךָ וְנַחֲלָתֶךָ.(לג) לָקַח מֹשֶׁה אֶת הַלּוּחוֹת וְהָיָה יוֹרֵד,(לד) וְהָיוּ הַכְּתוּבִין סוֹבְלִין אֶת עַצְמָן וְאֶת מֹשֶׁה עִמָּן.(לה) וּכְשֶׁרָאוּ(לו) אֶת הַתֻּפִּים וְאֶת הַמְּחוֹלוֹת וְאֶת הָעֵגֶל,(לז) בָּרְחוּ הַכְּתוּבִים(לח) וּפָרְחוּ מִן הַלּוּחוֹת, וְנִמְצְאוּ(לט) כְּבֵדִין עַל יְדֵי מֹשֶׁה.(מ) וְלֹא יָכוֹל מֹשֶׁה לִסְבּוֹל אֶת עַצְמוֹ וְלֹא אֶת הַלּוּחוֹת, וְהִשְׁלִיכָן מִיָּדָיו וְנִשְׁתַּבְּרוּ,(מא) שֶׁנֶּאֱמַר [שמות לב, יט] וַיְשַׁבֵּר אֹתָם תַּחַת הָהָר. אָמַר מֹשֶׁה לְאַהֲרֹן(מב) מֶה עָשִׂיתָ לָעָם הַזֶּה,(מג) פָּרַעְתָּ אֹתָם כְּאִשָּׁה(מד) פְּרוּעָה מִזְּנוּת, אָמַר לוֹ(מה) רָאִיתִי מֶה עָשׂוּ לְחוּר, וְיָרֵאתִי הַרְבֵּה מְאֹד: רַבִּי אוֹמֵר, כָּל(מו) הַנְּשִׂיאִים לֹא נִשְׁתַּתְּפוּ בְּמַעֲשֵׂה הָעֵגֶל, שֶׁנֶּאֱמַר [שם כד, יא] וְאֶל אֲצִילֵי בְּנֵי יִשְׂרָאֵל לֹא שָׁלַח יָדוֹ, וְאֵין אֲצִילֵי אֶלָּא הַנְּשִׂיאִים. לְפִיכָךְ זָכוּ לִרְאוֹת פְּנֵי הַשְּׁכִינָה,(מז) שֶׁנֶּאֱמַר [שם י] וַיִּרְאוּ אֵת אֱלֹהֵי יִשְׂרָאֵל. רַבִּי יְהוּדָה אוֹמֵר,(מח) אַף שֵׁבֶט לֵוִי לֹא שִׁתֵּף עַצְמוֹ בְּמַעֲשֵׂה הָעֵגֶל, שֶׁנֶּאֱמַר [שם לב, כו] וַיַּעֲמֹד מֹשֶׁה בְּשַׁעַר הַמַּחֲנֶה וְגוֹ' וַיֵּאָסְפוּ אֵלָיו כָּל בְּנֵי לֵוִי.(מט) רָאָה מֹשֶׁה שֶׁשֵּׁבֶט לֵוִי לֹא נִשְׁתַּתֵּף עִמָּהֶם, מִיָּד נִתְחַזֵּק וְנִתְגַבֵּר, וְלָקַח אֶת הָעֵגֶל וּשְׂרָפוֹ בָאֵשׁ(נ) וְכִתְּתוֹ כַּעֲפַר הָאָרֶץ,(נא) וְהִשְׁלִיךְ אֶת עֲפָרוֹ עַל פְּנֵי הַמַּיִם, שֶׁנֶּאֱמַר [שם כ] וַיִּקַּח אֶת הָעֵגֶל אֲשֶׁר עָשׂוּ. וְהָיָה מַשְׁקֶה לְיִשְׂרָאֵל,(נב) וְכָל מִי שֶׁהָיָה נוֹשֵׁק אֶת הָעֵגֶל בְּכָל לִבּוֹ, הָיוּ שְׂפָתָיו נַעֲשׂוֹת שֶׁל זָהָב, וְשֵׁבֶט לֵוִי הָיוּ הוֹרְגִין אוֹתוֹ, עַד שֶׁנָּפְלוּ מִיִשְׂרָאֵל כִּשְׁלשֶׁת אַלְפֵי אִישׁ. שָׁלַח הַקָּדוֹשׁ בָּרוּךְ הוּא(נג) חֲמִשָּׁה מַלְאָכִים לְהַשְׁחִית אֶת כָּל יִשְׂרָאֵל, וְאֵלּוּ הֵן: קֶצֶף, אַף, חֵימָה, מַשְׁחִית, חָרוֹן.(נד) שָׁמַע מֹשֶׁה(נה) וְיָצָא לִקְרַאת אַבְרָהָם יִצְחָק יַעֲקֹב וְאָמַר,(נו) אִם אַתֶּם מִבְּנֵי הָעוֹלָם הַבָּא עִמְדוּ לְפָנַי בַּשָּׁעָה הַזֹּאת, שֶׁהֲרֵי בְּנֵיכֶם נִתְּנוּ(נז) כְּצֹאן לְטִבְחָה,(נח) וְעָמְדוּ שָׁם לְפָנָיו שְׁלֹשֶׁת הָאָבוֹת. אָמַר מֹשֶׁה לְפָנָיו, רִבּוֹן כָּל הָעוֹלָמִים, לֹא כָךְ נִשְׁבַּעְתָּ לָאֵלּוּ לְהַרְבּוֹת אֶת זַרְעָם כְּכוֹכְבֵי הַשָּׁמַיִם, שֶׁנֶּאֱמַר [שם יג] זְכֹר לְאַבְרָהָם לְיִצְחָק וּלְיִשְׂרָאֵל וְגוֹ'. וּבִזְכוּת שְׁלֹשֶׁת הָאָבוֹת(נט) נֶעֶצְרוּ שְׁלֹשָׁה מַלְאָכִים מִיִשְׂרָאֵל, קֶצֶף אַף וְחֵימָה, וְנִשְׁתַּיְּרוּ שְׁנָיִם. אָמַר לְפָנָיו רִבּוֹנוֹ שֶׁל עוֹלָם, לְמַעַן הַשְּׁבוּעָה שֶׁנִּשְׁבַּעְתָּ לָהֶם עֲצֹר מַשְׁחִית מִיִּשְׂרָאֵל, שֶׁנֶּאֱמַר [שם] אֲשֶׁר נִשְׁבַּעְתָּ לָהֶם בָּךְ,(ס) וְנֶעֱצַר הַמַּשְׁחִית, שֶׁנֶּאֱמַר [תהלים עח, לח](סא) וְהוּא רַחוּם יְכַפֵּר עָוֹן וְלֹא יַשְׁחִית. וְעוֹד אָמַר מֹשֶׁה לְפָנָיו, לְמַעַן הַשְּׁבוּעָה שֶׁנִּשְׁבַּעְתָּ לִי עֲצֹר חָרוֹן מִיִּשְׂרָאֵל, שֶׁנֶּאֱמַר [שמות לב, יב] שׁוּב מֵחֲרוֹן אַפֶּךָ. מֶה עָשָׂה מֹשֶׁה, חָפַר בָּאָרֶץ(סב) כְּבֵית דִּירָה גְדוֹלָה,(סג) בְּנַחֲלַת בְּנֵי גָּד,(סד) וְטָמַן חֲרוֹן אַף בָּאָרֶץ כְּאָדָם שֶׁהוּא חָבוּשׁ בְּבֵית הָאֲסוּרִים. וּבְכָל זְמַן שֶּׁהָיוּ יִשְׂרָאֵל חוֹטְאִין,(סה) הוּא עוֹלֶה וּפוֹעֵר אֶת פִּיו לִנְשׁוֹךְ בְּרוּחוֹ וּלְהַשְׁחִית אֶת יִשְׂרָאֵל, לְפִיכָךְ נִקְרָא שְׁמוֹ פְּעוֹר.(סו) וְהָיָה מֹשֶׁה מַזְכִּיר עָלָיו אֶת הַשֵּׁם, וּמוֹרִידוֹ לְמַטָּה לָאָרֶץ. וּכְשֶׁמֵּת מֹשֶׁה, מֶה עָשָׂה הַקָּדוֹשׁ בָּרוּךְ הוּא, נָתַן אֶת קִבְרוֹ כְּנֶגְדּוֹ,(סז) וְכָל זְמַן שֶׁיִּשְׂרָאֵל חוֹטְאִין וְהוּא פוֹעֵר אֶת פִּיו לִנְשׁוֹךְ בְּרוּחוֹ וּלְהַשְׁמִיד אֶת יִשְׂרָאֵל, הוּא רוֹאֶה קִבְרוֹ שֶׁל מֹשֶׁה כְּנֶגְדּוֹ,(סח) הוּא מִתְפַּחֵד וְחוֹזֵר לַאֲחוֹרָיו, שֶׁנֶּאֱמַר [דברים לד, ו] וַיִּקְבֹּר אֹתוֹ בַּגַּי מוּל בֵּית פְּעוֹר.

רַבִּי שִׁמְעוֹן בֶּן יוֹחַאי אוֹמֵר,(א) רָצָה הַקָּדוֹשׁ בָּרוּךְ הוּא לְהַכְרִית וּלְהַשְׁמִיד כָּל זַרְעוֹ שֶׁל עֲמָלֵק,(ב) וְשָׁלַח לְשָׁאוּל בֶּן קִישׁ לְהַכְרִית וּלְהַשְׁמִיד כָּל זַרְעוֹ שֶׁל עֲמָלֵק.(ג) וְשָׁאוּל וְהָעָם שָׁמְעוּ(ד) וְלֹא חָסוּ עַל אָדָם בָּזוּי אֶלָּא עַל אֲגָג, שֶׁנֶּאֱמַר [ש"א טו, ט] וַיַּחְמֹל שָׁאוּל וְהָעָם עַל אֲגָג וְעַל מֵיטַב הַצֹּאן וְהַבָּקָר.(ה) וְשָׁמַע שְׁמוּאֵל וְיָצָא לִקְרָאתָם וְאָמַר לָהֶם,(ו) חֲמַלְתֶּם עַל עֲמָלֵק וְהוֹתַרְתֶּם שָׂרִיד מִמֶּנוּ,(ז) אָמְרוּ לוֹ, הַצֹּאן וְהַבָּקָר לַעֲשׂוֹת זְבָחִים לַה' אֱלֹהֶיךָ,(ח) אָמַר לָהֶם, אֵין חֵפֶץ לַה' בְּעוֹלוֹת וּזְבָחִים כִּי אִם לִשְׁמֹעַ בְּקוֹלוֹ, שֶׁנֶּאֱמַר [שם כב] וַיֹּאמֶר שְׁמוּאֵל הַחֵפֶץ לַה' בְּעֹלוֹת וּזְבָחִים כִּשְׁמֹעַ בְּקוֹל ה' הִנֵּה שְׁמֹעַ מִזֶּבַח טוֹב וְגוֹ': רַבִּי פִּינְחָס אוֹמֵר,(ט) צָפָה הַקָּדוֹשׁ בָּרוּךְ הוּא שֶׁעָתִיד לַעֲמֹד מֵאֲגָג(י) אִישׁ צַר(יא) וְאוֹיֵב גָּדוֹל לַיְּהוּדִים. וְאֵיזֶה, זֶה הָמָן הָרָשָׁע, שֶׁנֶּאֱמַר [אסתר ט, כד] כִּי הָמָן בֶּן הַמְּדָתָא הָאֲגָגִי צֹרֵר כָּל הַיְּהוּדִים. וּמִזֶּרַע שָׁאוּל גּוֹאֵל וְנוֹקֵם לַיְּהוּדִים מִיַּד הָמָן, וְאֵיזֶה, זֶה מָרְדְּכַי,(יג) שֶׁנֶּאֱמַר [שם ב, ה] אִישׁ יְהוּדִי הָיָה בְּשׁוּשַׁן הַבִּירָה.(יד) וְשָׁם עָמַד שְׁמוּאֵל מִתְפַּלֵּל וְאוֹמֵר, רִבּוֹן כָּל הָעוֹלָמִים אַל תִּשְׁכַּח צָרַת עֵשָׂו הָרָשָׁע שֶׁעָשָׂה לְאָבִיו, וְלָקַח נָשִׁים נָכְרִיּוֹת מְזַבְּחוֹת וּמְקַטְּרוֹת לַעֲבוֹדָה זָרָה לְמָרֵר חַיֵּי אֲבוֹתָיו, שֶׁנֶּאֱמַר [בראשית כו, לה] וַתִּהְיֶיןָ מֹרַת רוּחַ לְיִצְחָק וּלְרִבְקָה. זְכֹר חַטָּאתוֹ עַל בָּנָיו וְעַל בְּנֵי בָנָיו עַד סוֹף כָּל הַדּוֹרוֹת, שֶׁנֶּאֱמַר [תהלים קט, יד] יִזָּכֵר עֲוֹן אֲבוֹתָיו אֶל ה'.(כ) וְשָׁמַע שְׁמוּאֵל אֶת קוֹל אֲגָג לוֹחֵשׁ בְּפִיו וְאוֹמֵר אֶפְשָׁר שֶׁסָר מִמֶּנִּי הַמָּוֶת הָרַע, שֶׁנֶּאֱמַר [ש"א טו, לב] אָכֵן סָר מַר הַמָּוֶת. אָמַר לוֹ שְׁמוּאֵל, כַּאֲשֶׁר שִׁכְּלָה נָשִׁים חַרְבֶּךָ כֵּן תִּשְׁכַּל מִנָשִׁים אִמֶּךָ [שם טו, לג], כְּשֵׁם שֶׁשִּׁכְּלָה חֶרֶב עֲמָלֵק זְקֵנְךָ אֶת(כב) בַּחוּרֵי בְנֵי יִשְׂרָאֵל מֵאַחֲרֵי הֶעָנָן,(כג) וְיָשְׁבוּ נְשֵׁיהֶם שְׁכוּלוֹת וְאַלְמָנוֹת, כֵּן תִּשְׁכַּל מִנָשִׁים אִמֶּךָ.(כד) וּבִתְפִלַּת אֶסְתֵּר(כה) וְנַעֲרוֹתֶיהָ נֶהֶרְגוּ כָּל בַר עֲמָלֵק, וְיָשְׁבוּ נְשֵׁיהֶם שְׁכוּלוֹת וְאַלְמָנוֹת, שֶׁנֶּאֱמַר [שם] וַיֹּאמֶר שְׁמוּאֵל כַּאֲשֶׁר שִׁכְּלָה נָשִׁים חַרְבֶּךָ.(כו) וּתְפִלַּת שְׁמוּאֵל שִׁבֵּר חֵילָם שֶׁל בְּנֵי אֲגָג מֵעַל יִשְׂרָאֵל בַּגִּלְגָּל, שֶׁנֶּאֱמַר וַיְשַׁסֵּף שְׁמוּאֵל אֶת אֲגָג לִפְנֵי ה' בַּגִּלְגָּל:(כז) בַּכֹּל הוּא עוֹשֶׂה שְׁלִיחוּתוֹ,(כח) בַּגִּבּוֹרִים הוּא שׁוֹלֵחַ עֲלֵיהֶם דְּבָרִים נְמוּכִים, לְהוֹדִיעָן שֶׁאֵין כֹּחַ גְּבוּרָתָן מְאוּמָה.(כט) טִיטוֹס הָרָשָׁע נִכְנַס לְבֵית קָדְשֵׁי הַקֳּדָשִׁים, וְאָמַר אֵין צַר וְאוֹיֵב יָכוֹל לְפָנָי. מֶה עָשָׂה הַקָּדוֹשׁ בָּרוּךְ הוּא, שָׁלַח עָלָיו יִתּוּשׁ אֶחָד(ל) לְתוֹךְ חָטְמוֹ, וְהָיָה מְהַלֵּךְ עַד שֶׁהִגִיעַ לַמֹּחַ, וְנַעֲשָׂה אוֹתוֹ יִתּוּשׁ כְּמוֹ גוֹזָל בֶּן יוֹנָה מִשְׁקַל שְׁנֵי לִיטְרִין, לְהוֹדִיעוֹ שֶׁאֵין בְּכֹחַ גְּבוּרָתוֹ מְאוּמָה. וְיִשְׂרָאֵל(לא) שֶׁנִּכְנְסוּ לְבֵית קָדְשֵׁי הַקֳּדָשִׁים(לב) בְּגַּאֲוַת לִבָּם,(לג) וְהָיוּ אוֹמְרִים אֵין צַר וְאוֹיֵב יָכוֹל לַעֲמֹד לְפָנֵינוּ, מֶה עָשָׂה הַקָּדוֹשׁ בָּרוּךְ הוּא, שָׁלַח עֲלֵיהֶם נְבוּכַדְנֶצַּר הָרָשָׁע,(לד) שֶׁהָיָה לָהֶם אִישׁ צַר(לה) וּגְבַהּ רוּחַ, לְהוֹדִיעַ לַכֹּל כִּי לֹא בְכֹחַ יִגְבַּר אִישׁ [שם ב, ט]: רַבִּי חֲכִינַאי אוֹמֵר, לֹא(לו) נָתַן הַקָּדוֹשׁ בָּרוּךְ הוּא קִצְבָה לַמַּלְכֻיוֹת אֶלָּא(לז) לְשִׁעְבּוּד מִצְרַיִם, שֶׁנֶּאֱמַר [בראשית טו, יג] וַעֲבָדוּם וְעִנּוּ אֹתָם אַרְבַּע מֵאוֹת שָׁנָה.(לח) וְעָשָׂה עִמָּהֶם הַקָּדוֹשׁ בָּרוּךְ הוּא כְּרֹב רַחֲמָיו וּכְרֹב חֲסָדָיו(לט) וְקִצֵּר חֶצְיוֹ כְּמִנְיַן מָאתַיִם וְעֶשֶׂר שָׁנִים. וּלְמַלְכוּת בָּבֶל שִׁבְעִים שָׁנָה, שֶׁנֶּאֱמַר [ירמיה כט, י] כִּי לְפִי מְלֹאת לְבָבֶל שִׁבְעִים שָׁנָה: רַבִּי אַבָּהוּ אוֹמֵר, אַרְבָּעִים וְחָמֵשׁ שָׁנִים מָלַךְ נְבוּכַדְנֶצַּר. תֵּדַע לְךָ שֶׁהוּא כֵן, בֹּא וּרְאֵה,(מ) בְּשָׁנָה שֶׁמָלַךְ עָלָה לִירוּשָׁלַיִם וְכָבַשׁ אֶת יְהוֹיָקִים מֶלֶךְ יְהוּדָה, שֶׁנֶּאֱמַר [דניאל א, א] בִּשְׁנַת שָׁלֹשׁ לְמַלְכוּת יְהוֹיָקִים מֶלֶךְ יְהוּדָה בָּא נְבוּכַדְנֶאצַּר מֶלֶךְ בָּבֶל יְרוּשָׁלַיִם וַיָּצַר עָלֶיהָ. וּשְׁמוֹנֶה שָׁנִים מָלַךְ עַל מַלְכוּת יְהוֹיָקִים, וְאַחַת עֶשְׂרֵה שָׁנִים מָלַךְ עַל מַלְכוּת צִדְקִיָּהוּ, הֲרֵי תְּשַׁע עֶשְׂרֵה עַד שֶׁלֹּא נֶחֱרַב הַבָּיִת. וּמִשֶׁנֶּחֱרַב הַבַּיִת עֶשְׂרִים וָשֵׁשׁ שָׁנָה, תֵּדַע לְךָ שֶׁהוּא כֵן, בֹּא וּרְאֵה, מִן גָּלוּת יְהוֹיָכִין עַד שֶׁמָלַךְ אֱוִיל מְרוֹדַךְ בְּנוֹ שְׁלֹשִׁים וָשֶׁבַע שָׁנָה, שֶׁנֶּאֱמַר [מ"ב כה, כז] וַיְהִי(מא) בִשְׁלֹשִׁים וְשֶׁבַע שָׁנָה לְגָלוּת יְהוֹיָכִין מֶלֶךְ יְהוּדָה בִּשְׁנֵים עָשָׂר חֹדֶשׁ בְּעֶשְׂרִים וְשִׁבְעָה לַחֹדֶשׁ נָשָׂא אֱוִיל מְרֹדַךְ מֶלֶךְ בָּבֶל וְגוֹ' אֶת רֹאשׁ יְהוֹיָכִין מֶלֶךְ יְהוּדָה מִבֵּית כֶּלֶא. רַבִּי יוֹנָתָן אוֹמֵר,(מב) סוֹף מַלְכֵי מָדַי וּפָרַס הָיָה אַרְתַּחְשַׁסְתָּא מֶלֶךְ בָּבֶל,(מג) וּשְׁתֵּים עֶשְׂרֵה שָׁנָה מָלַךְ, שֶׁנֶּאֱמַר [נחמיה יג, ו] וּבְכָל זֶה לֹא הָיִיתִי בִּירוּשָׁלָיִם וגו': רַבִּי תַנְחוּם אוֹמֵר, בֹּא וּרְאֵה עָשְׁרוֹ שֶׁל אֲחַשְׁוֵרוֹשׁ, שֶׁהָיָה עָשִׁיר מִכָּל מַלְכֵי מָדַי וּפָרָס,(מד) וְעָלָיו הַכָּתוּב אוֹמֵר וְהָרְבִיעִי יַעֲשִׁיר עֹשֶׁר גָּדוֹל [דניאל יא, ב]. וּמֶה הָיָה עָשְׁרוֹ שֶׁל אֲחַשְׁוֵרוֹשׁ, שֶׁהֶעֱמִיד מִטּוֹת זָהָב וָכֶסֶף(מה) בִּרְחוֹב הָעִיר,(מו) לְהוֹדִיעַ לְכָל הָעוֹלָם אֶת עָשְׁרוֹ, שֶׁנֶּאֱמַר [אסתר א, ו] מִטּוֹת זָהָב וָכֶסֶף. וְכָל כְּלֵי תַשְׁמִישׁוֹ שֶׁל אֲחַשְׁוֵרוֹשׁ(מז) לֹא הָיָה בִּכְלֵי כֶסֶף אֶלָּא בִּכְלֵי זָהָב.(מח) וְהֵבִיא כְּלֵי בֵית הַמִּקְדָּשׁ, וּכְלֵי בֵיתוֹ הָיוּ מִשְׁתַּנִּים(מט) וְנַעֲשִׂים כַּעוֹפֶרֶת, שֶׁנֶּאֱמַר [שם ז] וְכֵלִים מִכֵּלִים שׁוֹנִים.(נ) וְכֹל רִצְפַּת פַּלְטֵרִין שֶׁלּוֹ(נא) אֲבָנִים טוֹבוֹת(נב) וּמַרְגָּלִיוֹת, שֶׁנֶּאֱמַר [אסתר א, ו] עַל רִצְפַּת בַּהַט וָשֵׁשׁ וְדַר וְסֹחָרֶת: רַבִּי אֱלִיעֶזֶר אוֹמֵר,(נג) חֲצִי שָׁנָה עָשָׂה אֲחַשְׁוֵרוֹשׁ(נד) סְעוּדוֹת גְּדוֹלוֹת לְכָל הָעַמִּים, שֶׁנֶּאֱמַר [שם ד] יָמִים רַבִּים שְׁמוֹנִים וּמְאַת יוֹם. וְכָל מִי שֶׁהָיָה מַאֲכָלוֹ(נה) בְּטָהֳרָה,(נו) הָיוּ נוֹתְנִין לוֹ מַאֲכָל בְּטָהֳרָה. וְכָל מִי שֶׁהָיָה מַאֲכָלוֹ בְּטֻמְאָה, נוֹתְנִין לוֹ מַאֲכָל בְּטֻמְאָה, לְקַיֵּם מַה שֶׁנֶּאֱמַר [שם ח](נז) לַעֲשׂוֹת כִּרְצוֹן אִישׁ וָאִישׁ. רַבִּי שִׁמְעוֹן אוֹמֵר, כָּל מִנְהָגָן(נח) שֶׁל מַלְכֵי מָדַי, כְּשֶׁהָיוּ אוֹכְלִין וְשׁוֹתִין הָיוּ מְבִיאִין לִנְשֵׁיהֶם גְּלוּיוֹת בְּשַׂר עֶרְוָה(ס) מְשַׂחֲקוֹת וּמְרַקְדוֹת לִפְנֵיהֶם, כְּדֵי לִרְאוֹת בְּיֹפִי תָאֳרָן.(סב) וּכְשֶׁנִּכְנָס הַיַּיִן בְּלִבּוֹ שֶׁל אֲחַשְׁוֵרוֹשׁ, בִּקֵּשׁ לַעֲשׂוֹת כֵּן לְוַשְׁתִּי,(סג) וּבַת מְלָכִים הָיְתָה(סד) וְלֹא קִבְּלָה עָלֶיהָ לַעֲשׂוֹת כֵּן שֶׁתָּבֹא לְפָנָיו עֲרֻמָּה. וְשָׁלְחָה וְאָמְרָה לוֹ אִישׁ שׁוֹטֶה שִׁכּוֹר, אִם אָבוֹא וְיִרְאוּ אוֹתִי שֶׁאֲנִי כְּעוּרָה יָבוּזוּ לְךָ בְּשֻׁלְחָנֶךָ, וְאִם יִרְאוּ אוֹתִי שֶׁאֲנִי יְפַת תֹּאַר, יַהֲרֹג אוֹתְךָ [נ"א: יחמוד אותי] אֶחָד מִן הַשָׂרִים [נ"א: הקרואים] שֶׁלְּךָ בִּשְׁעַת הַיָּיִן.(סו) שָׁמַע הַמֶּלֶךְ לִדְּבָרֶיהָ(סז) וְצִוָּה לְהָרְגָהּ.(סח) וּכְשֶׁיָּצָא הַיַּיִן מִלִּבּוֹ,(סט) בִּקֵּשׁ לְוַשְׁתִּי הַמַּלְכָּה וְלֹא מְצָאָהּ,(ע) אָמְרוּ לוֹ שָׂרָיו הַגְּזֵירָה שֶׁגָּזַר עָלֶיהָ,(עא) וְהִתְחִיל בּוֹכֶה עַל הַגְּזֵירָה שֶׁגָּזַר עָלֶיהָ.(עב) וְלָמָה נִגְזַר עָלֶיהָ אוֹתָהּ גְּזֵרָה, לְפִי שֶׁהָיְתָה מְבִיאָה אֶת בְּנוֹת יִשְׂרָאֵל עֲרֻמּוֹת, וְעוֹשִׂין לָהּ מְלָאכוֹת בַּשַּׁבָּת, וְעַל זֶה נִגְזַר עָלֶיהָ גְּזֵרָה בְּשַּׁבָּת שֶׁתִּשָׁחֵט עֲרֻמָּה, שֶׁנֶּאֱמַר [שם ב, א] זָכַר אֶת וַשְׁתִּי וְאֵת אֲשֶׁר עָשָׂתָה וְגוֹ': רַבִּי זְכַרְיָה אוֹמֵר, מְגַלְגְּלִין זְכוּת עַל יְדֵי זַכַּאי, עַל יְדֵי(עז) דָּנִיֵּאל שֶׁהָיָה מְמוּכָן, נִתְגַּלְגְּלָה הַמַּלְכוּת עַל אֶסְתֵּר.(עח) לְפִי שֶׁאָמַר לַמֶּלֶךְ אַל תִּבְכֶּה, שֶׁכָּל מַה שֶׁעָשִׂיתָ לְוַשְׁתִּי(עט) כַּתּוֹרָה עָשִׂיתָ, וְכָל הַמְקַיֵּם מִצְוַת הַתּוֹרָה, הַקָּדוֹשׁ בָּרוּךְ הוּא מְקַיֵּם מַלְכוּתוֹ,(פא) שֶׁכָּךְ כָּתוּב בַּתּוֹרָה וְהוּא יִמְשָׁל בָּךְ [בראשית ג, טז].(פג) וְשָׁלַח הַמֶּלֶךְ בְּכָל הַמְּדִינוֹת לַעֲשׂוֹת כִּדְבַר מְמוּכָן, שֶׁנֶּאֱמַר [אסתר א, כב] לִהְיוֹת כָּל אִישׁ שׂרֵר בְּבֵיתוֹ.(פד) וְעוֹד אָמַר יְבַקְשׁוּ לַמֶּלֶךְ נְעָרוֹת בְּתוּלוֹת וְגוֹ' [שם ב, ב].(פה) נְעָרוֹת וְלֹא כָּל נְעָרוֹת, אֶלָּא נְעָרוֹת יְפוֹת תֹּאַר, שֶׁנֶּאֱמַר [שם ד] וְהַנַּעֲרָה אֲשֶׁר תִּיטַב בְּעֵינֵי הַמֶּלֶךְ. וּכְתִיב לְהַלָּן וַתִּיטַב הַנַּעֲרָה בְּעֵינָיו [שם ט].(פו) וְהִטָּה הַקָּדוֹשׁ בָּרוּךְ הוּא חֵן וָחֶסֶד עָלֶיהָ בְּעֵינֵי כָּל רוֹאֶיהָ, שֶׁנֶּאֱמַר [שם טו] וַתְּהִי אֶסְתֵּר נֹשֵׂאת חֵן בְּעֵינֵי כָּל רֹאֶיהָ.

כָּל הַמַּלְשִׁין אָדָם בַּסֵּתֶר [נ"א: כל(א) המספר לשון הרע](ב) אֵין לוֹ חֵלֶק לָעוֹלָם הַבָּא, שֶׁנֶּאֱמַר [תהלים קא, ה] מְלָשְׁנִי בַסֵּתֶר רֵעֵהוּ אוֹתוֹ אַצְמִית.(ג) וְכָתוּב אֶחָד אוֹמֵר אָרוּר מַכֵּה רֵעֵהוּ בַּסָּתֶר [דברים כז, כד].(ד) תֵּדַע לְךָ שֶׁהוּא כֵּן, בֹּא וּרְאֵה מִן הַנָּחָשׁ(ה) שֶׁהִלְשִׁין לְאָדָם וּלְעֶזְרוֹ [נ"א: בין אדם להקדוש ברוך הוא], וְאֵרֲרוֹ הַקָּדוֹשׁ בָּרוּךְ הוּא שֶׁיְּהֵא מַאֲכָלוֹ עָפָר, שֶׁנֶּאֱמַר [בראשית ג, יד] וְעָפָר תֹּאכַל כָּל יְמֵי חַיֶּיךָ. רַבָּן גַּמְלִיאֵל אוֹמֵר, יִשְׂרָאֵל אָמְרוּ לָשׁוֹן הָרָע עַל הַקָּדוֹשׁ בָּרוּךְ הוּא,(ו) וְאָמְרוּ הַיֵּשׁ בַּה' כֹּחַ לָזוּן אוֹתָנוּ בַּמִּדְבָּר, שֶׁנֶּאֱמַר [תהלים עח, יט - כ](ז) וַיְדַבְּרוּ בֵאלֹהִים אָמְרוּ הֲיוּכַל אֵל לַעֲרֹךְ שֻׁלְחָן בַּמִּדְבָּר הֵן הִכָּה צוּר וַיָּזוּבוּ מַיִם וּנְחָלִים יִשְׁטֹפוּ. וְרָאָה הַקָּדוֹשׁ בָּרוּךְ הוּא(ח) שֶׁהִלְשִׁינוּ בִּכְבוֹדוֹ, וּכְבוֹדוֹ הוּא אֵשׁ אוֹכְלָה, שָׁלַח בָּהֶם אֵשׁ וְאָכְלָה אוֹתָם(ט) סָבִיב, שֶׁנֶּאֱמַר [במדבר יא, א] [וַיְהִי הָעָם כְּמִתְאוֹנְנִים רַע בְּעֵינֵי ה'] וַתִּבְעַר בָּם אֵשׁ ה' וַתֹּאכַל בִּקְצֵה הַמַּחֲנֶה. הָלְכוּ לָהֶם זִקְנֵי יִשְׂרָאֵל אֵצֶל מֹשֶׁה וְאָמְרוּ לוֹ,(י) מֹשֶׁה רַבֵּנוּ, תְּנֵנוּ כְּצֹאן לְטִבְחָה וְלֹא לְאֵשׁ אוֹכְלָה.(יא) וְרָאָה מֹשֶׁה בְּצָרָתָן שֶׁל יִשְׂרָאֵל עָמַד וְהִתְפַּלֵּל עֲלֵיהֶם [וְנֶעֱתַּר לוֹ],(יב) שֶׁנֶּאֱמַר [שם ב] וַיִצְעַק מֹשֶׁה אֶל ה' וַתִּשְׁקַע הָאֵשׁ: רַבִּי יְהוּדָה אוֹמֵר, אוֹתָהּ הָאֵשׁ שֶׁיָּרְדָה מִן הַשָּׁמַיִם,(יג) שָׁקְעָה בָאָרֶץ וְלֹא חָזְרָה לִמְקוֹמָהּ לַשָּׁמַיִם, אֶלָּא(יד) נִכְנְסָה לְאֹהֶל מוֹעֵד.(טו) וְכָל הַקָּרְבָּנוֹת שֶׁהָיוּ מַקְרִיבִין יִשְׂרָאֵל בַּמִּדְבָּר, [אוֹתָה אֵשׁ] הָיְתָה יוֹצֵאת וְאוֹכֶלֶת אוֹתָן, שֶׁנֶּאֱמַר [ויקרא י, ב] וַתֵּצֵא אֵשׁ מִלִּפְנֵי ה'.(טז) וַתֵּרֶד אֵשׁ אֵין כְּתִיב כָּאן, אֶלָּא(יז) וַתֵּצֵא אֵשׁ מִלִּפְנֵי ה' וְגוֹ'. [וְאוֹתָה הָאֵשׁ יָצָאתָה וְאָכְלָה בְּנֵי אַהֲרֹן, שֶׁנֶּאֱמַר וַתֵּצֵא אֵשׁ] וְהִיא הָאֵשׁ שֶׁאָכְלָה לַעֲדַת קֹרַח, שֶׁנֶּאֱמַר [במדבר טז, לה] וְאֵשׁ יָצְאָה מֵאֵת ה'. וְאֵין אָדָם נִפְטָר מִן הָעוֹלָם עַד שֶׁיַעֲבֹר עָלָיו(יח) מֵאוֹתָהּ הָאֵשׁ שֶׁשָּׁקְעָה בָאָרֶץ, שֶׁנֶּאֱמַר [שם יא, ב] וַתִּשְׁקַע הָאֵשׁ: וַתְּדַבֵּר מִרְיָם וְאַהֲרֹן בְּמֹשֶׁה עַל אֹדוֹת הָאִשָּׁה הַכֻּשִׁית אֲשֶׁר לָקָח וְגוֹ' [שם יב, א].(יט) וְכִי כֻּשִׁית הָיְתָה, אֶלָּא מָה הַכֻּשִׁי הַזֶּה מְשֻׁנֶּה בְעוֹרוֹ, כָּךְ הָיְתָה(כ) צִפּוֹרָה מְשֻׁנָּה בְּמַעֲשֶׂיהָ הַטּוֹבִים, [לְפִיכָךְ נִקְרֵאת כּוּשִׁית שֶׁנֶּאֱמַר כִּי אִשָׁה כּוּשִׁית לָקָח]. רַבִּי תַנְחוּם אוֹמֵר, אַף יִשְׂרָאֵל נִקְרְאוּ כּוּשִׁים, שֶׁנֶּאֱמַר [עמוס ט, ז] הֲלוֹא כִּבְנֵי כֻשִׁיִּים אַתֶּם לִי, [וְכִי כּוּשִׁים הָיוּ, אֶלָא] מָה הַכֻּשִׁי הַזֶּה [גוּפוֹ] מְשֻׁנֶּה מִכָּל הַבְּרִיּוֹת, כָּךְ יִשְׂרָאֵל מְשֻׁנִּים [בְּדַרְכֵיהֶם] וּבְמַעֲשֵׂיהֶם הַטּוֹבִים מִכָּל אֻמּוֹת הָעוֹלָם, לְכָךְ נִקְרְאוּ כּוּשִׁים. וְכָתוּב אֶחָד אוֹמֵר [ירמיה לח, ז] וַיִשְׁמַע עֶבֶד מֶלֶךְ הַכּוּשִׁי,(כא) וְכִי עֶבֶד כּוּשִׁי הָיָה, וַהֲלֹא(כב) בָּרוּךְ בֶּן נֵרִיָּה הָיָה, אֶלָּא מָה הַכּוּשִׁי הַזֶּה מְשֻׁנֶּה בְעוֹרוֹ, כָּךְ בָּרוּךְ בֶּן נֵרִיָּה מְשֻׁנֶּה בְּמַעֲשָׂיו הַטּוֹבִים מִשְּׁאָר בְּנֵי אָדָם. [וְכָתוּב אֶחָד אוֹמֵר] וַיֹּאמֶר יוֹאָב לַכּוּשִׁי לֵךְ הַגֵּד לַמֶּלֶךְ [ש"ב יח, כא],(כג) וְכִי כּוּשִׁי הָיָה וַהֲלֹא בֶּן יְמִינִי הָיָה, אֶלָּא מָה הַכּוּשִׁי הַזֶּה [גוּפוֹ] מְשֻׁנֶּה מִכָּל הַבְּרִיּוֹת, כָּךְ הָיָה בֶּן יְמִינִי מְשֻׁנֶּה בְּמַעֲשָׂיו וּבִדְרָכָיו הַטּוֹבִים מִכָּל יִשְׂרָאֵל. רַבִּי אֱלִיעֶזֶר אוֹמֵר בֹּא וּרְאֵה יַשְׁרוּתוֹ וְתֻמָּתוֹ(כד) שֶׁל אוֹתוֹ הָאִישׁ שֶׁאָמַר לְיוֹאָב, אִם תִּתֵּן אֶלֶף כֶּסֶף אֵינִי עוֹבֵר עַל מִצְוַת הַמֶּלֶךְ אֲשֶׁר צִוְּךָ, שֶׁנֶּאֱמַר [ש"ב יח, יב] וְלֻא אָנֹכִי שֹׁקֵל עַל כַּפַּי אֶלֶף כֶּסֶף. אָמַר לוֹ [יוֹאָב] בְּבַקָּשָׁה מִמְךָ הַרְאֵנִי אוֹתוֹ הַמָּקוֹם שֶׁאַבְשָׁלוֹם תָּלוּי בּוֹ. וְלֹא קִבֵּל עָלָיו.(כה) הִתְחִיל כּוֹרֵעַ וּמִשְׁתַּחֲוֶה לוֹ, [שֶׁנֶּאֱמַר [שם יד] וַיֹאמֶר] (אָמַר לוֹ) יוֹאָב לֹא כֵן אֹחִילָה לְפָנֶיךָ [שם], וְלָקַח אוֹתוֹ בִּזְרוֹעַ(וֹ)(כו) וְהֶרְאָה לוֹ [אֶת הַמָקוֹם שֶׁאַבְשָׁלוֹם תָּלוּי בוֹ: וְכָל עוֹבֵר עַל מִצְוַת כַּבֵּד אֶת אָבִיךָ כְּאִלּוּ עוֹבֵר עַל עֲשֶׂרֶת הַדִּבְּרוֹת], לְפִיכָךְ נִדְקְרוּ בּוֹ עֲשָׂרָה רְמָחִים, שֶׁנֶּאֱמַר [שם טו] וַיָסֹבּוּ עֲשָׂרָה נְעָרִים נֹשְׂאֵי כְּלֵי יוֹאָב:(כז) שִׁשָּׁה נִדְמוּ לְאָדָם הָרִאשׁוֹן וְכֻלָּן נֶהֶרְגוּ. וְאֵלוּ הֵן: שִׁמְשׁוֹן בְּכֹחוֹ, וְנֶהֱרַג בְּכֹחוֹ. שָׁאוּל בְּקוֹמָתוֹ, וְנֶהֱרַג בְּקוֹמָתוֹ. עֲשָׂהאֵל בִּמְרוּצָתוֹ, וְנֶהֱרַג בִּמְרוּצָתוֹ.(כח) יֹאשִׁיָּהוּ בִנְחִירָיו, וְנֶהֱרַג בִּנְחִירָיו.(כט) צִדְקִיָּהוּ בְעֵינָיו, וּלְכָךְ עִוְּרוּ עֵינָיו. אַבְשָׁלוֹם בְּשַׂעְרוֹ, וְנֶהֱרַג בְּשַׂעְרוֹ. וְאַבְשָׁלוֹם(ל) גִּבּוֹר כֹּחַ הָיָה בַמִּלְחָמָה, וְהָיְתָה חַרְבּוֹ מְצֻמֶּדֶת עַל מָתְנָיו. וּמִפְּנֵי מָה לֹא שָׁלַף חַרְבּוֹ וְכָרַת שְׂעַר רֹאשׁוֹ וְיָרַד, אֶלָּא רָאָה גֵּיהִנָּם שֶׁנִפְתְּחָה תַּחְתָּיו, אָמַר(לא) מוּטָב לִי לְהִתָּלוֹת בִּשְׂעַר רֹאשִׁי(לב) וְלֹא לֵירֵד בָּאֵשׁ, לְפִיכָךְ הָיָה תָּלוּי בָּאֵלָה [שֶׁנֶּאֱמַר [ש"ב יח, י] הִנֵּה רָאִיתִי אֶת אַבְשָׁלוֹם תָּלוּי בָּאֵלָה]. רַבִּי יוֹסֵי אוֹמֵר, שִׁבְעָה פְּתָחִים לְגֵיהִנָּם, וְנִכְנַס אַבְשָׁלוֹם עַד חֲמִשָּׁה פְּתָחִים שֶׁל גֵּיהִנָּם, וְשָׁמַע דָּוִד וְהִתְחִיל מְיַלֵּל וּמְסַפֵּד, וְקָרָא עַל אַבְשָׁלוֹם(לג) חֲמִשָּׁה [פְּעָמִים] בְּנִי בְּנִי, שֶׁנֶּאֱמַר [ש"ב יט, א] וַיִרְגַּז הַמֶּלֶךְ וַיַּעַל עַל עֲלִיַּת הַשַׁעַר וְגוֹ' בְּנִי אַבְשָׁלוֹם בְּנִי בְּנִי וְגוֹ'. וְהֶחֱזִירוֹ מֵחֲמִשָּׁה פְּתָחִים, [וְרָאָה דָוִד שֶׁנִיצַל מֵחֲמִשָּׁה פְּתָחִים], וְהִתְחִיל מְהַלֵּל וּמְשַׁבֵּחַ לְיוֹצְרוֹ וְאָמַר עֲשֵׂה עִמִּי אוֹת לְטוֹבָה וְיִרְאוּ שׂנְאַי וְיֵבֹשׁוּ כִּי אַתָּה ה' עֲזַרְתַּנִי וְנִחַמְתָּנִי [תהלים פו, יז]. עֲזַרְתַּנִי מִמִּלְחֶמֶת אַבְשָׁלוֹם,(לד) וְנִחַמְתָּנִי עַל אֶבְלִי.
